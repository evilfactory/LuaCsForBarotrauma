294  public void CheckIfNeedsUpdate()
295  {
296  if (!IsOn)
297  {
298  base.IsActive = false;
299  return;
300  }
301 
302  if ((item.body == null || !item.body.Enabled) &&
303  powerConsumption <= 0.0f && Parent == null && turret == null &&
304  (statusEffectLists == null || !statusEffectLists.ContainsKey(ActionType.OnActive)) &&
305  (IsActiveConditionals == null || IsActiveConditionals.Count == 0))
306  {
307  if (item.body == null || item.body.Enabled ||
308  (item.ParentInventory is ItemInventory itemInventory && !itemInventory.Container.HideItems))
309  {
310  lightBrightness = 1.0f;
311  SetLightSourceState(true, lightBrightness);
312  }
313  else
314  {
315  lightBrightness = 0.0f;
316  SetLightSourceState(false, 0.0f);
317  }
318  isOn = true;
319  SetLightSourceTransformProjSpecific();
320  base.IsActive = false;
321 #if CLIENT
322  Light.ParentSub = item.Submarine;
323 #endif
324  }

331  public override void Update(float deltaTime, Camera cam)
332  {
333  if (item.AiTarget != null)
334  {
335  UpdateAITarget(item.AiTarget);
336  }
337  UpdateOnActiveEffects(deltaTime);
338  //something in UpdateOnActiveEffects may deactivate the light -> return so we don't turn it back on
339  if (!IsActive) { return; }
340 
341 #if CLIENT
342  Light.ParentSub = item.Submarine;
343 #endif
344 
345 
346  bool visibleInContainer;
347  var ownerCharacter = item.GetRootInventoryOwner() as Character;
348  if (ownerCharacter != null && item.RootContainer?.GetComponent<Holdable>() is not { IsActive: true })
349  {
350  //if the item is in a character inventory, the light should only be visible if the character is holding the item
351  //(not if it's e.q. inside a wearable item, or in a rifle worn on the back)
352  visibleInContainer = false;
353  }
354  else
355  {
356  visibleInContainer = item.FindParentInventory(static it => it is ItemInventory { Container.HideItems: true }) == null;
357  }
358 
359  if ((item.Container != null && !visibleInContainer && ownerCharacter == null) ||
360  (ownerCharacter != null && ownerCharacter.InvisibleTimer > 0.0f))
361  {
362  lightBrightness = 0.0f;
363  SetLightSourceState(false, 0.0f);
364  return;
365  }
366  SetLightSourceTransformProjSpecific();
367 
368  PhysicsBody body = ParentBody ?? item.body;
369  if (body != null && !body.Enabled && !visibleInContainer)
370  {
371  lightBrightness = 0.0f;
372  SetLightSourceState(false, 0.0f);
373  return;
374  }
375 
376  TemporaryFlickerTimer -= deltaTime;
377 
378  //currPowerConsumption = powerConsumption;
379  if (Rand.Range(0.0f, 1.0f) < 0.05f && (Voltage < Rand.Range(0.0f, MinVoltage) || TemporaryFlickerTimer > 0.0f))
380  {
381 #if CLIENT
382  if (Voltage > 0.1f)
383  {
384  SoundPlayer.PlaySound("zap", item.WorldPosition, hullGuess: item.CurrentHull);
385  }
386 #endif
387  lightBrightness = 0.0f;
388  }
389  else
390  {
391  lightBrightness = MathHelper.Lerp(lightBrightness, powerConsumption <= 0.0f ? 1.0f : Math.Min(Voltage, 1.0f), 0.1f);
392  }
393 
394  SetLightSourceState(true, lightBrightness);
395  }

409  public override void ReceiveSignal(Signal signal, Connection connection)
410  {
411  switch (connection.Name)
412  {
413  case "toggle":
414  if (signal.value != "0")
415  {
416  if (!IgnoreContinuousToggle || lastToggleSignalTime < Timing.TotalTime - 0.1)
417  {
418  IsOn = !IsOn;
419  }
420  lastToggleSignalTime = Timing.TotalTime;
421  }
422  break;
423  case "set_state":
424  IsOn = signal.value != "0";
425  break;
426  case "set_color":
427  if (signal.value != prevColorSignal)
428  {
429  LightColor = XMLExtensions.ParseColor(signal.value, false);
430 #if CLIENT
431  SetLightSourceState(Light.Enabled, lightColorMultiplier);
432 #endif
433  prevColorSignal = signal.value;
434  }
435  break;
436  }
437  }
