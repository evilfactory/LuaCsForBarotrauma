115  public void ServerEventRead(IReadMessage msg, Client c)
116  {
117  var header = (CircuitBoxOpcode)msg.ReadByte();
118  switch (header)
119  {
120  case CircuitBoxOpcode.AddComponent:
121  {
122  var data = INetSerializableStruct.Read<CircuitBoxAddComponentEvent>(msg);
123  if (!CanAccessAndUnlocked(c)) { break; }
124 
125  var prefab = ItemPrefab.Prefabs.Find(p => p.UintIdentifier == data.PrefabIdentifier);
126  if (prefab is null)
127  {
128  ThrowError("Unable to add component because the prefab was not found.", c);
129  return;
130  }
131 
132  if (IsFull || !GetApplicableResourcePlayerHas(prefab, c.Character).TryUnwrap(out var resource)) { return; }
133 
134  ushort id = ICircuitBoxIdentifiable.FindFreeID(Components);
135  if (id is ICircuitBoxIdentifiable.NullComponentID)
136  {
137  ThrowError("Unable to add component because there are no available IDs left.", c);
138  return;
139  }
140 
141  bool result = AddComponentInternal(id, prefab, resource.Prefab, data.Position, c.Character, it =>
142  {
143  CreateServerEvent(new CircuitBoxServerCreateComponentEvent(it.ID, resource.Prefab.UintIdentifier, id, data.Position));
144  });
145 
146  if (!result)
147  {
148  ThrowError("Unable to add component because the component could not be created.", c);
149  return;
150  }
151 
152  GameServer.Log($"{NetworkMember.ClientLogName(c)} added a {prefab.Name} into a circuit box.", ServerLog.MessageType.Wiring);
153  RemoveItem(resource);
154  break;
155  }
156  case CircuitBoxOpcode.MoveComponent:
157  {
158  var data = INetSerializableStruct.Read<CircuitBoxMoveComponentEvent>(msg);
159  if (!item.CanClientAccess(c)) { break; }
160 
161  MoveNodesInternal(data.TargetIDs, data.IOs, data.LabelIDs, data.MoveAmount);
162  CreateServerEvent(data);
163  break;
164  }
165  case CircuitBoxOpcode.DeleteComponent:
166  {
167  var data = INetSerializableStruct.Read<CircuitBoxRemoveComponentEvent>(msg);
168  if (!data.TargetIDs.Any() || !CanAccessAndUnlocked(c)) { break; }
169 
170  CreateRefundItemsForUsedResources(data.TargetIDs, c.Character);
171  GameServer.Log($"{NetworkMember.ClientLogName(c)} removed {GetLogComponentName(data.TargetIDs)} from circuit box.", ServerLog.MessageType.Wiring);
172  RemoveComponentInternal(data.TargetIDs);
173  CreateServerEvent(data);
174  break;
175  }
176  case CircuitBoxOpcode.SelectComponents:
177  {
178  var data = INetSerializableStruct.Read<CircuitBoxSelectNodesEvent>(msg);
179  if (!item.CanClientAccess(c)) { break; }
180 
181  SelectComponentsInternal(data.TargetIDs, c.CharacterID, data.Overwrite);
182  SelectInputOutputInternal(data.IOs, c.CharacterID, data.Overwrite);
183  SelectLabelsInternal(data.LabelIDs, c.CharacterID, data.Overwrite);
184  BroadcastSelectionStatus();
185  break;
186  }
187  case CircuitBoxOpcode.SelectWires:
188  {
189  var data = INetSerializableStruct.Read<CircuitBoxSelectWiresEvent>(msg);
190  if (!item.CanClientAccess(c)) { break; }
191 
192  SelectWiresInternal(data.TargetIDs, c.CharacterID, data.Overwrite);
193  BroadcastSelectionStatus();
194  break;
195  }
196  case CircuitBoxOpcode.AddWire:
197  {
198  var data = INetSerializableStruct.Read<CircuitBoxClientAddWireEvent>(msg);
199  if (!CanAccessAndUnlocked(c)) { break; }
200 
201  var prefab = ItemPrefab.Prefabs.Find(p => p.UintIdentifier == data.SelectedWirePrefabIdentifier);
202  if (prefab is null)
203  {
204  ThrowError($"Unable to connect wire because wire by identifier \"{data.SelectedWirePrefabIdentifier}\" was not found.", c);
205  break;
206  }
207 
208  if (data.Start.FindConnection(this).TryUnwrap(out var start) &&
209  data.End.FindConnection(this).TryUnwrap(out var end))
210  {
211  bool result = Connect(start, end, wire =>
212  {
213  CreateServerEvent(new CircuitBoxServerCreateWireEvent(data with { Start = wire.Start, End = wire.End }, wire.ID, wire.Item.Select(static i => i.ID)));
214  }, prefab);
215 
216  if (!result)
217  {
218  ThrowError("Unable to connect wire because the circuit box rejected it.", c);
219  }
220 
221  GameServer.Log($"{NetworkMember.ClientLogName(c)} connected a wire from {start.Name} to {end.Name} in a circuit box.", ServerLog.MessageType.Wiring);
222  }
223  else
224  {
225  ThrowError($"Unable to connect wire because the start or end connection was not found. (start: {data.Start}, end: {data.End})", c);
226  }
227 
228  break;
229  }
230  case CircuitBoxOpcode.RemoveWire:
231  {
232  var data = INetSerializableStruct.Read<CircuitBoxRemoveWireEvent>(msg);
233  if (!data.TargetIDs.Any() || !CanAccessAndUnlocked(c)) { break; }
234 
235  GameServer.Log($"{NetworkMember.ClientLogName(c)} removed {GetLogWireName(data.TargetIDs)} from circuit box.", ServerLog.MessageType.Wiring);
236  RemoveWireInternal(data.TargetIDs);
237  CreateServerEvent(data);
238  break;
239  }
240  case CircuitBoxOpcode.RenameLabel:
241  {
242  var data = INetSerializableStruct.Read<CircuitBoxRenameLabelEvent>(msg);
243  if (!CanAccessAndUnlocked(c)) { break; }
244 
245  RenameLabelInternal(data.LabelId, data.Color, data.NewHeader, data.NewBody);
246  CreateServerEvent(data);
247  break;
248  }
249  case CircuitBoxOpcode.AddLabel:
250  {
251  var data = INetSerializableStruct.Read<CircuitBoxAddLabelEvent>(msg);
252  if (!CanAccessAndUnlocked(c)) { break; }
253 
254  ushort id = ICircuitBoxIdentifiable.FindFreeID(Labels);
255  if (id is ICircuitBoxIdentifiable.NullComponentID)
256  {
257  ThrowError("Unable to add label because there are no available IDs left.", c);
258  return;
259  }
260 
261  AddLabelInternal(id, data.Color, data.Position, data.Header, data.Body);
262  CreateServerEvent(new CircuitBoxServerAddLabelEvent(id, data.Position, new Vector2(256), data.Color, data.Header, data.Body));
263  break;
264  }
265  case CircuitBoxOpcode.RemoveLabel:
266  {
267  var data = INetSerializableStruct.Read<CircuitBoxRemoveLabelEvent>(msg);
268  if (!CanAccessAndUnlocked(c)) { break; }
269 
270  RemoveLabelInternal(data.TargetIDs);
271  CreateServerEvent(data);
272  break;
273  }
274  case CircuitBoxOpcode.ResizeLabel:
275  {
276  var data = INetSerializableStruct.Read<CircuitBoxResizeLabelEvent>(msg);
277  if (!CanAccessAndUnlocked(c)) { break; }
278 
279  ResizeLabelInternal(data.ID, data.Position, data.Size);
280  CreateServerEvent(data with { Size = Vector2.Max(data.Size, CircuitBoxLabelNode.MinSize) });
281  break;
282  }
283  default:
284  throw new ArgumentOutOfRangeException(nameof(header), header, "This opcode cannot be handled using entity events");
285  }
