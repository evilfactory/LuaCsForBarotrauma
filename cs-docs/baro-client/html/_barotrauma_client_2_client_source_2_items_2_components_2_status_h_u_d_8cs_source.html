221  public static void DrawThermalOverlay(SpriteBatch spriteBatch, Entity refEntity, Character user, Color overlayColor, float range, float effectState, bool showDeadCharacters)
222  {
223  spriteBatch.End();
224  float colorIntensityBase = 0.5f; //Multiplies the overlay color by this amount, the higher the value, the more bright/vibrant the color.
225  float colorIntensityVariance = 0.05f; //The variance of the pulse effect affecting the color's brightness/vibrance
226  GameMain.LightManager.SolidColorEffect.Parameters["color"].SetValue(overlayColor.ToVector4() * (colorIntensityBase + MathF.Sin(effectState) * colorIntensityVariance));
227  GameMain.LightManager.SolidColorEffect.CurrentTechnique = GameMain.LightManager.SolidColorEffect.Techniques["SolidColorBlur"];
228  GameMain.LightManager.SolidColorEffect.Parameters["blurDistance"].SetValue(0.01f + MathF.Sin(effectState) * 0.005f);
229  GameMain.LightManager.SolidColorEffect.CurrentTechnique.Passes[0].Apply();
230  spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.Additive, transformMatrix: Screen.Selected.Cam.Transform, effect: GameMain.LightManager.SolidColorEffect);
231 
232  foreach (Character c in Character.CharacterList)
233  {
234  if (c == user || !c.Enabled || c.Removed || c.Params.HideInThermalGoggles) { continue; }
235  if (!showDeadCharacters && c.IsDead) { continue; }
236 
237  float dist = Vector2.DistanceSquared(refEntity.WorldPosition, c.WorldPosition);
238  if (dist > range * range) { continue; }
239 
240  Sprite pingCircle = GUIStyle.UIThermalGlow.Value.Sprite;
241  foreach (Limb limb in c.AnimController.Limbs)
242  {
243  if (limb.Mass < 0.5f && limb != c.AnimController.MainLimb) { continue; }
244  float noise1 = PerlinNoise.GetPerlin((effectState + limb.Params.ID + c.ID) * 0.01f, (effectState + limb.Params.ID + c.ID) * 0.02f);
245  float noise2 = PerlinNoise.GetPerlin((effectState + limb.Params.ID + c.ID) * 0.01f, (effectState + limb.Params.ID + c.ID) * 0.008f);
246  Vector2 spriteScale = ConvertUnits.ToDisplayUnits(limb.body.GetSize()) / pingCircle.size * (noise1 * 0.5f + 2f);
247  Vector2 drawPos = new Vector2(limb.body.DrawPosition.X + (noise1 - 0.5f) * 100, -limb.body.DrawPosition.Y + (noise2 - 0.5f) * 100);
248  pingCircle.Draw(spriteBatch, drawPos, 0.0f, scale: Math.Max(spriteScale.X, spriteScale.Y));
249  }
250  }
251 
252  spriteBatch.End();
253  spriteBatch.Begin(SpriteSortMode.Deferred, BlendState.NonPremultiplied);
254  }

256  private void DrawCharacterInfo(SpriteBatch spriteBatch, Character target, float alpha = 1.0f)
257  {

259  hudPos += Vector2.UnitX * 50.0f;
260 
261  List<LocalizedString> texts = new List<LocalizedString>();
262  List<Color> textColors = new List<Color>();
263  texts.Add(target.Info == null ? target.DisplayName : target.Info.DisplayName);
264  Color nameColor = GUIStyle.TextColorNormal;
265  if (Character.Controlled != null && target.TeamID != Character.Controlled.TeamID)
266  {
267  nameColor = target.TeamID == CharacterTeamType.FriendlyNPC ? Color.SkyBlue : GUIStyle.Red;
268  }
269  textColors.Add(nameColor);
270 
271  if (target.IsDead)

273  texts.Add(TextManager.Get("Deceased"));
274  textColors.Add(GUIStyle.Red);
275  if (target.CauseOfDeath != null)
276  {
277  texts.Add(
278  target.CauseOfDeath.Affliction?.CauseOfDeathDescription ??
279  TextManager.AddPunctuation(':', TextManager.Get("CauseOfDeath"), TextManager.Get("CauseOfDeath." + target.CauseOfDeath.Type.ToString())));
280  textColors.Add(GUIStyle.Red);
281  }
282  }
283  else
284  {
285  if (!target.CustomInteractHUDText.IsNullOrEmpty() && target.AllowCustomInteract)
286  {
287  texts.Add(target.CustomInteractHUDText);
288  textColors.Add(GUIStyle.Green);
289  }
290  if (!target.IsIncapacitated && target.IsPet)
291  {
292  texts.Add(CharacterHUD.GetCachedHudText("PlayHint", InputType.Use));
293  textColors.Add(GUIStyle.Green);
294  }
295  if (equipper?.FocusedCharacter == target && target.CanBeHealedBy(equipper, checkFriendlyTeam: false))
296  {
297  texts.Add(CharacterHUD.GetCachedHudText("HealHint", InputType.Health));
298  textColors.Add(GUIStyle.Green);
299  }
300  if (target.CanBeDraggedBy(Character.Controlled))
301  {
302  texts.Add(CharacterHUD.GetCachedHudText("GrabHint", InputType.Grab));
303  textColors.Add(GUIStyle.Green);
304  }
305 
306  if (target.IsUnconscious)
307  {
308  texts.Add(TextManager.Get("Unconscious"));
309  textColors.Add(GUIStyle.Orange);
310  }
311  if (target.Stun > 0.01f)
312  {
313  texts.Add(TextManager.Get("Stunned"));
314  textColors.Add(GUIStyle.Orange);
315  }
316 
317  int oxygenTextIndex = MathHelper.Clamp((int)Math.Floor((1.0f - (target.Oxygen / 100.0f)) * OxygenTexts.Length), 0, OxygenTexts.Length - 1);
318  texts.Add(OxygenTexts[oxygenTextIndex]);
319  textColors.Add(Color.Lerp(GUIStyle.Red, GUIStyle.Green, target.Oxygen / 100.0f));
320 
321  if (target.Bleeding > 0.0f)
322  {
323  int bleedingTextIndex = MathHelper.Clamp((int)Math.Floor(target.Bleeding / 100.0f * BleedingTexts.Length), 0, BleedingTexts.Length - 1);
324  texts.Add(BleedingTexts[bleedingTextIndex]);
325  textColors.Add(Color.Lerp(GUIStyle.Orange, GUIStyle.Red, target.Bleeding / 100.0f));
326  }
327 
328  var allAfflictions = target.CharacterHealth.GetAllAfflictions();
329  Dictionary<AfflictionPrefab, float> combinedAfflictionStrengths = new Dictionary<AfflictionPrefab, float>();
330  foreach (Affliction affliction in allAfflictions)
331  {
332  if (affliction.Strength <= 0f) { continue; }
333  if (affliction.Strength < affliction.Prefab.ShowInHealthScannerThreshold)
334  {
335  if (target.IsHuman || target.IsOnPlayerTeam || (affliction.Prefab.AfflictionType != AfflictionPrefab.PoisonType && affliction.Prefab.AfflictionType != AfflictionPrefab.ParalysisType))
336  {
337  // Always show the poisons on monsters, because poisoning bigger monsters require multiple doses.
338  // The solution is hacky, but didn't want to introduce an extra property for this.
339  // We also want to have a relatively high thershold for showing the poisons on the scanner on humans, so that it's not instantly clear that a target is poisoned and especially not which poison was used.
340  // Paralysis is treated like a poison but isn't technically a poison, so that we can have multiple afflictions that still are treated the same.
341  continue;
342  }
343  }
344  if (combinedAfflictionStrengths.ContainsKey(affliction.Prefab))
345  {
346  combinedAfflictionStrengths[affliction.Prefab] += affliction.Strength;
347  }
348  else
349  {
350  combinedAfflictionStrengths[affliction.Prefab] = affliction.Strength;
351  }
352  }
353 
354  foreach (AfflictionPrefab affliction in combinedAfflictionStrengths.Keys)
355  {
356  texts.Add(TextManager.AddPunctuation(':', affliction.Name, Math.Max((int)combinedAfflictionStrengths[affliction], 1).ToString() + " %"));
357  textColors.Add(Color.Lerp(GUIStyle.Orange, GUIStyle.Red, combinedAfflictionStrengths[affliction] / affliction.MaxStrength));
358  }
359  }
360 
361  GUI.DrawString(spriteBatch, hudPos, texts[0].Value, textColors[0] * alpha, Color.Black * 0.7f * alpha, 2, GUIStyle.SubHeadingFont, ForceUpperCase.No);
362  hudPos.X += 5.0f * GUI.Scale;
363  hudPos.Y += GUIStyle.SubHeadingFont.MeasureString(texts[0].Value).Y;
364 
365  hudPos.X = (int)hudPos.X;
366  hudPos.Y = (int)hudPos.Y;
367 
368  for (int i = 1; i < texts.Count; i++)
369  {
370  GUI.DrawString(spriteBatch, hudPos, texts[i], textColors[i] * alpha, Color.Black * 0.7f * alpha, 2, GUIStyle.SmallFont);
371  hudPos.Y += (int)(GUIStyle.SubHeadingFont.MeasureString(texts[i].Value).Y);
372  }
373  }
