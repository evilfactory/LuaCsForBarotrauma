119  public void ClientAdminRead(IReadMessage incMsg)
120  {
121  bool hasPermission = incMsg.ReadBoolean();
122  if (!hasPermission)
123  {
124  incMsg.ReadPadBits();
125  return;
126  }
127 
128  bool isOwner = incMsg.ReadBoolean();
129  incMsg.ReadPadBits();
130 
131  bannedPlayers.Clear();
132  UInt32 bannedPlayerCount = incMsg.ReadVariableUInt32();
133 
134  for (int i = 0; i < (int)bannedPlayerCount; i++)
135  {
136  string name = incMsg.ReadString();
137  UInt32 uniqueIdentifier = incMsg.ReadUInt32();
138  bool includesExpiration = incMsg.ReadBoolean();
139  incMsg.ReadPadBits();
140 
141  DateTime? expiration = null;
142  if (includesExpiration)
143  {
144  double hoursFromNow = incMsg.ReadDouble();
145  expiration = DateTime.Now + TimeSpan.FromHours(hoursFromNow);
146  }
147 
148  string reason = incMsg.ReadString();
149 
150  Either<Address, AccountId> addressOrAccountId;
151  if (isOwner)
152  {
153  bool isAddress = incMsg.ReadBoolean();
154  incMsg.ReadPadBits();
155  string str = incMsg.ReadString();
156  if (isAddress && Address.Parse(str).TryUnwrap(out var address))
157  {
158  addressOrAccountId = address;
159  }
160  else if (AccountId.Parse(str).TryUnwrap(out var accountId))
161  {
162  addressOrAccountId = accountId;
163  }
164  else
165  {
166  continue;
167  }
168  }
169  else
170  {
171  addressOrAccountId = new UnknownAddress();
172  }
173  bannedPlayers.Add(new BannedPlayer(uniqueIdentifier, name, addressOrAccountId, reason, expiration));
174  }
175 
176  if (BanFrame != null)
177  {
178  var parent = BanFrame.Parent;
179  parent.RemoveChild(BanFrame);
180  CreateBanFrame(parent);
181  }
182  }
