121  private bool TryCreateTraitorEvents(EventManager eventManager, Level level)
122  {
123  var eventPrefabs = EventPrefab.Prefabs.Where(p => p is TraitorEventPrefab).Cast<TraitorEventPrefab>();
124  if (!eventPrefabs.Any())
125  {
126  DebugConsole.AddWarning("No traitor event available in any of the enabled content packages.");
127  return false;
128  }
129 
130  if (server.ConnectedClients.Count(IsClientViableTraitor) < server.ServerSettings.TraitorsMinPlayerCount)
131  {
132  DebugConsole.AddWarning("Not enough clients to create a traitor event. Active traitor events: " + activeEvents.Count);
133 #if DEBUG
134  DebugConsole.AddWarning("Starting a traitor event anyway because this is a debug build.");
135 #else
136  return false;
137 #endif
138 
139  }
140 
141  int maxDangerLevel = server.ServerSettings.TraitorDangerLevel;
142 
143  int playerCount = server.ConnectedClients.Count(c => c.Character != null && !c.Character.Removed);
144 
145  var campaign = GameMain.GameSession?.Campaign;
146  var suitablePrefabs = eventPrefabs
147  .Where(e => EventManager.IsLevelSuitable(e, level))
148  .Where(e => e.ReputationRequirementsMet(campaign))
149  .Where(e => e.MissionRequirementsMet(GameMain.GameSession))
150  .Where(e => e.LevelRequirementsMet(level))
151  .Where(e => e.DangerLevel <= maxDangerLevel)
152  .Where(e => playerCount >= e.MinPlayerCount)
153  .ToList();
154 
155  if (!suitablePrefabs.Any())
156  {
157  //this is normal, there e.g. might be no missions for an abandoned outpost or end level
158  DebugConsole.Log("No suitable traitor missions available for the level.");
159  return false;
160  }
161 
162  foreach (var previousEvent in previousTraitorEvents.Reverse<PreviousTraitorEvent>().DistinctBy(e => e.Traitor))
163  {
164  if (previousEvent.State == TraitorEvent.State.Completed &&
165  previousEvent.TraitorEvent.IsChainable &&
166  IsClientViableTraitor(previousEvent.Traitor))
167  {
168  GameServer.Log($"{NetworkMember.ClientLogName(previousEvent.Traitor)} successfully completed a traitor event ({previousEvent.TraitorEvent.Identifier}) on a previous round. Attempting to give choose them a new, more dangerous event...", ServerLog.MessageType.Traitors);
169 
170  var suitablePrefab =
171  //try finding an event that's continuation from the previous one (= requires the previous one to be completed 1st)
172  suitablePrefabs.GetRandomUnsynced(p => p.RequiredCompletedTags.Any(t =>
173  previousEvent.TraitorEvent.Identifier == t || previousEvent.TraitorEvent.Tags.Contains(t)));
174 
175  suitablePrefab ??=
176  //otherwise try finding some higher-difficult event for the same faction
177  suitablePrefabs.GetRandomUnsynced(p =>
178  p.RequiredCompletedTags.None() &&
179  p.DangerLevel > previousEvent.TraitorEvent.DangerLevel &&
180  p.Faction == previousEvent.TraitorEvent.Faction);
181 
182  if (suitablePrefab != null)
183  {
184  CreateTraitorEvent(eventManager, suitablePrefab, previousEvent.Traitor);
185  return true;
186  }
187  else
188  {
189  GameServer.Log($"Could not find a suitable, more difficult traitor event for {NetworkMember.ClientLogName(previousEvent.Traitor)}.", ServerLog.MessageType.Traitors);
190  }
191  }
192  }
193 
194  TraitorEventPrefab selectedPrefab;
195  if (GameMain.GameSession?.Campaign == null)
196  {
197  selectedPrefab = suitablePrefabs.GetRandomByWeight(GetTraitorEventPrefabCommonness, Rand.RandSync.Unsynced);
198  }
199  else
200  {
201  //events that are suitable as initial traitor events (not requiring some other event to be completed first)
202  var suitableInitialPrefabs = suitablePrefabs.FindAll(e => e.RequiredCompletedTags.None() && IsSuitableDangerLevel(e));
203 
204  bool IsSuitableDangerLevel(TraitorEventPrefab prefab)
205  {
206  if (prefab.DangerLevel == TraitorEventPrefab.MinDangerLevel) { return true; }
207  //events that require another event to be completed are handled earlier in the method
208  if (prefab.RequirePreviousDangerLevelCompleted) { return false; }
209  return
210  prefab.RequiredPreviousDangerLevel < TraitorEventPrefab.MinDangerLevel ||
211  previousTraitorEvents.Any(e2 => e2.TraitorEvent.DangerLevel >= prefab.RequiredPreviousDangerLevel);
212  }
213 
214  selectedPrefab = suitableInitialPrefabs.GetRandomByWeight(GetTraitorEventPrefabCommonness, Rand.RandSync.Unsynced);
215  if (selectedPrefab == null)
216  {
217  GameServer.Log($"Could not find a suitable danger level {TraitorEventPrefab.MinDangerLevel} traitor event. Choosing a random event instead.", ServerLog.MessageType.Traitors);
218  selectedPrefab = suitablePrefabs.GetRandomByWeight(GetTraitorEventPrefabCommonness, Rand.RandSync.Unsynced);
219  }
220  }
221 
222  if (selectedPrefab != null)
223  {
224  var selectedTraitor = SelectRandomTraitor();
225  if (selectedTraitor == null)
226  {
227  DebugConsole.ThrowError($"Could not find a suitable traitor for the event \"{selectedPrefab.Identifier}\".",
228  contentPackage: selectedPrefab.ContentPackage);
229  return false;
230  }
231  CreateTraitorEvent(eventManager, selectedPrefab, selectedTraitor);
232  return true;
233  }
234 
235  return false;
236  }

250  private IEnumerable<Client> SelectSecondaryTraitors(TraitorEvent traitorEvent, Client mainTraitor)
251  {
252  if (traitorEvent.Prefab.SecondaryTraitorPercentage <= 0.0f &&
253  traitorEvent.Prefab.SecondaryTraitorAmount <= 0)
254  {
255  return Enumerable.Empty<Client>();
256  }
257 
258  var viableTraitors = server.ConnectedClients.Where(c => c != mainTraitor && IsClientViableTraitor(c)).ToList();
259 
260  int amountToChoose = (int)Math.Ceiling(viableTraitors.Count * (traitorEvent.Prefab.SecondaryTraitorPercentage / 100.0f));
261  amountToChoose = Math.Max(amountToChoose, traitorEvent.Prefab.SecondaryTraitorAmount);
262 
263  if (amountToChoose > viableTraitors.Count)
264  {
265  DebugConsole.ThrowError(
266  $"Error in traitor event {traitorEvent.Prefab.Identifier}. Not enough players to choose {amountToChoose} secondary traitors. " +
267  $"Make sure the {nameof(traitorEvent.Prefab.MinPlayerCount)} of the event is high enough to support to desired amount of secondary traitors.",
268  contentPackage: traitorEvent.Prefab.ContentPackage);
269  amountToChoose = viableTraitors.Count;
270  }
271 
272  List<Client> traitors = new List<Client>();
273  for (int i = 0; i < amountToChoose; i++)
274  {
275  var traitor = viableTraitors.GetRandomUnsynced();
276  if (traitor != null)
277  {
278  viableTraitors.Remove(traitor);
279  traitors.Add(traitor);
280  }
281  }
282  return traitors;
283  }

295  private float GetTraitorEventPrefabCommonness(TraitorEventPrefab prefab)
296  {
297  int? roundsSinceLastSelected = GetRoundsSinceLastSelected(e => e.TraitorEvent == prefab);
298  if (roundsSinceLastSelected.HasValue)
299  {
300  float normalizedRoundsSinceLastSelected = MathUtils.InverseLerp(0, MaxPreviousEventHistory, roundsSinceLastSelected.Value);
301  //exponentially decreasing commonness the closer the last time the event was selected
302  return prefab.Commonness * normalizedRoundsSinceLastSelected * normalizedRoundsSinceLastSelected;
303  }
304  else
305  {
306  return prefab.Commonness;
307  }
308  }
309 
310  private float GetTraitorProbability(Client client)
311  {
312  int? roundsSinceLastSelected = GetRoundsSinceLastSelected(e => e.Traitor == client);
313  if (roundsSinceLastSelected.HasValue)
314  {
315  float normalizedRoundsSinceLastSelected = MathUtils.InverseLerp(0, MaxPreviousEventHistory, roundsSinceLastSelected.Value);
316  //exponentially decreasing commonness the closer the last time the event was selected
317  return normalizedRoundsSinceLastSelected * normalizedRoundsSinceLastSelected;
318  }
319  else
320  {
321  return 1.0f;
322  }
323  }

325  private int? GetRoundsSinceLastSelected(Func<PreviousTraitorEvent, bool> condition)
326  {
327  //most recent events are at the end of the list, start from there
328  for (int i = previousTraitorEvents.Count - 1; i >= 0; i--)
329  {
330  if (condition(previousTraitorEvents[i]))
331  {
332  return previousTraitorEvents.Count - i;
333  }
334  }
335  return null;

338  private void CreateTraitorEvent(EventManager eventManager, TraitorEventPrefab selectedPrefab, Client traitor)
339  {
340  if (selectedPrefab.TryCreateInstance<TraitorEvent>(eventManager.RandomSeed, out var newEvent))
341  {
342  var secondaryTraitors = SelectSecondaryTraitors(newEvent, traitor);
343 
344  string logMessage = $"{NetworkMember.ClientLogName(traitor)} was selected as a traitor. Selected event: {selectedPrefab.Name}";
345  if (secondaryTraitors.Any())
346  {
347  logMessage += $", secondary traitors: {string.Join(", ", secondaryTraitors.Select(c => NetworkMember.ClientLogName(c)))}";
348  }
349  GameServer.Log(logMessage, ServerLog.MessageType.Traitors);
350 
351  newEvent.OnStateChanged += () => SendCurrentState(newEvent);
352  activeEvents.Add(new ActiveTraitorEvent(traitor, newEvent));
353  newEvent.SetTraitor(traitor);
354  newEvent.SetSecondaryTraitors(secondaryTraitors);
355  eventManager.ActivateEvent(newEvent);
356  SendCurrentState(newEvent);
357  }
358  else
359  {
360  DebugConsole.ThrowError($"Failed to create an instance of the traitor event prefab \"{selectedPrefab.Identifier}\"!",
361  contentPackage: selectedPrefab.ContentPackage);
362  }

387  public void Update(float deltaTime)
388  {
389  if (!Enabled) { return; }
390  if (!started)
391  {
392  if (level?.LevelData is { Type: LevelData.LevelType.LocationConnection })
393  {
394  if (Submarine.MainSub != null && Submarine.MainSub.WorldPosition.X > level.Size.X / 2)
395  {
396  //try starting ASAP if the submarine is already half-way through the level
397  //(brief delay regardless, because otherwise we might retry every frame if finding a suitable event fails below)
398  startTimer = Math.Min(startTimer, 10.0f);
399  }
400  }
401  startTimer -= deltaTime;
402  if (startTimer >= 0.0f) { return; }
403  if (eventManager == null)
404  {
405  throw new InvalidOperationException("EventManager was null. TraitorManager may not have been initialized properly.");
406  }
407  if (level == null)
408  {
409  throw new InvalidOperationException("Level was null. TraitorManager may not have been initialized properly.");
410  }
411  if (TryCreateTraitorEvents(eventManager, level))
412  {
413  started = true;
414  }
415  else
416  {
417  //restart timer, we might be able to start a mission later if more clients join
418  startTimer = Rand.Range(StartDelayMin, StartDelayMax);
419  }
420  }

423  public void EndRound()
424  {
425  Client? votedAsTraitor = GetClientAccusedAsTraitor();
426  foreach (var activeEvent in activeEvents)
427  {
428  if (results != null)
429  {
430  DebugConsole.AddWarning("Multiple traitor events active during the round, only displaying the results for the last one.");
431  }
432  results = new TraitorResults(votedAsTraitor, activeEvent.TraitorEvent);
433  if (results.Value.MoneyPenalty > 0)
434  {
435  GameMain.GameSession?.Campaign?.Bank?.TryDeduct(results.Value.MoneyPenalty);
436  }
437 
438  if (activeEvent.TraitorEvent.CurrentState != TraitorEvent.State.Completed)
439  {
440  activeEvent.TraitorEvent.CurrentState = TraitorEvent.State.Failed;
441  }
442  GameServer.Log(
443  NetworkMember.ClientLogName(activeEvent.Traitor) +
444  (activeEvent.TraitorEvent.CurrentState == TraitorEvent.State.Completed ?
445  " completed their traitor objective successfully." : " failed to complete their traitor objective."),
446  ServerLog.MessageType.Traitors);
447 
448  //consider the event failed if the traitor was identifier correctly,
449  //so the traitor doesn't get rewards or get assigned a follow-up event
450  if (results.Value.VotedCorrectTraitor)
451  {
452  GameServer.Log(
453  NetworkMember.ClientLogName(activeEvent.Traitor) + " was correctly identified as the traitor, and will not receive any rewards.",
454  ServerLog.MessageType.Traitors);
455  activeEvent.TraitorEvent.CurrentState = TraitorEvent.State.Failed;
456  }
457  previousTraitorEvents.Add(new PreviousTraitorEvent(
458  activeEvent.TraitorEvent.Prefab,
459  activeEvent.TraitorEvent.CurrentState,
460  activeEvent.Traitor));
461 
462  if (activeEvent.TraitorEvent.CurrentState == TraitorEvent.State.Completed)
463  {
464  AchievementManager.OnTraitorWin(activeEvent.TraitorEvent.Traitor?.Character);
465  foreach (var secondaryTraitor in activeEvent.TraitorEvent.SecondaryTraitors)
466  {
467  AchievementManager.OnTraitorWin(secondaryTraitor?.Character);
468  }
469  }
470  }
471  if (previousTraitorEvents.Count > MaxPreviousEventHistory)
472  {
473  previousTraitorEvents.RemoveRange(0, previousTraitorEvents.Count - MaxPreviousEventHistory);
474  }
475  activeEvents.Clear();
476  }
