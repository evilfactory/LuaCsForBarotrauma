243  if (element.GetAttribute("leveltype") != null)
244  {
245  string levelTypeStr = element.GetAttributeString("leveltype", "");
246  if (Enum.TryParse(levelTypeStr, out LevelData.LevelType parsedLevelType))
247  {
248  LevelType = parsedLevelType;
249  }
250  else

252  DebugConsole.ThrowError($"Error in outpost generation parameters \"{Identifier}\". \"{levelTypeStr}\" is not a valid level type.", contentPackage: element.ContentPackage);
253  }

256  OutpostFilePath = element.GetAttributeContentPath(nameof(OutpostFilePath));
257 
258  var humanPrefabCollections = new List<NpcCollection>();
259  foreach (var subElement in element.Elements())
260  {
261  switch (subElement.Name.ToString().ToLowerInvariant())
262  {
263  case "modulecount":
264  moduleCounts.Add(new ModuleCount(subElement));
265  break;
266  case "npcs":
267  var newCollection = new NpcCollection();
268  foreach (var npcElement in subElement.Elements())
269  {
270  Identifier from = npcElement.GetAttributeIdentifier("from", Identifier.Empty);
271  Identifier faction = npcElement.GetAttributeIdentifier("faction", Identifier.Empty);
272  if (from != Identifier.Empty)
273  {
274  newCollection.Add(from, npcElement.GetAttributeIdentifier("identifier", Identifier.Empty), faction);
275  }
276  else
277  {
278  newCollection.Add(new HumanPrefab(npcElement, file, npcSetIdentifier: from), faction);
279  }
280  }
281  humanPrefabCollections.Add(newCollection);
282  break;
283  }
284  }
