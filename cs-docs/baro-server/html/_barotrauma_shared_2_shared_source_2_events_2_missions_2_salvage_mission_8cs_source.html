542  protected override void UpdateMissionSpecific(float deltaTime)
543  {
544  //make body dynamic when picked up
545  foreach (var target in targets)
546  {
547  var root = target.Item?.RootContainer ?? target.Item;
548  if (root == null) { continue; }
549  if (target.Item.ParentInventory != null && target.Item.body != null) { target.Item.body.FarseerBody.BodyType = BodyType.Dynamic; }
550  }
551 
552  if (IsClient) { return; }
553 
554  bool atLeastOneTargetWasRetrieved = false;
555  for (int i = 0; i < targets.Count; i++)
556  {
557  var target = targets[i];
558  if (i > 0 && !targets[i - 1].AllowContinueBeforeRetrieved && !targets[i - 1].Retrieved) { break; }
559  if (target.Item == null)
560  {
561 #if DEBUG
562  DebugConsole.ThrowError("Error in salvage mission " + Prefab.Identifier + " (item was null)",
563  contentPackage: Prefab.ContentPackage);
564 #endif
565  return;
566  }
567 
568  Entity rootInventoryOwner = target.Item.GetRootInventoryOwner();
569  Submarine parentSub = target.Item.CurrentHull?.Submarine ?? rootInventoryOwner?.Submarine;
570  bool inPlayerSub = parentSub != null && parentSub.Info.Type == SubmarineType.Player;
571  switch (target.State)
572  {
573  case Target.RetrievalState.None:
574  {
575  if (target.Interacted)
576  {
577  TrySetRetrievalState(Target.RetrievalState.Interact);
578  }
579  var root = target.Item?.RootContainer ?? target.Item;
580  if (root.ParentInventory?.Owner is Character character && character.TeamID == CharacterTeamType.Team1)
581  {
582  TrySetRetrievalState(Target.RetrievalState.PickedUp);
583  }
584  if (inPlayerSub)
585  {
586  TrySetRetrievalState(Target.RetrievalState.RetrievedToSub);
587  }
588  }
589  break;
590  case Target.RetrievalState.PickedUp:
591  case Target.RetrievalState.RetrievedToSub:
592  {
593 
594  bool inPlayerInventory = false;
595  bool playerInFriendlySub = false;
596  if (rootInventoryOwner is Character character && character.TeamID == CharacterTeamType.Team1)
597  {
598  inPlayerInventory = true;
599  if (character.Submarine != null)
600  {
601  playerInFriendlySub =
602  character.IsInFriendlySub ||
603  (character.Submarine == Level.Loaded?.StartOutpost && Level.IsLoadedFriendlyOutpost && GameMain.GameSession?.Campaign.CurrentLocation is not { IsFactionHostile: true });
604  }
605  }
606 
607  if (inPlayerSub || (inPlayerInventory && playerInFriendlySub))
608  {
609  TrySetRetrievalState(Target.RetrievalState.RetrievedToSub);
610  }
611  else
612  {
613  target.State = Target.RetrievalState.PickedUp;
614  }
615  }
616  break;
617  }

620  {
621  if (retrievalState < target.State || target.State == retrievalState) { return; }
622  bool wasRetrieved = target.Retrieved;
623  target.State = retrievalState;
624  //increment the mission state if the target became retrieved
625  if (!wasRetrieved && target.Retrieved)
626  {
627  State = Math.Max(i + 1, State);
628  atLeastOneTargetWasRetrieved = true;

630  }
631  }

634  {
635  TryShowRetrievedMessage();

637 #endif

641  }
