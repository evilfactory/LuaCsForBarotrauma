74  }

76  if (unlockLocation == null && CreateLocationIfNotFound)

81  mustAllowLocationTypeChanges: true,

88  requireCorrectFaction: false);

98  }
99  }
100  }
101 
102  if (unlockLocation != null)
103  {
104  if (!MissionIdentifier.IsEmpty)
105  {
106  unlockedMission = unlockLocation.UnlockMissionByIdentifier(MissionIdentifier);
107  }
108  else if (!MissionTag.IsEmpty)
109  {
110  unlockedMission = unlockLocation.UnlockMissionByTag(MissionTag, random);
111  }
112  if (campaign is MultiPlayerCampaign mpCampaign)

137 #endif
138  }
139  }

148  private Location FindUnlockLocation(int minDistance, bool unlockFurtherOnMap, IEnumerable<Identifier> locationTypes, bool mustAllowLocationTypeChanges, bool requireCorrectFaction = true)
149  {
150  var campaign = GameMain.GameSession.GameMode as CampaignMode;
151  if (LocationTypes.Length == 0 && minDistance <= 1)
152  {
153  return campaign.Map.CurrentLocation;
154  }
155 
156  var currentLocation = campaign.Map.CurrentLocation;
157  int distance = 0;
158  HashSet<Location> checkedLocations = new HashSet<Location>();
159  HashSet<Location> pendingLocations = new HashSet<Location>() { currentLocation };
160  do
161  {
162  List<Location> currentLocations = pendingLocations.ToList();
163  pendingLocations.Clear();
164  foreach (var location in currentLocations)
165  {
166  checkedLocations.Add(location);
167  if (IsLocationValid(currentLocation, location, unlockFurtherOnMap, distance, minDistance, locationTypes, mustAllowLocationTypeChanges, requireCorrectFaction))
168  {
169  return location;
170  }
171  else
172  {
173  foreach (LocationConnection connection in location.Connections)
174  {
175  var otherLocation = connection.OtherLocation(location);
176  if (checkedLocations.Contains(otherLocation)) { continue; }
177  pendingLocations.Add(otherLocation);
178  }
179  }
180  }
181  distance++;
182  } while (pendingLocations.Any());
183 
184  return null;
185  }

187  private bool IsLocationValid(Location currLocation, Location location, bool unlockFurtherOnMap, int distance, int minDistance, IEnumerable<Identifier> locationTypes, bool mustAllowLocationTypeChanges, bool requireCorrectFaction)
188  {
189  if (mustAllowLocationTypeChanges && location.LocationTypeChangesBlocked)
190  {
191  return false;
192  }
193  if (requireCorrectFaction && !RequiredFaction.IsEmpty)
194  {
195  if (location.Faction?.Prefab.Identifier != RequiredFaction &&
196  location.SecondaryFaction?.Prefab.Identifier != RequiredFaction)
197  {
198  return false;
199  }
200  }
201  if (!locationTypes.Contains(location.Type.Identifier) && !(location.HasOutpost() && locationTypes.Contains("AnyOutpost".ToIdentifier())))
202  {
203  return false;
204  }
205  if (distance < minDistance)
206  {
207  return false;
208  }
209  if (unlockFurtherOnMap && location.MapPosition.X < currLocation.MapPosition.X)
210  {
211  return false;
212  }
213  return true;
214  }
