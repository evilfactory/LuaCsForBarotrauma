256  recalculationCount = 0;

262

266

273  if (light.LightSprite != null)
274  {
275  float spriteRange = Math.Max(
276  light.LightSprite.size.X * light.SpriteScale.X * (0.5f + Math.Abs(light.LightSprite.RelativeOrigin.X - 0.5f)),
277  light.LightSprite.size.Y * light.SpriteScale.Y * (0.5f + Math.Abs(light.LightSprite.RelativeOrigin.Y - 0.5f)));

285 
286  int i = 0;
287  while (i < activeLights.Count && light.Priority < activeLights[i].Priority)
288  {
289  i++;
290  }
291  activeLights.Insert(i, light);

295  float lightPriority(float range, LightSource light)

301  (light.LightSourceParams.OverrideLightSpriteAlpha ?? (light.Color.A / 255.0f)) *
302  light.PriorityMultiplier;

306  activeLightsWithLightVolume.Clear();

396  SpriteEffects.None,
397  layerDepth: 0);
398  }

544  {

546  {

559  }
560  }

562

564  graphics.SetRenderTarget(HighlightMap);
565  SolidColorEffect.CurrentTechnique = SolidColorEffect.Techniques["SolidColor"];
566  SolidColorEffect.Parameters["color"].SetValue(Color.LightBlue.ToVector4());
567  SolidColorEffect.CurrentTechnique.Passes[0].Apply();

569  DeformableSprite.Effect.Parameters["solidColor"].SetValue(Color.LightBlue.ToVector4());
570  DeformableSprite.Effect.CurrentTechnique.Passes[0].Apply();

572  foreach (Entity highlighted in highlightedEntities)
573  {
574  if (highlighted is Item item)
575  {
576  if (item.IconStyle != null && (item != Character.Controlled.FocusedItem || Character.Controlled.FocusedItem == null))
577  {
578  //wait until next pass
579  }
580  else
581  {
582  item.Draw(spriteBatch, false, true);
583  }
584  }
585  else if (highlighted is Character character)
586  {
587  character.Draw(spriteBatch, cam);
588  }
589  }
590  spriteBatch.End();
591 
592  //draw items with iconstyles in the style's color
593  spriteBatch.Begin(SpriteSortMode.Immediate, BlendState.Additive, samplerState: SamplerState.LinearWrap, effect: SolidColorEffect, transformMatrix: spriteBatchTransform);
594  foreach (Entity highlighted in highlightedEntities)
595  {
596  if (highlighted is Item item)

610  Vector4 overlayColor = Color.Black.ToVector4() * MathHelper.Lerp(0.5f, 0.9f, phase);
611  SolidColorEffect.Parameters["color"].SetValue(overlayColor);
612  SolidColorEffect.CurrentTechnique = SolidColorEffect.Techniques["SolidColorBlur"];
613  SolidColorEffect.CurrentTechnique.Passes[0].Apply();
614  DeformableSprite.Effect.Parameters["solidColor"].SetValue(overlayColor);
615  DeformableSprite.Effect.CurrentTechnique.Passes[0].Apply();
616  spriteBatch.Begin(SpriteSortMode.Immediate, BlendState.NonPremultiplied, samplerState: SamplerState.LinearWrap, effect: SolidColorEffect, transformMatrix: spriteBatchTransform);
617  foreach (Entity highlighted in highlightedEntities)
618  {
619  if (highlighted is Item item)
620  {
621  SolidColorEffect.Parameters["blurDistance"].SetValue(0.02f);
622  item.Draw(spriteBatch, false, true);
623  }
624  else if (highlighted is Character character)
625  {
626  SolidColorEffect.Parameters["blurDistance"].SetValue(0.05f);
627  character.Draw(spriteBatch, cam);
628  }
629  }

633  spriteBatch.Begin(blendState: BlendState.NonPremultiplied, samplerState: SamplerState.LinearWrap);

667  public void UpdateObstructVision(GraphicsDevice graphics, SpriteBatch spriteBatch, Camera cam, Vector2 lookAtPosition)
668  {
669  if ((!LosEnabled || LosMode == LosMode.None) && !ObstructVision) { return; }
670  if (ViewTarget == null) return;
671 
672  graphics.SetRenderTarget(LosTexture);
673 
674  if (ObstructVision)
675  {
676  graphics.Clear(Color.Black);
677  Vector2 diff = lookAtPosition - ViewTarget.WorldPosition;
678  diff.Y = -diff.Y;
679  if (diff.LengthSquared() > 20.0f * 20.0f) { losOffset = diff; }
680  float rotation = MathUtils.VectorToAngle(losOffset);
681 
682  Vector2 scale = new Vector2(
683  MathHelper.Clamp(losOffset.Length() / 256.0f, 4.0f, 5.0f), 3.0f);
684 
685  spriteBatch.Begin(SpriteSortMode.Deferred, transformMatrix: cam.Transform * Matrix.CreateScale(new Vector3(GameSettings.CurrentConfig.Graphics.LightMapScale, GameSettings.CurrentConfig.Graphics.LightMapScale, 1.0f)));
686  spriteBatch.Draw(visionCircle, new Vector2(ViewTarget.WorldPosition.X, -ViewTarget.WorldPosition.Y), null, Color.White, rotation,
687  new Vector2(visionCircle.Width * 0.2f, visionCircle.Height / 2), scale, SpriteEffects.None, 0.0f);
688  spriteBatch.End();
689  }
690  else
691  {
692  graphics.Clear(Color.White);
693  }
694 
695 
696  //--------------------------------------
697 
698  if (LosEnabled && LosMode != LosMode.None && ViewTarget != null)
699  {
700  Vector2 pos = ViewTarget.DrawPosition;
701  bool centeredOnHead = false;
702  if (ViewTarget is Character character &&
703  character.AnimController?.GetLimb(LimbType.Head) is Limb head &&
704  !head.IsSevered && !head.Removed)
705  {
706  pos = head.body.DrawPosition;
707  centeredOnHead = true;
708  }
709 
710  Rectangle camView = new Rectangle(cam.WorldView.X, cam.WorldView.Y - cam.WorldView.Height, cam.WorldView.Width, cam.WorldView.Height);
711  Matrix shadowTransform = cam.ShaderTransform
712  * Matrix.CreateOrthographic(GameMain.GraphicsWidth, GameMain.GraphicsHeight, -1, 1) * 0.5f;
713 
714  var convexHulls = ConvexHull.GetHullsInRange(ViewTarget.Position, cam.WorldView.Width * 0.75f, ViewTarget.Submarine);
715 
716  //make sure the head isn't peeking through any LOS segments, and if it is,
717  //center the LOS on the character's collider instead
718  if (centeredOnHead)
719  {
720  foreach (var ch in convexHulls)
721  {
722  if (!ch.Enabled) { continue; }
723  Vector2 currentViewPos = pos;
724  Vector2 defaultViewPos = ViewTarget.DrawPosition;
725  if (ch.ParentEntity?.Submarine != null)
726  {
727  defaultViewPos -= ch.ParentEntity.Submarine.DrawPosition;
728  currentViewPos -= ch.ParentEntity.Submarine.DrawPosition;
729  }
730  //check if a line from the character's collider to the head intersects with the los segment (= head poking through it)
731  if (ch.LosIntersects(defaultViewPos, currentViewPos))
732  {
733  pos = ViewTarget.DrawPosition;
734  }
735  }
736  }
737 
738  if (convexHulls != null)
739  {
740  List<VertexPositionColor> shadowVerts = new List<VertexPositionColor>();
741  List<VertexPositionTexture> penumbraVerts = new List<VertexPositionTexture>();
742  foreach (ConvexHull convexHull in convexHulls)
743  {
744  if (!convexHull.Enabled || !convexHull.Intersects(camView)) { continue; }
745 
746  Vector2 relativeViewPos = pos;
747  if (convexHull.ParentEntity?.Submarine != null)
748  {
749  relativeViewPos -= convexHull.ParentEntity.Submarine.DrawPosition;
750  }
751 
752  convexHull.CalculateLosVertices(relativeViewPos);
753 
754  for (int i = 0; i < convexHull.ShadowVertexCount; i++)
755  {
756  shadowVerts.Add(convexHull.ShadowVertices[i]);
757  }
758 
759  for (int i = 0; i < convexHull.PenumbraVertexCount; i++)
760  {
761  penumbraVerts.Add(convexHull.PenumbraVertices[i]);
762  }
763  }
764 
765  if (shadowVerts.Count > 0)
766  {
767  ConvexHull.shadowEffect.World = shadowTransform;
768  ConvexHull.shadowEffect.CurrentTechnique.Passes[0].Apply();
769  graphics.DrawUserPrimitives(PrimitiveType.TriangleList, shadowVerts.ToArray(), 0, shadowVerts.Count / 3, VertexPositionColor.VertexDeclaration);
770 
771  if (penumbraVerts.Count > 0)
772  {
773  ConvexHull.penumbraEffect.World = shadowTransform;
774  ConvexHull.penumbraEffect.CurrentTechnique.Passes[0].Apply();
775  graphics.DrawUserPrimitives(PrimitiveType.TriangleList, penumbraVerts.ToArray(), 0, penumbraVerts.Count / 3, VertexPositionTexture.VertexDeclaration);
776  }
777  }
778  }
779  }
780  graphics.SetRenderTarget(null);
781  }
