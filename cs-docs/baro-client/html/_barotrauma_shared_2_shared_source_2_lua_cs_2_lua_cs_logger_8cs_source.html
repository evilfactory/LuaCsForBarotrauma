33  public static void HandleException(Exception ex, LuaCsMessageOrigin origin)
34  {
35  switch (ex)
36  {
37  case NetRuntimeException netRuntimeException:
38  if (netRuntimeException.DecoratedMessage == null)
39  {
40  LogError(netRuntimeException.ToString(), origin);
41  }
42  else
43  {
44  // FIXME: netRuntimeException.ToString() doesn't print the InnerException's stack trace...
45  LogError($"{netRuntimeException.DecoratedMessage}: {netRuntimeException}", origin);
46  }
47  break;
48  case InterpreterException interpreterException:
49  if (interpreterException.DecoratedMessage == null)
50  {
51  LogError(interpreterException.ToString(), origin);
52  }
53  else
54  {
55  LogError(interpreterException.DecoratedMessage, origin);
56  }
57  break;
58  default:
59  var msg = ex.StackTrace != null
60  ? ex.ToString()
61  : $"{ex}\n{Environment.StackTrace}";
62  LogError(msg, origin);
63  break;
64  }
65  }
