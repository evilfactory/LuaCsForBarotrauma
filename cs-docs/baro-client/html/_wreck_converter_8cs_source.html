17  {
18  "circuitboxcomponent",

22  public static XElement ConvertToWreck(XElement submarineElement)
23  {
24  ImmutableHashSet<Identifier> availableWreckContainerTags = ItemPrefab.Prefabs
25  .SelectMany(ip => ip.PreferredContainers.SelectMany(pc => pc.Primary.Union(pc.Secondary)))
26  .Where(t => !ItemPrefab.Prefabs.ContainsKey(t) && t.StartsWith("wreck"))
27  .ToImmutableHashSet();
28 
29  bool monsterSpawnPointCreated = false;
30 
31  List<string> warnings = new List<string>();
32 
33  var wreckElement = new XElement(submarineElement);
34  foreach (var element in wreckElement.Elements().ToList())
35  {
36  var identifier = element.GetAttributeIdentifier("identifier", Identifier.Empty);
37  if (identifier.IsEmpty)
38  {
39  if (element.NameAsIdentifier() == "waypoint")
40  {
41  if (element.GetAttributeEnum("spawn", SpawnType.Path) != SpawnType.Human) { continue; }
42  if (element.GetAttributeIdentifier("job", Identifier.Empty) == Identifier.Empty)
43  {
44  element.SetAttributeValue("spawn", SpawnType.Enemy);
45  DebugConsole.NewMessage("Converted a non-job-specific spawnpoint to an enemy spawnpoint.");
46  monsterSpawnPointCreated = true;
47  }
48  else
49  {
50  element.SetAttributeValue("spawn", SpawnType.Corpse);
51  }
52  }
53  continue;
54  }
55 
56  //remove if set to be removed
57  var tags = element.GetAttributeIdentifierArray("tags", Array.Empty<Identifier>());
58  if (itemsToRemove.Any(it => tags.Contains(it.ToIdentifier())))
59  {
60  element.Remove();
61  continue;
62  }
63 
64  bool tagsModified = false;
65  for (int i = 0; i < tags.Length; i++)
66  {
67  Identifier wreckTag = ("wreck" + tags[i]).ToIdentifier();
68  if (availableWreckContainerTags.Contains(wreckTag))
69  {
70  DebugConsole.NewMessage($"Replaced tag {tags[i]} with {wreckTag} in item \"{identifier}\".");
71  tags[i] = wreckTag;
72  tagsModified = true;
73  }
74  }
75  if (tagsModified)
76  {
77  element.SetAttributeValue("tags", string.Join(",", tags.Select(t => t.ToString())));
78  }
79 
80  Identifier[] wreckedIdentifiers =
81  {
82  (identifier + "wrecked").ToIdentifier(),
83  (identifier + "_wrecked").ToIdentifier(),
84  };
85 
86  //turn to wrecked version if one is available
87  foreach (var wreckedIdentifier in wreckedIdentifiers)
88  {
89  var wreckedPrefab = MapEntityPrefab.FindByIdentifier(wreckedIdentifier);
90  if (wreckedPrefab == null) { continue; }
91 
92  var oldPrefab = MapEntityPrefab.FindByIdentifier(identifier);
93  element.SetAttributeValue("identifier", wreckedIdentifier);
94  float currentScale = element.GetAttributeFloat("scale", oldPrefab.Scale);
95  element.SetAttributeValue("scale", currentScale * (wreckedPrefab.Scale / oldPrefab.Scale));
96 
97  if (wreckedPrefab is ItemPrefab wreckedItemPrefab)
98  {
99  //remove connections that don't exist in the wreck version
100  var originalConnectionPanelElement = element.GetChildElement(nameof(ConnectionPanel));
101  var wreckedConnectionPanelElement = wreckedItemPrefab.ConfigElement.GetChildElement(nameof(ConnectionPanel));
102  if (originalConnectionPanelElement != null && wreckedConnectionPanelElement != null)
103  {
104  foreach (var connectionElement in originalConnectionPanelElement.Elements().ToList())
105  {
106  var elementName = connectionElement.NameAsIdentifier();
107  if (elementName != "input" && elementName != "output") { continue; }
108  string connectionName = connectionElement.GetAttributeString("name", string.Empty);
109  if (wreckedConnectionPanelElement
110  .GetChildElements(connectionElement.Name.LocalName)
111  .None(c => c.GetAttributeString("name", string.Empty) == connectionName))
112  {
113  connectionElement.Remove();
114  }
115  }
116  }
117  }
118  else if (wreckedPrefab is StructurePrefab wreckedStructurePrefab)
119  {
120  //if the dimensions of the structures are different, rescale
121  //ignore small differences, they tend to be just irrelevant differences in how the sourcerect is scaled
122  const int MaximumSizeDifference = 5;
123  Rectangle rect = element.GetAttributeRect("rect", Rectangle.Empty);
124  if (!wreckedStructurePrefab.ResizeHorizontal)
125  {
126  if (Math.Abs(wreckedStructurePrefab.ScaledSize.X - rect.Width) > MaximumSizeDifference)
127  {
128  DebugConsole.NewMessage($"The prefab {wreckedStructurePrefab.Name} has different dimensions than the original one. Changing the width from {rect.Width} to {(int)wreckedStructurePrefab.ScaledSize.X}.", Color.Yellow);
129  }
130  rect.Width = (int)wreckedStructurePrefab.ScaledSize.X;
131  }
132  if (!wreckedStructurePrefab.ResizeVertical)
133  {
134  if (Math.Abs(wreckedStructurePrefab.ScaledSize.Y - rect.Height) > MaximumSizeDifference)
135  {
136  DebugConsole.NewMessage($"The prefab {wreckedStructurePrefab.Name} has different dimensions than the original one. Changing the height from {rect.Height} to {(int)wreckedStructurePrefab.ScaledSize.Y}.", Color.Yellow);
137  }
138  rect.Height = (int)wreckedStructurePrefab.ScaledSize.Y;
139  }
140  element.SetAttributeValue("rect", XMLExtensions.RectToString(rect));
141  }
142  break;
143  }
144 
145  var itemContainerElement = element.GetChildElement(nameof(ItemContainer));
146  if (itemContainerElement != null)
147  {
148  string containedString = itemContainerElement.GetAttributeString("contained", "");
149  string[] itemIdStrings = containedString.Split(',');
150  var itemIds = new HashSet<ushort>();
151  foreach (string idListStr in itemIdStrings)
152  {
153  foreach (string idStr in idListStr.Split(';'))
154  {
155  if (int.TryParse(idStr, out int id)) { itemIds.Add((UInt16)id); }
156  }
157  }
158  if (itemIds.Any())
159  {
160  List<string> containedItemNames = new List<string>();
161  foreach (var itemElement in wreckElement.Elements())
162  {
163  var id = itemElement.GetAttributeUInt16("id", Entity.NullEntityID);
164  if (itemIds.Contains(id))
165  {
166  containedItemNames.Add(itemElement.GetAttributeString("identifier", string.Empty));
167  }
168  }
169  warnings.Add($"Potential issue in container \"{identifier}\". The following items are pre-placed, and may interfere with the loot generated in the wreck: " + string.Join(", ", containedItemNames));
170  }
171  }
172 
173  //set to 0 condition if repairable, exclude doors and hatches
174  if (element.GetChildElement(nameof(Repairable)) != null && element.GetChildElement(nameof(Door)) == null)
175  {
176  element.SetAttributeValue("conditionpercentage", 0.0f);
177  }
178  }
179 
180  foreach (var warning in warnings)
181  {
182  DebugConsole.AddWarning(warning);
183  }
184 
185  if (!monsterSpawnPointCreated)
186  {
187  DebugConsole.ThrowError("There are no monster spawnpoints in the wreck. Remember to add some for monsters to spawn properly!");
188  }
189 
190  return wreckElement;
191  }
192  }
193 }
