614  private void GetName(Rand.RandSync randSync, out string name)
615  {
616  ContentXElement nameElement = CharacterConfigElement.GetChildElement("names") ?? CharacterConfigElement.GetChildElement("name");
617  ContentPath namesXmlFile = nameElement?.GetAttributeContentPath("path") ?? ContentPath.Empty;
618  XElement namesXml = null;
619  if (!namesXmlFile.IsNullOrEmpty()) //names.xml is defined
620  {
621  XDocument doc = XMLExtensions.TryLoadXml(namesXmlFile);
622  namesXml = doc.Root;
623  }
624  else //the legacy firstnames.txt/lastnames.txt shit is defined
625  {
626  namesXml = new XElement("names", new XAttribute("format", "[firstname] [lastname]"));
627  string firstNamesPath = nameElement == null ? string.Empty : ReplaceVars(nameElement.GetAttributeContentPath("firstname")?.Value ?? "");
628  string lastNamesPath = nameElement == null ? string.Empty : ReplaceVars(nameElement.GetAttributeContentPath("lastname")?.Value ?? "");
629  if (File.Exists(firstNamesPath) && File.Exists(lastNamesPath))
630  {
631  var firstNames = File.ReadAllLines(firstNamesPath);
632  var lastNames = File.ReadAllLines(lastNamesPath);
633  namesXml.Add(firstNames.Select(n => new XElement("firstname", new XAttribute("value", n))));
634  namesXml.Add(lastNames.Select(n => new XElement("lastname", new XAttribute("value", n))));
635  }
636  else //the files don't exist, just fall back to the vanilla names
637  {
638  XDocument doc = XMLExtensions.TryLoadXml("Content/Characters/Human/names.xml");
639  namesXml = doc.Root;
640  }
641  }
642  name = namesXml.GetAttributeString("format", "");
643  Dictionary<Identifier, List<string>> entries = new Dictionary<Identifier, List<string>>();
644  foreach (var subElement in namesXml.Elements())
645  {
646  Identifier elemName = subElement.NameAsIdentifier();
647  if (!entries.ContainsKey(elemName))
648  {
649  entries.Add(elemName, new List<string>());
650  }
651  ImmutableHashSet<Identifier> identifiers = subElement.GetAttributeIdentifierArray("tags", Array.Empty<Identifier>()).ToImmutableHashSet();
652  if (identifiers.IsSubsetOf(Head.Preset.TagSet))
653  {
654  entries[elemName].Add(subElement.GetAttributeString("value", ""));
655  }
656  }
657 
658  foreach (var k in entries.Keys)
659  {
660  name = name.Replace($"[{k}]", entries[k].GetRandom(randSync), StringComparison.OrdinalIgnoreCase);
661  }
662  }
