114  public void Draw(SpriteBatch spriteBatch, bool editing, float itemDepth = -1, Color? overrideColor = null)
115  {
116  if (target == null || target.Removed) { return; }
117  if (target.ParentInventory != null) { return; }
118  if (source is Limb limb && limb.Removed) { return; }
119  if (source is Entity e && e.Removed) { return; }
120 
121  Vector2 startPos = GetSourcePos(useDrawPosition: true);
122  startPos.Y = -startPos.Y;
123  if ((source as Item)?.GetComponent<Turret>() is { } turret)
124  {
125  if (turret.BarrelSprite != null)
126  {
127  startPos += new Vector2((float)Math.Cos(turret.Rotation), (float)Math.Sin(turret.Rotation)) * turret.BarrelSprite.size.Y * turret.BarrelSprite.RelativeOrigin.Y * turret.Item.Scale * BarrelLengthMultiplier;
128  }
129  startPos -= turret.GetRecoilOffset();
130  }
131  Vector2 endPos = new Vector2(target.DrawPosition.X, target.DrawPosition.Y);
132  Vector2 flippedPos = target.Sprite.size * target.Scale * (Origin - new Vector2(0.5f));
133  if (target.body.Dir < 0.0f)
134  {
135  flippedPos.X = -flippedPos.X;
136  }
137  endPos += Vector2.Transform(flippedPos, Matrix.CreateRotationZ(target.body.Rotation));
138  endPos.Y = -endPos.Y;
139 
140  if (Snapped)
141  {
142  float snapState = 1.0f - snapTimer / SnapAnimDuration;
143  Vector2 diff = target.DrawPosition - new Vector2(startPos.X, -startPos.Y);
144  diff.Y = -diff.Y;
145 
146  int width = (int)(SpriteWidth * snapState);
147  if (width > 0.0f)
148  {
149  float positionMultiplier = snapState;
150  if (BreakFromMiddle)
151  {
152  positionMultiplier /= 2;
153  DrawRope(spriteBatch, endPos - diff * positionMultiplier, endPos, width);
154  }
155  DrawRope(spriteBatch, startPos, startPos + diff * positionMultiplier, width);
156  }
157  }
158  else
159  {
160  DrawRope(spriteBatch, startPos, endPos, SpriteWidth);
161  }
162 
163  if (startSprite != null || endSprite != null)
164  {
165  Vector2 dir = endPos - startPos;
166  float angle = (float)Math.Atan2(dir.Y, dir.X);
167  if (startSprite != null)
168  {
169  float depth = Math.Min(item.GetDrawDepth() + (startSprite.Depth - item.Sprite.Depth), 0.999f);
170  startSprite?.Draw(spriteBatch, startPos, overrideColor ?? SpriteColor, angle, depth: depth);
171  }
172  if (endSprite != null && (!Snapped || BreakFromMiddle))
173  {
174  float depth = Math.Min(item.GetDrawDepth() + (endSprite.Depth - item.Sprite.Depth), 0.999f);
175  endSprite?.Draw(spriteBatch, endPos, overrideColor ?? SpriteColor, angle, depth: depth);
176  }
177  }

226  {

237  }

242  reelSoundChannel.FrequencyMultiplier = MathHelper.Lerp(ReelSoundPitchSlide.X, ReelSoundPitchSlide.Y, MathUtils.InverseLerp(MinPullDistance, MaxLength, currentRopeLength));
243  }
244  }

252  {
253  snapped = msg.ReadBoolean();
254 
255  if (!snapped)

257  ushort targetId = msg.ReadUInt16();
258  ushort sourceId = msg.ReadUInt16();
259  byte limbIndex = msg.ReadByte();
260 
261  if (Entity.FindEntityByID(targetId) is not Item target) { return; }
262  var source = Entity.FindEntityByID(sourceId);
263  switch (source)
264  {
265  case Character sourceCharacter when limbIndex >= 0 && limbIndex < sourceCharacter.AnimController.Limbs.Length:
266  {
267  Limb sourceLimb = sourceCharacter.AnimController.Limbs[limbIndex];
268  Attach(sourceLimb, target);
269  sourceCharacter.AnimController.DragWithRope();
270  break;
271  }
272  case ISpatialEntity spatialEntity:
273  Attach(spatialEntity, target);
274  break;
275  }
