28  public void SellItems(Identifier storeIdentifier, List<SoldItem> itemsToSell, Client client)
29  {
30  var store = Location.GetStore(storeIdentifier);
31  if (store == null) { return; }
32  bool canAddToRemoveQueue = (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer) && Entity.Spawner != null;
33  IEnumerable<Item> sellableItemsInSub = Enumerable.Empty<Item>();
34  if (canAddToRemoveQueue && itemsToSell.Any(i => i.Origin == SoldItem.SellOrigin.Submarine && i.ID == Entity.NullEntityID && !i.Removed))
35  {
36  sellableItemsInSub = GetSellableItemsFromSub();
37  }
38  var itemsSoldAtStore = SoldItems.GetValueOrDefault(storeIdentifier);
39  // Check all the prices before starting the transaction to make sure the modifiers stay the same for the whole transaction
40  var sellValues = GetSellValuesAtCurrentLocation(storeIdentifier, itemsToSell.Select(i => i.ItemPrefab));
41  foreach (var item in itemsToSell)
42  {
43  int itemValue = sellValues[item.ItemPrefab];
44  // check if the store can afford the item and if the item hasn't been removed already
45  if (store.Balance < itemValue || item.Removed) { continue; }
46  // Server determines the items that are sold from the sub in multiplayer
47  if (item.Origin == SoldItem.SellOrigin.Submarine && item.ID == Entity.NullEntityID && !item.Removed)
48  {
49  var matchingItem = sellableItemsInSub.FirstOrDefault(i => !i.Removed && i.Prefab == item.ItemPrefab &&
50  itemsToSell.None(itemToSell => itemToSell.ItemPrefab == i.Prefab && itemToSell.ID == i.ID));
51  // This is a failsafe for scenarios where a client is trying to sell more items than there's available on the sub
52  if (matchingItem == null) { continue; }
53  item.SetItemId(matchingItem.ID);
54  }
55  if (!item.Removed && canAddToRemoveQueue && Entity.FindEntityByID(item.ID) is Item entity)
56  {
57  item.Removed = true;
58  Entity.Spawner.AddItemToRemoveQueue(entity);
59  }
60  itemsSoldAtStore?.Add(item);
61  store.Balance -= itemValue;
62  campaign.GetWallet(client).Give(itemValue);
63  GameAnalyticsManager.AddMoneyGainedEvent(itemValue, GameAnalyticsManager.MoneySource.Store, item.ItemPrefab.Identifier.Value);
64  }
65  OnSoldItemsChanged?.Invoke(this);
66  }
