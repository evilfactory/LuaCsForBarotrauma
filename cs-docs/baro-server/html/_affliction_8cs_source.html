385  public virtual void Update(CharacterHealth characterHealth, Limb targetLimb, float deltaTime)
386  {
387  foreach (AfflictionPrefab.PeriodicEffect periodicEffect in Prefab.PeriodicEffects)
388  {
389  PeriodicEffectTimers[periodicEffect] -= deltaTime;
390  if (PeriodicEffectTimers[periodicEffect] <= 0.0f)
391  {
392  if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient)
393  {
394  PeriodicEffectTimers[periodicEffect] = 0.0f;
395  }
396  else
397  {
398  foreach (StatusEffect statusEffect in periodicEffect.StatusEffects)
399  {
400  ApplyStatusEffect(ActionType.OnActive, statusEffect, 1.0f, characterHealth, targetLimb);
401  PeriodicEffectTimers[periodicEffect] = Rand.Range(periodicEffect.MinInterval, periodicEffect.MaxInterval);
402  }
403  }
404  }
405  }
406 
407  AfflictionPrefab.Effect currentEffect = GetActiveEffect();
408  if (currentEffect == null) { return; }
409 
410  fluctuationTimer += deltaTime * currentEffect.ScreenEffectFluctuationFrequency;
411  fluctuationTimer %= 1.0f;
412 
413  if (currentEffect.StrengthChange < 0) // Only apply StrengthDiminish.Multiplier if affliction is being weakened
414  {
415  float stat = characterHealth.Character.GetStatValue(
416  Prefab.IsBuff
417  ? StatTypes.BuffDurationMultiplier
418  : StatTypes.DebuffDurationMultiplier);
419 
420  float durationMultiplier = 1f / (1f + stat);
421 
422  _strength += currentEffect.StrengthChange * deltaTime * StrengthDiminishMultiplier.Value * durationMultiplier;
423  }
424  else if (currentEffect.StrengthChange > 0) // Reduce strengthening of afflictions if resistant
425  {
426  _strength += currentEffect.StrengthChange * deltaTime * (1f - characterHealth.GetResistance(Prefab));
427  }
428  // Don't use the property, because it's virtual and some afflictions like husk overload it for external use.
429  _strength = MathHelper.Clamp(_strength, 0.0f, Prefab.MaxStrength);
430  activeEffectDirty |= !MathUtils.NearlyEqual(prevActiveEffectStrength, _strength);
431 
432  foreach (StatusEffect statusEffect in currentEffect.StatusEffects)
433  {
434  ApplyStatusEffect(ActionType.OnActive, statusEffect, deltaTime, characterHealth, targetLimb);
435  }
436 
437  float amount = deltaTime;
438  if (Prefab.GrainBurst > 0)
439  {
440  amount /= Prefab.GrainBurst;
441  }
442  if (PendingGrainEffectStrength >= 0)
443  {
444  GrainEffectStrength += amount;
445  PendingGrainEffectStrength -= deltaTime;
446  }
447  else if (GrainEffectStrength > 0)
448  {
449  GrainEffectStrength -= amount;
450  }
451  #if SERVER
452  GameMain.LuaCs.Hook.Call("afflictionUpdate", new object[] { this, characterHealth, targetLimb, deltaTime });
453  #endif
454  }

469  public void ApplyStatusEffect(ActionType type, StatusEffect statusEffect, float deltaTime, CharacterHealth characterHealth, Limb targetLimb)
470  {

477  }
478  if (targetLimb != null && statusEffect.HasTargetType(StatusEffect.TargetType.Limb))
479  {

481  }
482  if (characterHealth?.Character?.AnimController?.Limbs != null && statusEffect.HasTargetType(StatusEffect.TargetType.AllLimbs))

484  statusEffect.Apply(type, deltaTime, characterHealth.Character, targets: characterHealth.Character.AnimController.Limbs);

487  statusEffect.HasTargetType(StatusEffect.TargetType.NearbyCharacters))
