98  private void GenerateOutpost(Submarine submarine)
99  {

101  outpost.SetPosition(Vector2.Zero);
102 
103  float closestDistance = 0.0f;
104  DockingPort myPort = null, outPostPort = null;
105  foreach (DockingPort port in DockingPort.List)
106  {
107  if (port.IsHorizontal || port.Docked) { continue; }
108  if (port.Item.Submarine == outpost)
109  {
110  outPostPort = port;
111  continue;
112  }
113  if (port.Item.Submarine != submarine) { continue; }
114 
115  //the submarine port has to be at the top of the sub
116  if (port.Item.WorldPosition.Y < submarine.WorldPosition.Y) { continue; }
117 
118  float dist = Vector2.DistanceSquared(port.Item.WorldPosition, outpost.WorldPosition);
119  if ((myPort == null || dist < closestDistance || port.MainDockingPort) && !(myPort?.MainDockingPort ?? false))
120  {
121  myPort = port;
122  closestDistance = dist;
123  }
124  }
125 
126  if (myPort != null && outPostPort != null)
127  {
128  Vector2 portDiff = myPort.Item.WorldPosition - submarine.WorldPosition;
129  Vector2 spawnPos = (outPostPort.Item.WorldPosition - portDiff) - Vector2.UnitY * outPostPort.DockedDistance;
130 
131  submarine.SetPosition(spawnPos);
132  myPort.Dock(outPostPort);
133  myPort.Lock(true);
134  }
135 
136  if (Character.Controlled != null)
