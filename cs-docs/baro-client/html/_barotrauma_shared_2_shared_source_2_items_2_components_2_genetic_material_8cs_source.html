273  public CombineResult Combine(GeneticMaterial otherGeneticMaterial, Character user, out Item itemToDestroy)
274  {
275  var combineRefineResult = GetCombineRefineResult(otherGeneticMaterial);
276 
277  float randomQualityIncrease = Rand.Range(ConditionIncreaseOnCombineMin, ConditionIncreaseOnCombineMax);
278  float talentIncrease = user?.GetStatValue(StatTypes.GeneticMaterialRefineBonus) ?? 0.0f;
279 
280  bool perfectQuality = item.IsFullCondition || otherGeneticMaterial.item.IsFullCondition;
281 
282  itemToDestroy = otherGeneticMaterial.item;
283 
284  if (combineRefineResult == CombineResult.Refined)
285  {
286  float maxQuality = Math.Max(item.Condition, otherGeneticMaterial.item.Condition);
287  float minQuality = Math.Min(item.Condition, otherGeneticMaterial.item.Condition);
288  bool oneIsCombined = IsCombined || otherGeneticMaterial.IsCombined;
289 
290  float minQualityProportionalIncreaseBonus = MathHelper.Lerp(ConditionIncreaseOnLowQualityCombine, ConditionIncreaseOnHighQualityCombine, Math.Clamp(minQuality / 80.0f, 0f, 1f));
291  float totalQualityIncrease = minQualityProportionalIncreaseBonus + randomQualityIncrease + talentIncrease;
292  if (oneIsCombined) { totalQualityIncrease /= 2f; }
293 
294  float newQuality = maxQuality + totalQualityIncrease;
295 
296  // the deconstructor wants to remove and delete the item for otherGeneticMaterial,
297  // we want to keep the type that's not being refined here, so we move around the items
298  if (!IsCombined && otherGeneticMaterial.IsCombined)
299  {
300  if (item.Prefab == otherGeneticMaterial.item.Prefab)
301  {
302  // main items share type, nest the non-shared item
303  item.OwnInventory?.TryPutItem(otherGeneticMaterial.NestedMaterial.item, user: null);
304  }
305  else
306  {
307  // the non-shared item is the main item in otherGeneticMaterial,
308  // we need to nest it...
309  item.OwnInventory?.TryPutItem(otherGeneticMaterial.item, user: null);
310 
311  // ...and get rid of the now triple-nested item inside it
312  var otherNestedItem = otherGeneticMaterial.NestedMaterial.item;
313  otherGeneticMaterial.item.OwnInventory?.RemoveItem(otherNestedItem);
314  itemToDestroy = otherNestedItem;
315  }
316  }
317 
318  // note: the condition of combined items represents the quality of both materials,
319  // and the condition of the nested item should equal that of the housing item
320  item.Condition = newQuality;
321  // this can become combined as a result of the item shuffling above
322  if (IsCombined) { NestedMaterial.item.Condition = newQuality; }
323 
324  // if one item is 100% condition, remove taint when refining
325  if (CanBeUntainted && perfectQuality)
326  {
327  SetTainted(false, affectsNestedGene: true);
328  }
329  else if (GetTaintedProbabilityOnRefine(otherGeneticMaterial, user) >= Rand.Range(0.0f, 1.0f))
330  {
331  SetTainted(true);
332  }
333  }
334  else if (combineRefineResult == CombineResult.Combined)
335  {
336  float averageQuality = (item.Condition + otherGeneticMaterial.Item.Condition) / 2.0f;
337  item.Condition = otherGeneticMaterial.Item.Condition = averageQuality + randomQualityIncrease + talentIncrease;
338  item.OwnInventory?.TryPutItem(otherGeneticMaterial.Item, user: null);
339 
340  // if one item is 100% condition, remove taint when combining
341  if (CanBeUntainted && perfectQuality)
342  {
343  SetTainted(false, affectsNestedGene: true);
344  }
345  else if (GetTaintedProbabilityOnCombine(user) >= Rand.Range(0.0f, 1.0f))
346  {
347  SetTainted(true);
348  }
349  }
350 
351  return combineRefineResult;
352  }
