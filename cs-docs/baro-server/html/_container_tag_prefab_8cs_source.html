62  public ImmutableArray<ItemAndProbability> GetItemsAndSpawnProbabilities()
63  {
64  var items = ImmutableArray.CreateBuilder<ItemAndProbability>();
65  foreach (ItemPrefab ip in ItemPrefab.Prefabs)
66  {
67  bool found = false;
68  float spawnProbability = 0f;
69  float campaignSpawnProbability = 0f;
70 
71  foreach (PreferredContainer pc in ip.PreferredContainers)
72  {
73  if (!pc.Primary.Contains(Identifier) && !pc.Secondary.Contains(Identifier)) { continue; }
74 
75  found = true;
76  spawnProbability = Math.Max(pc.SpawnProbability, spawnProbability);
77  if (!pc.NotCampaign)
78  {
79  campaignSpawnProbability = Math.Max(spawnProbability, campaignSpawnProbability);
80  }
81 
82  if (!pc.NotCampaign || pc.CampaignOnly)
83  {
84  campaignSpawnProbability = Math.Max(pc.SpawnProbability, campaignSpawnProbability);
85  }
86  }
87 
88  if (found)
89  {
90  items.Add(new ItemAndProbability(ip, spawnProbability, campaignSpawnProbability));
91  }
92  }
93  return items.ToImmutable();
94  }

96  public static void CheckForContainerTagErrors()
97  {
98  var allContainerTagsInTheGame = new HashSet<Identifier>();
99  var vanillaContainerTags = new HashSet<Identifier>();
100 
101  foreach (var prefab in ItemPrefab.Prefabs)
102  {
103  foreach (Identifier tag in prefab.PreferredContainers.SelectMany(pc => Enumerable.Union(pc.Primary, pc.Secondary)))
104  {
105  allContainerTagsInTheGame.Add(tag);
106  if (prefab.ContentPackage == GameMain.VanillaContent && !TagExistsInItemOrCharacterPrefab(tag))
107  {
108  vanillaContainerTags.Add(tag);
109  }
110  }
111  }

113  static bool TagExistsInItemOrCharacterPrefab(Identifier tag)
114  {
115  if (CharacterPrefab.Prefabs.TryGet(tag, out _))
116  {
117  return true;
118  }
119 
120  foreach (var prefab in ItemPrefab.Prefabs)
121  {
122  if (prefab.Tags.Contains(tag) || prefab.Identifier == tag) { return true; }
123  }
124 
125  return false;
126  }
127 
128  // Find container tags that are defined in a ContainerTagPrefab but not used in any item prefabs.
129  foreach (var prefab in Prefabs)
130  {
131  if (!allContainerTagsInTheGame.Contains(prefab.Identifier))
132  {
133  DebugConsole.AddWarning($"Container tag \"{prefab.Identifier}\" defined in ContainerTagPrefab is not used in any item prefabs, did you misspell it? It's also possible mods override container tags in a way that causes some of the pre-defined tags to become unused.",
134  contentPackage: prefab.ContentPackage);
135  }
136  }
137 
138  // Find container tags that are used in vanilla item prefabs but not defined in a ContainerTagPrefab.
139  // We only check vanilla item prefabs because we don't want to force modders to define all vanilla container tags.
140  foreach (var vanillaTag in vanillaContainerTags)
141  {
142  //mission types can be appended to the respawn container tags for mission-specific respawn items,
143  //check those too so we don't unnecessarily log a warning
144  if (vanillaTag.StartsWith(Tags.RespawnContainer))
145  {
146  bool isMissionSpecificRespawnContainer = false;
147  foreach (var missionType in MissionPrefab.Prefabs.Select(p => p.Type).Distinct())
148  {
149  if (vanillaTag == Tags.RespawnContainer + "_" + missionType)
150  {
151  isMissionSpecificRespawnContainer = true;
152  break;
153  }
154  }
155  if (isMissionSpecificRespawnContainer) { continue; }
156 
157  }
158  if (Prefabs.None(p => p.Identifier == vanillaTag))
159  {
160  DebugConsole.AddWarning($"Container tag \"{vanillaTag}\" is used in vanilla item prefabs but not defined in a ContainerTagPrefab.");
161  }
162  }
163  }
