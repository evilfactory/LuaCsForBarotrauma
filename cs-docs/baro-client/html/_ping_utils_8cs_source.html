134  private static void GetIPAddressPing(ServerInfo serverInfo, IPEndPoint endPoint, Action<ServerInfo> onPingDiscovered)
135  {
136  if (IPAddress.IsLoopback(endPoint.Address))
137  {
138  serverInfo.Ping = Option<int>.Some(0);
139  onPingDiscovered(serverInfo);
140  }
141  else
142  {
143  lock (activePings)
144  {
145  if (activePings.ContainsKey(endPoint)) { return; }
146  activePings.Add(endPoint, activePings.Any() ? activePings.Values.Max() + 1 : 0);
147  }
148  serverInfo.Ping = Option<int>.None();
149  TaskPool.Add($"PingServerAsync ({endPoint})", PingServerAsync(endPoint, 1000),
150  rtt =>
151  {
152  if (!rtt.TryGetResult(out serverInfo.Ping)) { serverInfo.Ping = Option<int>.None(); }
153  onPingDiscovered(serverInfo);
154  lock (activePings)
155  {
156  activePings.Remove(endPoint);
157  }
158  });
159  }

162  private static async Task<Option<int>> PingServerAsync(IPEndPoint endPoint, int timeOut)
163  {
164  await Task.Yield();
165  bool shouldGo = false;
166  while (!shouldGo)
167  {
168  lock (activePings)
169  {
170  shouldGo = activePings.Count(kvp => kvp.Value < activePings[endPoint]) < 25;
171  }
172  await Task.Delay(25);
173  }
174 
175  if (endPoint?.Address == null) { return Option<int>.None(); }
176 
177  //don't attempt to ping if the address is IPv6 and it's not supported
178  if (endPoint.Address.AddressFamily == AddressFamily.InterNetworkV6 && !Socket.OSSupportsIPv6) { return Option<int>.None(); }
179 
180  Ping ping = new Ping();
181  byte[] buffer = new byte[32];
182  try
183  {
184  PingReply pingReply = await ping.SendPingAsync(endPoint.Address, timeOut, buffer, new PingOptions(128, true));
185 
186  return pingReply.Status switch
187  {
188  IPStatus.Success => Option<int>.Some((int)pingReply.RoundtripTime),
189  _ => Option<int>.None(),
190  };
191  }
192  catch (Exception ex)
193  {
194  GameAnalyticsManager.AddErrorEventOnce("ServerListScreen.PingServer:PingException" + endPoint.Address, GameAnalyticsManager.ErrorSeverity.Warning, "Failed to ping a server - " + (ex?.InnerException?.Message ?? ex.Message));
195 #if DEBUG
196  DebugConsole.NewMessage("Failed to ping a server (" + endPoint.Address + ") - " + (ex?.InnerException?.Message ?? ex.Message), Color.Red);
197 #endif
198 
199  return Option<int>.None();
200  }
201  }
