616  public static int ParsePercentage(string value, Identifier attribute = default, XElement? sourceElement = null, bool suppressWarnings = false)
617  {
618  string? line = sourceElement?.ToString().Split('\n')[0].Trim();
619  bool doWarnings = !suppressWarnings && !attribute.IsEmpty && sourceElement != null && line != null;
620 
621  if (string.IsNullOrWhiteSpace(value))
622  {
623  if (doWarnings)
624  {
625  DebugConsole.AddWarning($"Attribute \"{attribute}\" not found at {sourceElement!.Document?.ParseContentPathFromUri()} @ '{line}'.\n " +
626  "Value has been assumed to be '0'.");
627  }
628 
629  return 1;
630  }
631 
632  if (!int.TryParse(value, NumberStyles.Number, CultureInfo.InvariantCulture, out var price))
633  {
634  string str = value;
635 
636  if (str.Length > 1 && str[0] == '+') { str = str.Substring(1); }
637 
638  if (str.Length > 1 && str[^1] == '%') { str = str.Substring(0, str.Length - 1); }
639 
640  if (int.TryParse(str, out price))
641  {
642  return price;
643  }
644  }
645  else
646  {
647  return price;
648  }
649 
650  if (doWarnings)
651  {
652  DebugConsole.AddWarning($"Value in attribute \"{attribute}\" is not formatted correctly\n " +
653  $"at {sourceElement!.Document?.ParseContentPathFromUri()} @ '{line}'.\n " +
654  "It should be an integer with optionally a '+' or '-' at the front and/or '%' at the end.\n" +
655  "The value has been assumed to be '0'.");
656  }
