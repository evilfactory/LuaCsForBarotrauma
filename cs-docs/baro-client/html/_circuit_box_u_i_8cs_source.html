474  public void Update(float deltaTime)
475  {
476  cursorPos = camera.ScreenToWorld(PlayerInput.MousePosition);
477  foreach (CircuitBoxWire wire in CircuitBox.Wires)
478  {
479  wire.Update();
480  }

482  bool foundSelected = false;
483  foreach (var node in CircuitBox.Components)
484  {
485  if (!node.IsSelectedByMe) { continue; }
486 
487  foundSelected = true;
488  if (circuitComponent is not null)
489  {
490  node.UpdateEditing(circuitComponent.RectTransform);
491  }
492  break;
493  }
494 
495  if (!foundSelected)
496  {
497  CircuitBoxComponent.RemoveEditingHUD();
498  }

500  bool isMouseOn = GUI.MouseOn == circuitComponent;
501 
502  if (isMouseOn)

505  }
506  camera.MoveCamera(deltaTime, allowMove: true, allowZoom: isMouseOn, allowInput: isMouseOn, followSub: false);
507 
508  if (camera.TargetPos != Vector2.Zero && MathUtils.NearlyEqual(camera.Position, camera.TargetPos, 0.01f))
509  {
510  camera.TargetPos = Vector2.Zero;
511  }
512 
513  if (isMouseOn)
514  {
515  if (CircuitBox.HeldComponent.IsNone() && PlayerInput.PrimaryMouseButtonDown())
516  {

518  }
519 
520  if (PlayerInput.MidButtonHeld() || (PlayerInput.IsAltDown() && PlayerInput.PrimaryMouseButtonHeld()))
521  {
522  Vector2 moveSpeed = PlayerInput.MouseSpeed / camera.Zoom;
523  moveSpeed.X = -moveSpeed.X;
524  camera.Position += moveSpeed;
525  }
526 
527  if (PlayerInput.PrimaryMouseButtonHeld())
528  {
529  MouseSnapshotHandler.UpdateDrag(GetCursorPosition());
530  }
531 
532  if (MouseSnapshotHandler.IsWiring && MouseSnapshotHandler.LastConnectorUnderCursor.TryUnwrap(out var c))
533  {
534  Vector2 start = c.Rect.Center,
535  end = GetCursorPosition();
536 
537  end.Y = -end.Y;
538 
539  if (!c.IsOutput)
540  {
541  (start, end) = (end, start);
542  }
543 
544  if (DraggedWire.TryUnwrap(out var wire))
545  {
546  wire.Recompute(start, end, CircuitBoxWire.SelectedWirePrefab.SpriteColor);
547  }
548  else
549  {
550  DraggedWire = Option.Some(new CircuitBoxWireRenderer(Option.None,start, end, GUIStyle.Red, CircuitBox.WireSprite));
551  }
552  }
553  else
554  {
555  DraggedWire = Option.None;
556  }
557 
558  if (PlayerInput.SecondaryMouseButtonClicked())

560  OpenContextMenu();
561  }
562 
563  if (PlayerInput.PrimaryMouseButtonClicked())
564  {
565  if (CircuitBox.HeldComponent.TryUnwrap(out ItemPrefab? prefab))
566  {
567  CircuitBox.AddComponent(prefab, cursorPos);
568  }
569  else
570  {
571  if (MouseSnapshotHandler.IsDragging && PlayerInput.PrimaryMouseButtonReleased())
572  {
573  CircuitBox.MoveComponent(MouseSnapshotHandler.GetDragAmount(cursorPos), MouseSnapshotHandler.GetMoveAffectedComponents());
574  }
575  else if (!MouseSnapshotHandler.IsWiring)
576  {
577  TrySelectComponentsUnderCursor();
578  }
579  }
580 
581  if (MouseSnapshotHandler.IsWiring && MouseSnapshotHandler.LastConnectorUnderCursor.TryUnwrap(out var one))
582  {
583  if (MouseSnapshotHandler.FindConnectorUnderCursor(cursorPos).TryUnwrap(out var two))
584  {
585  CircuitBox.AddWire(one, two);
586  }
587  }
588 
589  CircuitBox.SelectWires(MouseSnapshotHandler.LastWireUnderCursor.TryUnwrap(out var wire) ? ImmutableArray.Create(wire) : ImmutableArray<CircuitBoxWire>.Empty, !PlayerInput.IsShiftDown());
590 
591  CircuitBox.HeldComponent = Option.None;
592  MouseSnapshotHandler.EndDragging();
593  }
594 
595  if (MouseSnapshotHandler.GetLastComponentsUnderCursor().IsEmpty && MouseSnapshotHandler.LastConnectorUnderCursor.IsNone())
596  {
597  UpdateSelection();
598  }
599 
600  // Allow using both Delete key and Ctrl+D for those who don't have a Delete key
601  bool hitDeleteCombo = PlayerInput.KeyHit(Keys.Delete) || (PlayerInput.IsCtrlDown() && PlayerInput.KeyHit(Keys.D));
602 
603  if (GUI.KeyboardDispatcher.Subscriber is null && hitDeleteCombo)

606  CircuitBox.RemoveWires(CircuitBox.Wires.Where(static wire => wire.IsSelectedByMe).ToImmutableArray());
607  }
608  }
609 
610  if (componentMenu is { } menu && toggleMenuButton is { } button)
611  {
612  componentMenuOpenState = componentMenuOpen ? Math.Min(componentMenuOpenState + deltaTime * 5.0f, 1.0f) : Math.Max(componentMenuOpenState - deltaTime * 5.0f, 0.0f);
613 
614  menu.RectTransform.ScreenSpaceOffset = Vector2.Lerp(new Vector2(0.0f, menu.Rect.Height - 10), Vector2.Zero, componentMenuOpenState).ToPoint();
615  button.RectTransform.AbsoluteOffset = new Point(menu.Rect.X + ((menu.Rect.Width / 2) - (button.Rect.Width / 2)), menu.Rect.Y - button.Rect.Height);
616  }
