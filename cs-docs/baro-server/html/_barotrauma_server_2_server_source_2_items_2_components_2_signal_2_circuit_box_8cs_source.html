118  public void ServerEventRead(IReadMessage msg, Client c)
119  {
120  var header = (CircuitBoxOpcode)msg.ReadByte();
121  switch (header)
122  {
123  case CircuitBoxOpcode.AddComponent:
124  {
125  var data = INetSerializableStruct.Read<CircuitBoxAddComponentEvent>(msg);
126  if (!CanAccessAndUnlocked(c)) { break; }
127 
128  var prefab = ItemPrefab.Prefabs.Find(p => p.UintIdentifier == data.PrefabIdentifier);
129  if (prefab is null)
130  {
131  ThrowError("Unable to add component because the prefab was not found.", c);
132  return;
133  }
134 
135  if (IsFull || !GetApplicableResourcePlayerHas(prefab, c.Character).TryUnwrap(out var resource)) { return; }
136 
137  ushort id = ICircuitBoxIdentifiable.FindFreeID(Components);
138  if (id is ICircuitBoxIdentifiable.NullComponentID)
139  {
140  ThrowError("Unable to add component because there are no available IDs left.", c);
141  return;
142  }
143 
144  bool result = AddComponentInternal(id, prefab, resource.Prefab, data.Position, c.Character, it =>
145  {
146  CreateServerEvent(new CircuitBoxServerCreateComponentEvent(it.ID, resource.Prefab.UintIdentifier, id, data.Position));
147  });
148 
149  if (!result)
150  {
151  ThrowError("Unable to add component because the component could not be created.", c);
152  return;
153  }
154 
155  GameServer.Log($"{NetworkMember.ClientLogName(c)} added a {prefab.Name} into a circuit box.", ServerLog.MessageType.Wiring);
156  RemoveItem(resource);
157  break;
158  }
159  case CircuitBoxOpcode.MoveComponent:
160  {
161  var data = INetSerializableStruct.Read<CircuitBoxMoveComponentEvent>(msg);
162  if (!item.CanClientAccess(c)) { break; }
163 
164  MoveNodesInternal(data.TargetIDs, data.IOs, data.LabelIDs, data.MoveAmount);
165  CreateServerEvent(data);
166  break;
167  }
168  case CircuitBoxOpcode.DeleteComponent:
169  {
170  var data = INetSerializableStruct.Read<CircuitBoxRemoveComponentEvent>(msg);
171  if (!data.TargetIDs.Any() || !CanAccessAndUnlocked(c)) { break; }
172 
173  CreateRefundItemsForUsedResources(data.TargetIDs, c.Character);
174  GameServer.Log($"{NetworkMember.ClientLogName(c)} removed {GetLogComponentName(data.TargetIDs)} from circuit box.", ServerLog.MessageType.Wiring);
175  RemoveComponentInternal(data.TargetIDs);
176  CreateServerEvent(data);
177  break;
178  }
179  case CircuitBoxOpcode.SelectComponents:
180  {
181  var data = INetSerializableStruct.Read<CircuitBoxSelectNodesEvent>(msg);
182  if (!item.CanClientAccess(c)) { break; }
183 
184  SelectComponentsInternal(data.TargetIDs, c.CharacterID, data.Overwrite);
185  SelectInputOutputInternal(data.IOs, c.CharacterID, data.Overwrite);
186  SelectLabelsInternal(data.LabelIDs, c.CharacterID, data.Overwrite);
187  BroadcastSelectionStatus();
188  break;
189  }
190  case CircuitBoxOpcode.SelectWires:
191  {
192  var data = INetSerializableStruct.Read<CircuitBoxSelectWiresEvent>(msg);
193  if (!item.CanClientAccess(c)) { break; }
194 
195  SelectWiresInternal(data.TargetIDs, c.CharacterID, data.Overwrite);
196  BroadcastSelectionStatus();
197  break;
198  }
199  case CircuitBoxOpcode.AddWire:
200  {
201  var data = INetSerializableStruct.Read<CircuitBoxClientAddWireEvent>(msg);
202  if (!CanAccessAndUnlocked(c)) { break; }
203 
204  var prefab = ItemPrefab.Prefabs.Find(p => p.UintIdentifier == data.SelectedWirePrefabIdentifier);
205  if (prefab is null)
206  {
207  ThrowError($"Unable to connect wire because wire by identifier \"{data.SelectedWirePrefabIdentifier}\" was not found.", c);
208  break;
209  }
210 
211  if (data.Start.FindConnection(this).TryUnwrap(out var start) &&
212  data.End.FindConnection(this).TryUnwrap(out var end))
213  {
214  bool result = Connect(start, end, wire =>
215  {
216  CreateServerEvent(new CircuitBoxServerCreateWireEvent(data with { Start = wire.Start, End = wire.End }, wire.ID, wire.Item.Select(static i => i.ID)));
217  }, prefab);
218

225  }
226  else
227  {
228  ThrowError($"Unable to connect wire because the start or end connection was not found. (start: {data.Start}, end: {data.End})", c);
229  }
230

232  }
233  case CircuitBoxOpcode.RemoveWire:
