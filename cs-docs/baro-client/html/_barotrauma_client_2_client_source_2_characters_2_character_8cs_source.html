662  public Item FindClosestItem(List<Item> itemCollection, Vector2 simPosition, float aimAssistModifier = 0.0f)
663  {
664  if (Submarine != null)
665  {
666  simPosition += Submarine.SimPosition;
667  }
668 
669  debugInteractablesInRange.Clear();
670  debugInteractablesAtCursor.Clear();
671  debugInteractablesNearCursor.Clear();
672 
673  bool draggingItemToWorld = CharacterInventory.DraggingItemToWorld;
674 
675  //reduce the amount of aim assist if an item has been selected
676  //= can't switch selection to another item without deselecting the current one first UNLESS the cursor is directly on the item
677  //otherwise it would be too easy to accidentally switch the selected item when rewiring items
678  float aimAssistAmount = SelectedItem == null ? 100.0f * aimAssistModifier : 1.0f;
679 
680  Vector2 displayPosition = ConvertUnits.ToDisplayUnits(simPosition);
681 
682  Item closestItem = null;
683  float closestItemDistance = Math.Max(aimAssistAmount, 2.0f);
684  foreach (var item in itemCollection)
685  {
686  if (draggingItemToWorld)
687  {
688  if (item.OwnInventory == null ||
689  !item.OwnInventory.Container.AllowDragAndDrop ||
690  !item.OwnInventory.CanBePut(CharacterInventory.DraggingItems.First()) ||
691  !CanAccessInventory(item.OwnInventory))
692  {
693  continue;
694  }
695  }
696 
697  float distanceToItem = float.PositiveInfinity;
698  if (item.IsInsideTrigger(displayPosition, out Rectangle transformedTrigger))
699  {
700  debugInteractablesAtCursor.Add(item);
701  //distance is between 0-1 when the cursor is directly on the item
702  distanceToItem =
703  Math.Abs(transformedTrigger.Center.X - displayPosition.X) / transformedTrigger.Width +
704  Math.Abs((transformedTrigger.Y - transformedTrigger.Height / 2.0f) - displayPosition.Y) / transformedTrigger.Height;
705  //modify the distance based on the size of the trigger (preferring smaller items)
706  distanceToItem *= MathHelper.Lerp(0.05f, 2.0f, (transformedTrigger.Width + transformedTrigger.Height) / 250.0f);
707  }
708  else if (!item.Prefab.RequireCursorInsideTrigger)
709  {
710  Rectangle itemDisplayRect = new Rectangle(item.InteractionRect.X, item.InteractionRect.Y - item.InteractionRect.Height, item.InteractionRect.Width, item.InteractionRect.Height);
711 
712  if (itemDisplayRect.Contains(displayPosition))
713  {
714  debugInteractablesAtCursor.Add(item);
715  //distance is between 0-1 when the cursor is directly on the item
716  distanceToItem =
717  Math.Abs(itemDisplayRect.Center.X - displayPosition.X) / itemDisplayRect.Width +
718  Math.Abs(itemDisplayRect.Center.Y - displayPosition.Y) / itemDisplayRect.Height;
719  //modify the distance based on the size of the item (preferring smaller ones)
720  distanceToItem *= MathHelper.Lerp(0.05f, 2.0f, (itemDisplayRect.Width + itemDisplayRect.Height) / 250.0f);
721  }
722  else
723  {
724  if (closestItemDistance < 2.0f) { continue; }
725  //get the point on the itemDisplayRect which is closest to the cursor
726  Vector2 rectIntersectionPoint = new Vector2(
727  MathHelper.Clamp(displayPosition.X, itemDisplayRect.X, itemDisplayRect.Right),
728  MathHelper.Clamp(displayPosition.Y, itemDisplayRect.Y, itemDisplayRect.Bottom));
729  distanceToItem = 2.0f + Vector2.Distance(rectIntersectionPoint, displayPosition);
730  }
731  }
