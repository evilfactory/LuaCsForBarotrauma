202  private static T GetAnimParams<T>(Identifier speciesName, Identifier animSpecies, Identifier fallbackSpecies, AnimationType animType, Either<string, ContentPath> file, bool throwErrors = true) where T : AnimationParams, new()
203  {
204  Debug.Assert(!speciesName.IsEmpty);
205  Debug.Assert(!animSpecies.IsEmpty);
206  ContentPath contentPath = null;
207  string fileName = null;
208  if (file != null)
209  {
210  if (!file.TryGet(out fileName))
211  {
212  file.TryGet(out contentPath);
213  }
214  Debug.Assert(!fileName.IsNullOrWhiteSpace() || !contentPath.IsNullOrWhiteSpace());
215  }
216  ContentPackage contentPackage = contentPath?.ContentPackage ?? CharacterPrefab.FindBySpeciesName(speciesName)?.ContentPackage;
217  Debug.Assert(contentPackage != null);
218  if (!allAnimations.TryGetValue(speciesName, out Dictionary<string, AnimationParams> animations))
219  {
220  animations = new Dictionary<string, AnimationParams>();
221  allAnimations.Add(speciesName, animations);
222  }
223  string key = fileName ?? contentPath?.Value ?? GetDefaultFileName(animSpecies, animType);
224  if (animations.TryGetValue(key, out AnimationParams anim) && anim.AnimationType == animType)
225  {
226  // Already cached.
227  return (T)anim;
228  }
229  if (!contentPath.IsNullOrEmpty())
230  {
231  // Load the animation from path.
232  T animInstance = new T();
233  if (animInstance.Load(contentPath, speciesName))
234  {
235  if (animInstance.AnimationType == animType)
236  {
237  animations.TryAdd(contentPath.Value, animInstance);
238  return animInstance;
239  }
240  else
241  {
242  errorMessages.Add($"[AnimationParams] Animation type mismatch. Expected: {animType}, Actual: {animInstance.AnimationType}. Using the default animation.");
243  }
244  }
245  else
246  {
247  errorMessages.Add($"[AnimationParams] Failed to load an animation {animInstance} of type {animType} from {contentPath.Value} for the character {speciesName}. Using the default animation.");
248  }
249  }
250  // Seek the correct animation from the character's animation folder.
251  string selectedFile = null;
252  string folder = GetFolder(animSpecies);
253  if (Directory.Exists(folder))
254  {
255  string[] files = Directory.GetFiles(folder);
256  if (files.None())
257  {
258  errorMessages.Add($"[AnimationParams] Could not find any animation files from the folder: {folder}. Using the default animation.");
259  }
260  else
261  {
262  var filteredFiles = FilterAndSortFiles(files, animType);
263  if (filteredFiles.None())
264  {
265  errorMessages.Add($"[AnimationParams] Could not find any animation files that match the animation type {animType} from the folder: {folder}. Using the default animation.");
266  }
267  else if (string.IsNullOrEmpty(fileName))
268  {
269  // Files found, but none specified -> Get a matching animation from the specified folder.
270  // First try to find a file that matches the default file name. If that fails, just take any file.
271  string defaultFileName = GetDefaultFileName(animSpecies, animType);
272  selectedFile = filteredFiles.FirstOrDefault(path => PathMatchesFile(path, defaultFileName)) ?? filteredFiles.First();
273  }
274  else
275  {
276  selectedFile = filteredFiles.FirstOrDefault(path => PathMatchesFile(path, fileName));
277  if (selectedFile == null)
278  {
279  errorMessages.Add($"[AnimationParams] Could not find an animation file that matches the name {fileName} and the animation type {animType}. Using the default animations.");
280  }
281  }
282  }
283  }
284  else
285  {
286  errorMessages.Add($"[AnimationParams] Invalid directory: {folder}. Using the default animation.");
287  }
288  selectedFile ??= GetDefaultFile(fallbackSpecies, animType);
289  Debug.Assert(selectedFile != null);
290  if (errorMessages.None())
291  {
292  DebugConsole.Log($"[AnimationParams] Loading animations from {selectedFile}.");
293  }
294  T animationInstance = new T();
295  if (animationInstance.Load(ContentPath.FromRaw(contentPackage, selectedFile), speciesName))
296  {
297  animations.TryAdd(key, animationInstance);
298  }
299  else
300  {
301  errorMessages.Add($"[AnimationParams] Failed to load an animation {animationInstance} at {selectedFile} of type {animType} for the character {speciesName}");
302  }
303  foreach (string errorMsg in errorMessages)
304  {
305  if (throwErrors)
306  {
307  DebugConsole.ThrowError(errorMsg, contentPackage: contentPackage);
308  }
309  else
310  {
311  DebugConsole.Log("Logging a supressed (potential) error: " + errorMsg);
312  }
313  }
314  errorMessages.Clear();
315  return animationInstance;

366  public static T Create<T>(string fullPath, Identifier speciesName, AnimationType animationType) where T : AnimationParams, new()
367  {
368  if (animationType == AnimationType.NotDefined)
369  {
370  throw new Exception("Cannot create an animation file of type " + animationType.ToString());
371  }
372  if (!allAnimations.TryGetValue(speciesName, out Dictionary<string, AnimationParams> anims))
373  {
374  anims = new Dictionary<string, AnimationParams>();
375  allAnimations.Add(speciesName, anims);
376  }
377  string fileName = IO.Path.GetFileNameWithoutExtension(fullPath);
378  if (anims.ContainsKey(fileName))
379  {
380  DebugConsole.NewMessage($"[AnimationParams] Removing the old animation of type {animationType}.", Color.Red);
381  anims.Remove(fileName);
382  }
383  var instance = new T();
384  XElement animationElement = new XElement(GetDefaultFileName(speciesName, animationType), new XAttribute("animationtype", animationType.ToString()));
385  instance.doc = new XDocument(animationElement);
386  var characterPrefab = CharacterPrefab.FindBySpeciesName(speciesName);
387  Debug.Assert(characterPrefab != null);
388  var contentPath = ContentPath.FromRaw(characterPrefab.ContentPackage, fullPath);
389  instance.UpdatePath(contentPath);
390  instance.IsLoaded = instance.Deserialize(animationElement);
391  instance.Save();
392  instance.Load(contentPath, speciesName);
393  anims.Add(fileName, instance);
394  DebugConsole.NewMessage($"[AnimationParams] New animation file of type {animationType} created.", Color.GhostWhite);
395  return instance;
396  }
