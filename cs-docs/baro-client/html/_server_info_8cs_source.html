122  {

124  {
125  GameMode = GameMain.NetLobbyScreen.SelectedMode?.Identifier ?? Identifier.Empty,
126  GameStarted = Screen.Selected != GameMain.NetLobbyScreen,

129  ContentPackages = ContentPackageManager.EnabledPackages.All.Select(p => new ContentPackageInfo(p)).ToImmutableArray(),
130  Ping = GameMain.Client.Ping,
131 
132  // -------------------------------------
133  // Settings that cannot be copied via
134  // SerializableProperty because they do
135  // not implement the attribute
136  ServerName = serverSettings.ServerName,
137  ServerMessage = serverSettings.ServerMessageText,
138  // -------------------------------------
139  // Settings that cannot be copied via
140  // SerializableProperty due to name mismatch
141  HasPassword = serverSettings.HasPassword,
142  VoipEnabled = serverSettings.VoiceChatEnabled,
143  FriendlyFireEnabled = serverSettings.AllowFriendlyFire,
144  // -------------------------------------
145 
146  Checked = true
147  };
148 
149  var serverInfoSerializableProperties
150  = SerializableProperty.GetProperties(serverInfo);
151  var serverSettingsSerializableProperties
152  = SerializableProperty.GetProperties(serverSettings);
153 
154  var intersection = serverInfoSerializableProperties.Keys
155  .Where(serverSettingsSerializableProperties.ContainsKey);
156 
157  foreach (var key in intersection)
158  {
159  var propToGet = serverSettingsSerializableProperties[key];
160  var propToSet = serverInfoSerializableProperties[key];
161  if (!propToGet.PropertyInfo.CanRead) { continue; }
162  if (!propToSet.PropertyInfo.CanWrite) { continue; }
163  propToSet.SetValue(
164  serverInfo,
165  propToGet.GetValue(serverSettings));
166  }
167 
168  return serverInfo;
169  }

455  public void UpdateInfo(Func<string, string?> valueGetter)
456  {
457  ServerMessage = valueGetter("message") ?? "";

464  if (Enum.TryParse(valueGetter("modeselectionmode"), out SelectionMode modeSelectionMode)) { ModeSelectionMode = modeSelectionMode; }
465  if (Enum.TryParse(valueGetter("subselectionmode"), out SelectionMode subSelectionMode)) { SubSelectionMode = subSelectionMode; }
466 
467  HasPassword = getBool("haspassword");
468  GameStarted = getBool("gamestarted");
469  KarmaEnabled = getBool("karmaenabled");
470  FriendlyFireEnabled = getBool("friendlyfireenabled");
471  AllowSpectating = getBool("allowspectating");
472  AllowRespawn = getBool("allowrespawn");
473  VoipEnabled = getBool("voicechatenabled");
474  GameMode = valueGetter("gamemode")?.ToIdentifier() ?? Identifier.Empty;
475  if (float.TryParse(valueGetter("traitors"), NumberStyles.Any, CultureInfo.InvariantCulture, out float traitorProbability)) { TraitorProbability = traitorProbability; }
476  if (Enum.TryParse(valueGetter("playstyle"), out PlayStyle playStyle)) { PlayStyle = playStyle; }
477  Language = valueGetter("language")?.ToLanguageIdentifier() ?? LanguageIdentifier.None;
478  SelectedSub = valueGetter("submarine") ?? string.Empty;

480  ContentPackages = ExtractContentPackageInfo(ServerName, valueGetter).ToImmutableArray();
481  ContentPackageCount = ContentPackages.Length;
482  if (int.TryParse(valueGetter("packagecount"), out int packageCount)) { ContentPackageCount = packageCount; }
483 
484  bool getBool(string key)
485  {
486  string? data = valueGetter(key);
487  return bool.TryParse(data, out var result) && result;
488  }

491  private static ContentPackageInfo[] ExtractContentPackageInfo(string serverName, Func<string, string?> valueGetter)
492  {
493  //workaround to ServerRules queries truncating the values to 255 bytes
494  int individualPackageIndex = 0;
495  string? individualPackage = valueGetter($"contentpackage{individualPackageIndex}");
496  if (!individualPackage.IsNullOrEmpty())
497  {
498  List<ContentPackageInfo> contentPackages = new List<ContentPackageInfo>();
499  do
500  {
501  string[] splitPackageInfo = individualPackage.Split(',');
502  if (splitPackageInfo.Length != 3)
503  {
504  DebugConsole.Log(
505  $"Error in a server's content package list: malformed content package info ({individualPackage}).");
506  return Array.Empty<ContentPackageInfo>();
507  }
508  string name = splitPackageInfo[0];
509  string hash = splitPackageInfo[1];
510  ulong.TryParse(splitPackageInfo[2], out ulong id);
511  contentPackages.Add(new ContentPackageInfo(name, hash, Option<ContentPackageId>.Some(new SteamWorkshopId(id))));
512 
513  individualPackageIndex++;
514  individualPackage = valueGetter($"contentpackage{individualPackageIndex}");
515  } while (!individualPackage.IsNullOrEmpty());
516  return contentPackages.ToArray();
517  }
518 
519  string? joinedNames = valueGetter("contentpackage");
520  string? joinedHashes = valueGetter("contentpackagehash");
521  string? joinedWorkshopIds = valueGetter("contentpackageid");
522 
523  string[] contentPackageNames = joinedNames.IsNullOrEmpty() ? Array.Empty<string>() : joinedNames.Split(',');
524  string[] contentPackageHashes = joinedHashes.IsNullOrEmpty() ? Array.Empty<string>() : joinedHashes.Split(',');
525  #warning TODO: genericize
526  ulong[] contentPackageIds = joinedWorkshopIds.IsNullOrEmpty() ? new ulong[1] : SteamManager.ParseWorkshopIds(joinedWorkshopIds).ToArray();
527 
528  if (contentPackageNames.Length != contentPackageHashes.Length || contentPackageHashes.Length != contentPackageIds.Length)
529  {
530  DebugConsole.Log(
531  $"The number of names, hashes and Workshop IDs on server \"{serverName}\"" +
532  $" doesn't match: {contentPackageNames.Length} names ({string.Join(", ", contentPackageNames)}), {contentPackageHashes.Length} hashes, {contentPackageIds.Length} ids)");
533  return Array.Empty<ContentPackageInfo>();
534  }
535 
536  return contentPackageNames
537  .Zip(contentPackageHashes, (name, hash) => (name, hash))
538  .Zip(contentPackageIds, (t1, id) =>
539  new ContentPackageInfo(
540  t1.name,
541  t1.hash,
542  Option<ContentPackageId>.Some(new SteamWorkshopId(id))))
543  .ToArray();
544  }

546  public static Option<ServerInfo> FromXElement(XElement element)
547  {
548  string endpointStr
549  = element.GetAttributeString("Endpoint", null)
550  ?? element.GetAttributeString("OwnerID", null)
551  ?? $"{element.GetAttributeString("IP", "")}:{element.GetAttributeInt("Port", 0)}";
552 
553  if (!Endpoint.Parse(endpointStr).TryUnwrap(out var endpoint)) { return Option<ServerInfo>.None(); }
554 
555  var gameVersionStr = element.GetAttributeString("GameVersion", "");
556  if (!Version.TryParse(gameVersionStr, out var gameVersion)) { gameVersion = GameMain.Version; }
557  var info = new ServerInfo(endpoint)
558  {
559  GameVersion = gameVersion
560  };
561  SerializableProperty.DeserializeProperties(info, element);
562 
563  info.MetadataSource = DataSource.Parse(element);
564 
565  return Option<ServerInfo>.Some(info);
566  }
