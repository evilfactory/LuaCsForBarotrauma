246  public void SavePlayers()
247  {
248  //refresh the character data of clients who are still in the server
249  foreach (Client c in GameMain.Server.ConnectedClients)
250  {
251  //ignore if the character is controlling a monster
252  //(we'll just use the previously saved campaign data if there's any)
253  if (c.Character != null && c.Character.Info == null)
254  {
255  c.Character = null;
256  }
257  //use the info of the character the client is currently controlling
258  // or the previously saved info if not (e.g. if the client has been spectating or died)
259  var characterInfo = c.Character?.Info;
260  var matchingCharacterData = characterData.Find(d => d.MatchesClient(c));
261  if (matchingCharacterData != null)
262  {
263  //hasn't spawned this round -> don't touch the data
264  if (!matchingCharacterData.HasSpawned) { continue; }
265  characterInfo ??= matchingCharacterData.CharacterInfo;
266  }
267  if (characterInfo == null || characterInfo.Discarded) { continue; }
268  //reduce skills if the character has died
269  if (characterInfo.CauseOfDeath != null && characterInfo.CauseOfDeath.Type != CauseOfDeathType.Disconnected)
270  {
271  characterInfo.ApplyDeathEffects();
272  }
273  c.CharacterInfo = characterInfo;
274  SetClientCharacterData(c);
275  }
276 
277  //refresh the character data of clients who aren't in the server anymore
278  List<CharacterCampaignData> prevCharacterData = new List<CharacterCampaignData>(characterData);
279  foreach (CharacterCampaignData data in prevCharacterData)
280  {
281  if (data.HasSpawned && !GameMain.Server.ConnectedClients.Any(c => data.MatchesClient(c)))
282  {
283  var character = Character.CharacterList.Find(c => c.Info == data.CharacterInfo && !c.IsHusk);
284  if (character != null &&
285  (!character.IsDead || character.CauseOfDeath?.Type == CauseOfDeathType.Disconnected))
286  {
287  //character still alive (or killed by Disconnect) -> save it as-is
288  characterData.RemoveAll(cd => cd.IsDuplicate(data));
289  data.Refresh(character, refreshHealthData: character.CauseOfDeath?.Type != CauseOfDeathType.Disconnected);
290  characterData.Add(data);
291  }
292  //check the cause of death in the CharacterInfo too (the character instance may have despawned, so we can't just rely on that)
293  else if (data.CharacterInfo.CauseOfDeath is not { Type: CauseOfDeathType.Disconnected })
294  {
295  //character dead or removed -> reduce skills, remove items, health data, etc
296  data.CharacterInfo.ApplyDeathEffects();
297  data.Reset();
298  }
299  }
300  }
301 
302  MoveDiscardedCharacterBalancesToBank();
303 
304  characterData.ForEach(cd => cd.HasSpawned = false);
305  foreach (var cd in characterData)
306  {
307  //remove from crewmanager - we don't need to save the data there if it's been saved as CharacterCampaignData
308  //(e.g. if a client has taken over a bot, we need to do this to prevent it being saved twice)
309  CrewManager.RemoveCharacterInfo(cd.CharacterInfo);
310  }
311 
312  SavePets();
313 
314  //remove all items that are in someone's inventory
315  foreach (Character c in Character.CharacterList)
316  {
317  if (c.Inventory == null) { continue; }
318  if (Level.Loaded.Type == LevelData.LevelType.Outpost && c.Submarine != Level.Loaded.StartOutpost)
319  {
320  Map.CurrentLocation.RegisterTakenItems(c.Inventory.AllItems.Where(it => it.SpawnedInCurrentOutpost && it.OriginalModuleIndex > 0));
321  }
322 
323  if (c.Info != null && c.IsBot)
324  {
325  if (c.IsDead && c.CauseOfDeath?.Type != CauseOfDeathType.Disconnected) { CrewManager.RemoveCharacterInfo(c.Info); }
326  c.Info.HealthData = new XElement("health");
327  c.CharacterHealth.Save(c.Info.HealthData);
328  c.Info.InventoryData = new XElement("inventory");
329  c.SaveInventory();
330  c.Info.SaveOrderData();
331  }
332 
333  c.Inventory.DeleteAllItems();
334  }

354  protected override IEnumerable<CoroutineStatus> DoLevelTransition(TransitionType transitionType, LevelData newLevel, Submarine leavingSub, bool mirror)
355  {
356  IncrementAllLastUpdateIds();
357 
358  switch (transitionType)
359  {
360  case TransitionType.None:
361  throw new InvalidOperationException("Level transition failed (no transitions available).");
362  case TransitionType.ReturnToPreviousLocation:
363  //deselect destination on map
364  map.SelectLocation(-1);
365  break;
366  case TransitionType.ProgressToNextLocation:
367  Map.MoveToNextLocation();
368  TotalPassedLevels++;
369  break;
370  case TransitionType.End:
371  EndCampaign();
372  IsFirstRound = true;
373  break;
374  case TransitionType.ProgressToNextEmptyLocation:
375  Map.Visit(Map.CurrentLocation);
376  TotalPassedLevels++;
377  break;
378  }
379 
380  Map.ProgressWorld(this, transitionType, GameMain.GameSession.RoundDuration);
381 
382  bool success = GameMain.Server.ConnectedClients.Any(c => c.InGame && c.Character != null && !c.Character.IsDead);
383  if (success)
384  {
385  foreach (Client c in GameMain.Server.ConnectedClients)
386  {
387  if (c.Character?.HasAbilityFlag(AbilityFlags.RetainExperienceForNewCharacter) ?? false)
388  {
389  (GameMain.GameSession?.GameMode as MultiPlayerCampaign)?.SaveExperiencePoints(c);
390  }
391  }
392  // Event history must be registered before ending the round or it will be cleared
393  GameMain.GameSession.EventManager.RegisterEventHistory();
394  }
395 
396  //store the currently active missions at this point so we can communicate their states to clients, they're cleared in EndRound
397  List<Mission> missions = GameMain.GameSession.Missions.ToList();
398 
399  GameMain.GameSession.EndRound("", transitionType);
400 
401  //--------------------------------------
402 
403  if (success)
404  {
405  SavePlayers();
406  yield return CoroutineStatus.Running;
407  LeaveUnconnectedSubs(leavingSub);
408  NextLevel = newLevel;
409  GameMain.GameSession.SubmarineInfo = new SubmarineInfo(GameMain.GameSession.Submarine);
410  SaveUtil.SaveGame(GameMain.GameSession.DataPath);
411  }
412  else
413  {
414  PendingSubmarineSwitch = null;
415  GameMain.Server.EndGame(TransitionType.None, wasSaved: false);
416  LoadCampaign(GameMain.GameSession.DataPath, client: null);
417  LastSaveID++;
418  IncrementAllLastUpdateIds();
419  yield return CoroutineStatus.Success;
420  }
421 
422  CrewManager?.ClearCurrentOrders();
423 
424  //--------------------------------------
425 
426  GameMain.Server.EndGame(transitionType, wasSaved: true, missions);
427 
428  ForceMapUI = false;
429 
430  NextLevel = newLevel;
431  MirrorLevel = mirror;
432 
433 
434  yield return new WaitForSeconds(EndTransitionDuration * 0.5f);
435 
436  //don't start the next round automatically if we just finished the campaign
437  if (transitionType != TransitionType.End)
438  {
439  GameMain.Server.TryStartGame();
440  }
441 
442  yield return CoroutineStatus.Success;
443  }

659  msg.WriteUInt16(map.SelectedLocationIndex == -1 ? UInt16.MaxValue : (UInt16)map.SelectedLocationIndex);
