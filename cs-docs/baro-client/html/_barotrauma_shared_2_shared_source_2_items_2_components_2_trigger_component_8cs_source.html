296  public override void Update(float deltaTime, Camera cam)
297  {
298  if (item.Submarine != null && MoveOutsideSub)
299  {
300  item.SetTransform(ConvertUnits.ToSimUnits(item.WorldPosition), item.Rotation);
301  item.CurrentHull = null;
302  item.Submarine = null;
303  SetPhysicsBodyPosition();
304  PhysicsBody.Submarine = item.Submarine;
305  }
306  else if (item.body is { BodyType: BodyType.Dynamic })
307  {
308  SetPhysicsBodyPosition();
309  PhysicsBody.Submarine = item.Submarine;
310  }
311 
312  LevelTrigger.RemoveInActiveTriggerers(PhysicsBody, triggerers);
313 
314  if (triggerOnce)
315  {
316  if (TriggeredOnce) { return; }
317  if (triggerers.Count > 0)
318  {
319  TriggeredOnce = true;
320  IsActive = false;
321  }
322  }
323 
324  TriggerActive = triggerers.Any();
325  if (TriggerActive && conditionals != null)
326  {
327  switch (conditionals.LogicalOperator)
328  {
329  case PropertyConditional.LogicalOperatorType.And:
330  {
331  if (triggerers.Any(t => !PropertyConditional.CheckConditionals((ISerializableEntity)t, conditionals.Conditionals, conditionals.LogicalOperator)))
332  {
333  // Some of the conditionals doesn't match
334  IsActive = false;
335  }
336  break;
337  }
338  case PropertyConditional.LogicalOperatorType.Or:
339  {
340  if (triggerers.None(t => !PropertyConditional.CheckConditionals((ISerializableEntity)t, conditionals.Conditionals, conditionals.LogicalOperator)))
341  {
342  // None of the conditionals match
343  IsActive = false;
344  }
345  break;
346  }
347  }
348  }
349 
350  if (ForceFluctuation && TriggerActive && (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer))
351  {
352  ForceFluctuationTimer += deltaTime;
353  if (ForceFluctuationTimer >= ForceFluctuationInterval)
354  {
355  float v = MathF.Sin(2 * MathF.PI * ForceFluctuationFrequency * TimeInLevel);
356  float amount = MathUtils.InverseLerp(-1.0f, 1.0f, v);
357  CurrentForceFluctuation = MathHelper.Lerp(1.0f - ForceFluctuationStrength, 1.0f, amount);
358  ForceFluctuationTimer = 0.0f;
359  GameMain.NetworkMember?.CreateEntityEvent(this);
360  }
361  }
362 
363  foreach (Entity triggerer in triggerers)
364  {
365  LevelTrigger.ApplyStatusEffects(statusEffects, item.WorldPosition, triggerer, deltaTime, statusEffectTargets, targetItem: Item);
366 
367  if (triggerer is IDamageable damageable)
368  {
369  LevelTrigger.ApplyAttacks(attacks, damageable, item.WorldPosition, deltaTime);
370  }
371  else if (triggerer is Submarine submarine)
372  {
373  LevelTrigger.ApplyAttacks(attacks, item.WorldPosition, deltaTime);
374  foreach (Character c2 in Character.CharacterList)
375  {
376  if (c2.Submarine == submarine)
377  {
378  LevelTrigger.ApplyAttacks(attacks, c2, item.WorldPosition, deltaTime);
379  }
380  }
381  }
382 
383  if (Math.Abs(Force) < 0.01f)
384  {
385  // Just ignore very minimal forces
386  continue;
387  }
388  else if (triggerer is Character c)
389  {
390  if (c.AnimController.Collider.BodyType == BodyType.Dynamic)
391  {
392  if (c.AnimController.Collider.Enabled)
393  {
394  ApplyForce(c.AnimController.Collider);
395  }
396  foreach (var limb in c.AnimController.Limbs)
397  {
398  ApplyForce(limb.body, multiplier: limb.Mass * c.AnimController.Collider.Mass / c.AnimController.Mass);
399  }
400  }
401  }
402  else if (triggerer is Submarine s)
403  {
404  ApplyForce(s.SubBody.Body);
405  }
406  else if (triggerer is Item i && i.body != null)
407  {
408  ApplyForce(i.body);
409  }
410  }
411 
412  item.SendSignal(IsActive ? "1" : "0", "state_out");
413  }

415  private void ApplyForce(PhysicsBody body, float multiplier = 1.0f)
416  {
417  Vector2 diff = ConvertUnits.ToDisplayUnits(item.SimPosition - body.SimPosition);
418  if (diff.LengthSquared() < 0.0001f) { return; }
419  float distanceFactor = DistanceBasedForce ? LevelTrigger.GetDistanceFactor(body, PhysicsBody, RadiusInDisplayUnits) : 1.0f;
420  if (distanceFactor <= 0.0f) { return; }
421  Vector2 force = distanceFactor * (CurrentForceFluctuation * Force) * Vector2.Normalize(diff) * multiplier;
422  if (force.LengthSquared() < 0.01f) { return; }
423  if (body.Mass < 1)
424  {
425  //restrict the force if the body is very light, otherwise it can end up moving at a speed that breaks physics
426  force *= body.Mass;
427  }
428  body.ApplyForce(force);
429  }
