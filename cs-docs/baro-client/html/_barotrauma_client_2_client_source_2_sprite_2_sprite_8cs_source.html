303  Vector2 position,
304  Vector2 targetSize,

318  bool flipVertical = spriteEffects.Value.HasFlag(SpriteEffects.FlipVertically);

320  float addedRotation = rotation + this.rotation;
321  if (flipHorizontal != flipVertical) { addedRotation = -addedRotation; }
322 
323  Vector2 advanceX = addedRotation == 0.0f ? Vector2.UnitX : new Vector2((float)Math.Cos(addedRotation), (float)Math.Sin(addedRotation));
324  Vector2 advanceY = new Vector2(-advanceX.Y, advanceX.X);
325 
326  //Init optional values
327  Vector2 drawOffset = startOffset ?? Vector2.Zero;
328  Vector2 scale = textureScale ?? Vector2.One;
329  Color drawColor = color ?? Color.White;
330  Vector2 transformedOrigin = origin ?? Vector2.Zero;
331 
332  transformedOrigin = advanceX * transformedOrigin.X + advanceY * transformedOrigin.Y;

334  void drawSection(Vector2 slicePos, Rectangle sliceRect)
335  {
336  Vector2 transformedPos = slicePos;
337 
338  if (flipHorizontal)
339  {
340  transformedPos.X = targetSize.X - transformedPos.X - sliceRect.Width * scale.X;
341  }
342  if (flipVertical)
343  {
344  transformedPos.Y = targetSize.Y - transformedPos.Y - sliceRect.Height * scale.Y;
345  }
346 
347  transformedPos = advanceX * transformedPos.X + advanceY * transformedPos.Y;
348  transformedPos += position - transformedOrigin;
349  spriteBatch.Draw(
350  texture: texture,
351  position: transformedPos,
352  sourceRectangle: sliceRect,
353  color: drawColor,
354  rotation: addedRotation,
355  origin: Vector2.Zero,
356  scale: scale,
357  effects: spriteEffects.Value,
358  layerDepth: depth ?? this.depth);
359  }
360 
361  //wrap the drawOffset inside the sourceRect
362  drawOffset.X = (drawOffset.X / scale.X) % sourceRect.Width;
363  drawOffset.Y = (drawOffset.Y / scale.Y) % sourceRect.Height;
364 
365  //how many times the texture needs to be drawn on the x-axis
366  int xTiles = (int)Math.Ceiling((targetSize.X + drawOffset.X * scale.X) / (sourceRect.Width * scale.X));
367  //how many times the texture needs to be drawn on the y-axis
368  int yTiles = (int)Math.Ceiling((targetSize.Y + drawOffset.Y * scale.Y) / (sourceRect.Height * scale.Y));
369 
370  //where the current tile is being drawn;
371  Vector2 currDrawPosition = -drawOffset;
372  //which part of the texture we are currently drawing
373  Rectangle texPerspective = sourceRect;
374 
375 
376  for (int x = 0; x < xTiles; x++)
377  {
378  texPerspective.X = sourceRect.X;
379  texPerspective.Width = sourceRect.Width;
380  texPerspective.Height = sourceRect.Height;
381 
382  //offset to the left, draw a partial slice
383  if (currDrawPosition.X < 0)
384  {
385  float diff = -currDrawPosition.X;
386  currDrawPosition.X += diff;
387  texPerspective.Width -= (int)diff;
388 
389  texPerspective.X += (int)diff;
390  }
391  //make sure the rightmost tiles don't go over the right side
392  if (x == xTiles - 1)
393  {
394  int diff = (int)(((currDrawPosition.X + texPerspective.Width * scale.X) - targetSize.X) / scale.X);
395  texPerspective.Width -= diff;
396  }
397 
398  currDrawPosition.Y = -drawOffset.Y;
399 
400  for (int y = 0; y < yTiles; y++)
401  {
402  texPerspective.Y = sourceRect.Y;
403  texPerspective.Height = sourceRect.Height;
404 
405  //offset above the top, draw a partial slice
406  if (currDrawPosition.Y < 0f)
407  {
408  float diff = -currDrawPosition.Y;
409  currDrawPosition.Y += diff;
410  texPerspective.Height -= (int)diff;
411  texPerspective.Y += (int)diff;
412  }
413 
414  //make sure the bottommost tiles don't go over the bottom
415  if (y == yTiles - 1)
416  {
417  int diff = (int)(((currDrawPosition.Y + texPerspective.Height * scale.Y) - targetSize.Y) / scale.Y);
418  texPerspective.Height -= diff;
419  }
420 
421  drawSection(currDrawPosition, texPerspective);
422 
423  currDrawPosition.Y += texPerspective.Height * scale.Y;
424  }
425 
426  currDrawPosition.X += texPerspective.Width * scale.X;
427  }
