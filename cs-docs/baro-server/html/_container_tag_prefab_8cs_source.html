61  public ImmutableArray<ItemAndProbability> GetItemsAndSpawnProbabilities()
62  {
63  var items = ImmutableArray.CreateBuilder<ItemAndProbability>();
64  foreach (ItemPrefab ip in ItemPrefab.Prefabs)
65  {
66  bool found = false;
67  float spawnProbability = 0f;
68  float campaignSpawnProbability = 0f;
69 
70  foreach (PreferredContainer pc in ip.PreferredContainers)
71  {
72  if (!pc.Primary.Contains(Identifier) && !pc.Secondary.Contains(Identifier)) { continue; }
73 
74  found = true;
75  spawnProbability = Math.Max(pc.SpawnProbability, spawnProbability);
76  if (!pc.NotCampaign)
77  {
78  campaignSpawnProbability = Math.Max(spawnProbability, campaignSpawnProbability);
79  }
80 
81  if (!pc.NotCampaign || pc.CampaignOnly)
82  {
83  campaignSpawnProbability = Math.Max(pc.SpawnProbability, campaignSpawnProbability);
84  }
85  }
86 
87  if (found)
88  {
89  items.Add(new ItemAndProbability(ip, spawnProbability, campaignSpawnProbability));
90  }
91  }
92  return items.ToImmutable();
93  }

95  public static void CheckForContainerTagErrors()
96  {
97  var allContainerTagsInTheGame = new HashSet<Identifier>();
98  var vanillaContainerTags = new HashSet<Identifier>();
99 
100  foreach (var prefab in ItemPrefab.Prefabs)
101  {
102  foreach (Identifier tag in prefab.PreferredContainers.SelectMany(pc => Enumerable.Union(pc.Primary, pc.Secondary)))
103  {
104  allContainerTagsInTheGame.Add(tag);
105  if (prefab.ContentPackage == GameMain.VanillaContent && !TagExistsInItemOrCharacterPrefab(tag))
106  {
107  vanillaContainerTags.Add(tag);
108  }
109  }
110  }

112  static bool TagExistsInItemOrCharacterPrefab(Identifier tag)
113  {
114  if (CharacterPrefab.Prefabs.TryGet(tag, out _))
115  {
116  return true;
117  }
118 
119  foreach (var prefab in ItemPrefab.Prefabs)
120  {
121  if (prefab.Tags.Contains(tag) || prefab.Identifier == tag) { return true; }
122  }
123 
124  return false;
125  }
126 
127  // Find container tags that are defined in a ContainerTagPrefab but not used in any item prefabs.
128  foreach (var prefab in Prefabs)
129  {
130  if (!allContainerTagsInTheGame.Contains(prefab.Identifier))
131  {
132  DebugConsole.AddWarning($"Container tag \"{prefab.Identifier}\" defined in ContainerTagPrefab is not used in any item prefabs, did you misspell it? It's also possible mods override container tags in a way that causes some of the pre-defined tags to become unused.",
133  contentPackage: prefab.ContentPackage);
134  }
135  }
136 
137  // Find container tags that are used in vanilla item prefabs but not defined in a ContainerTagPrefab.
138  // We only check vanilla item prefabs because we don't want to force modders to define all vanilla container tags.
139  foreach (var vanillaTag in vanillaContainerTags)
140  {
141  if (Prefabs.All(p => p.Identifier != vanillaTag))
142  {
143  DebugConsole.AddWarning($"Container tag \"{vanillaTag}\" is used in vanilla item prefabs but not defined in a ContainerTagPrefab.");
144  }
145  }
