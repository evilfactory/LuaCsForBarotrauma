121  private bool TryCreateTraitorEvents(EventManager eventManager, Level level)
122  {
123  var eventPrefabs = EventPrefab.Prefabs.Where(p => p is TraitorEventPrefab).Cast<TraitorEventPrefab>();
124  if (!eventPrefabs.Any())
125  {
126  DebugConsole.AddWarning("No traitor event available in any of the enabled content packages.");
127  return false;
128  }
129 
130  if (server.ConnectedClients.Count(IsClientViableTraitor) < server.ServerSettings.TraitorsMinPlayerCount)
131  {
132  DebugConsole.AddWarning("Not enough clients to create a traitor event. Active traitor events: " + activeEvents.Count);
133 #if DEBUG
134  DebugConsole.AddWarning("Starting a traitor event anyway because this is a debug build.");
135 #else
136  return false;
137 #endif
138 
139  }
140 
141  int maxDangerLevel = server.ServerSettings.TraitorDangerLevel;
142 
143  int playerCount = server.ConnectedClients.Count(c => c.Character != null && !c.Character.Removed);
144 
145  var campaign = GameMain.GameSession?.Campaign;
146  var suitablePrefabs = eventPrefabs
147  .Where(e => EventManager.IsLevelSuitable(e, level))
148  .Where(e => e.ReputationRequirementsMet(campaign))
149  .Where(e => e.MissionRequirementsMet(GameMain.GameSession))
150  .Where(e => e.LevelRequirementsMet(level))
151  .Where(e => e.DangerLevel <= maxDangerLevel)
152  .Where(e => playerCount >= e.MinPlayerCount)
153  .ToList();
154 
155  if (!suitablePrefabs.Any())
156  {
157  //this is normal, there e.g. might be no missions for an abandoned outpost or end level
158  DebugConsole.Log("No suitable traitor missions available for the level.");
159  return false;
160  }
161 
162  foreach (var previousEvent in previousTraitorEvents.Reverse<PreviousTraitorEvent>().DistinctBy(e => e.Traitor))
163  {
164  if (previousEvent.State == TraitorEvent.State.Completed &&
165  previousEvent.TraitorEvent.IsChainable &&
166  IsClientViableTraitor(previousEvent.Traitor))
167  {
168  GameServer.Log($"{NetworkMember.ClientLogName(previousEvent.Traitor)} successfully completed a traitor event ({previousEvent.TraitorEvent.Identifier}) on a previous round. Attempting to give choose them a new, more dangerous event...", ServerLog.MessageType.Traitors);
169 
170  var suitablePrefab =
171  //try finding an event that's continuation from the previous one (= requires the previous one to be completed 1st)
172  suitablePrefabs.GetRandomUnsynced(p => p.RequiredCompletedTags.Any(t =>
173  previousEvent.TraitorEvent.Identifier == t || previousEvent.TraitorEvent.Tags.Contains(t)));
174 
175  suitablePrefab ??=
176  //otherwise try finding some higher-difficult event for the same faction
177  suitablePrefabs.GetRandomUnsynced(p =>
178  p.RequiredCompletedTags.None() &&
179  p.DangerLevel > previousEvent.TraitorEvent.DangerLevel &&
180  p.Faction == previousEvent.TraitorEvent.Faction);
181 
182  if (suitablePrefab != null)
183  {
184  CreateTraitorEvent(eventManager, suitablePrefab, previousEvent.Traitor);
185  return true;
186  }
187  else
188  {
189  GameServer.Log($"Could not find a suitable, more difficult traitor event for {NetworkMember.ClientLogName(previousEvent.Traitor)}.", ServerLog.MessageType.Traitors);
190  }
191  }
192  }
193 
194  TraitorEventPrefab selectedPrefab;
195  if (GameMain.GameSession?.Campaign == null)
196  {
197  selectedPrefab = suitablePrefabs.GetRandomByWeight(GetTraitorEventPrefabCommonness, Rand.RandSync.Unsynced);
198  }
199  else
200  {
201  //events that are suitable as initial traitor events (not requiring some other event to be completed first)
202  var suitableInitialPrefabs = suitablePrefabs.FindAll(e => e.RequiredCompletedTags.None() && IsSuitableDangerLevel(e));
203 
204  bool IsSuitableDangerLevel(TraitorEventPrefab prefab)
205  {
206  if (prefab.DangerLevel == TraitorEventPrefab.MinDangerLevel) { return true; }
207  //events that require another event to be completed are handled earlier in the method
208  if (prefab.RequirePreviousDangerLevelCompleted) { return false; }
209  return
210  prefab.RequiredPreviousDangerLevel < TraitorEventPrefab.MinDangerLevel ||
211  previousTraitorEvents.Any(e2 => e2.TraitorEvent.DangerLevel >= prefab.RequiredPreviousDangerLevel);
212  }
213 
214  selectedPrefab = suitableInitialPrefabs.GetRandomByWeight(GetTraitorEventPrefabCommonness, Rand.RandSync.Unsynced);
215  if (selectedPrefab == null)
216  {
217  GameServer.Log($"Could not find a suitable danger level {TraitorEventPrefab.MinDangerLevel} traitor event. Choosing a random event instead.", ServerLog.MessageType.Traitors);
218  selectedPrefab = suitablePrefabs.GetRandomByWeight(GetTraitorEventPrefabCommonness, Rand.RandSync.Unsynced);
219  }
220  }
221 
222  if (selectedPrefab != null)
223  {
224  var selectedTraitor = SelectRandomTraitor();
225  if (selectedTraitor == null)
226  {
227  DebugConsole.ThrowError($"Could not find a suitable traitor for the event \"{selectedPrefab.Identifier}\".",
228  contentPackage: selectedPrefab.ContentPackage);
229  return false;
230  }
231  CreateTraitorEvent(eventManager, selectedPrefab, selectedTraitor);
232  return true;
233  }
234 
235  return false;
236  }

250  private IEnumerable<Client> SelectSecondaryTraitors(TraitorEvent traitorEvent, Client mainTraitor)
251  {
252  if (traitorEvent.Prefab.SecondaryTraitorPercentage <= 0.0f &&
253  traitorEvent.Prefab.SecondaryTraitorAmount <= 0)
254  {
255  return Enumerable.Empty<Client>();
256  }
257 
258  var viableTraitors = server.ConnectedClients.Where(c => c != mainTraitor && IsClientViableTraitor(c)).ToList();
259 
260  int amountToChoose = (int)Math.Ceiling(viableTraitors.Count * (traitorEvent.Prefab.SecondaryTraitorPercentage / 100.0f));
261  amountToChoose = Math.Max(amountToChoose, traitorEvent.Prefab.SecondaryTraitorAmount);
262 
263  if (amountToChoose > viableTraitors.Count)
264  {
265  DebugConsole.ThrowError(
266  $"Error in traitor event {traitorEvent.Prefab.Identifier}. Not enough players to choose {amountToChoose} secondary traitors. " +
267  $"Make sure the {nameof(traitorEvent.Prefab.MinPlayerCount)} of the event is high enough to support to desired amount of secondary traitors.",
268  contentPackage: traitorEvent.Prefab.ContentPackage);
269  amountToChoose = viableTraitors.Count;
270  }
271 
272  List<Client> traitors = new List<Client>();
273  for (int i = 0; i < amountToChoose; i++)
274  {
275  var traitor = viableTraitors.GetRandomUnsynced();
276  viableTraitors.Remove(traitor);
277  traitors.Add(traitor);
278  }
279  return traitors;
280  }

292  private float GetTraitorEventPrefabCommonness(TraitorEventPrefab prefab)
293  {
294  int? roundsSinceLastSelected = GetRoundsSinceLastSelected(e => e.TraitorEvent == prefab);
295  if (roundsSinceLastSelected.HasValue)
296  {
297  float normalizedRoundsSinceLastSelected = MathUtils.InverseLerp(0, MaxPreviousEventHistory, roundsSinceLastSelected.Value);
298  //exponentially decreasing commonness the closer the last time the event was selected
299  return prefab.Commonness * normalizedRoundsSinceLastSelected * normalizedRoundsSinceLastSelected;
300  }
301  else
302  {
303  return prefab.Commonness;
304  }
305  }
306 
307  private float GetTraitorProbability(Client client)
308  {
309  int? roundsSinceLastSelected = GetRoundsSinceLastSelected(e => e.Traitor == client);
310  if (roundsSinceLastSelected.HasValue)
311  {
312  float normalizedRoundsSinceLastSelected = MathUtils.InverseLerp(0, MaxPreviousEventHistory, roundsSinceLastSelected.Value);
313  //exponentially decreasing commonness the closer the last time the event was selected
314  return normalizedRoundsSinceLastSelected * normalizedRoundsSinceLastSelected;
315  }
316  else
317  {
318  return 1.0f;
319  }
320  }

322  private int? GetRoundsSinceLastSelected(Func<PreviousTraitorEvent, bool> condition)
323  {
324  //most recent events are at the end of the list, start from there
325  for (int i = previousTraitorEvents.Count - 1; i >= 0; i--)
326  {
327  if (condition(previousTraitorEvents[i]))
328  {
329  return previousTraitorEvents.Count - i;
330  }
331  }
332  return null;

335  private void CreateTraitorEvent(EventManager eventManager, TraitorEventPrefab selectedPrefab, Client traitor)
336  {
337  if (selectedPrefab.TryCreateInstance<TraitorEvent>(eventManager.RandomSeed, out var newEvent))
338  {
339  var secondaryTraitors = SelectSecondaryTraitors(newEvent, traitor);
340 
341  string logMessage = $"{NetworkMember.ClientLogName(traitor)} was selected as a traitor. Selected event: {selectedPrefab.Name}";
342  if (secondaryTraitors.Any())
343  {
344  logMessage += $", secondary traitors: {string.Join(", ", secondaryTraitors.Select(c => NetworkMember.ClientLogName(c)))}";
345  }
346  GameServer.Log(logMessage, ServerLog.MessageType.Traitors);
347 
348  newEvent.OnStateChanged += () => SendCurrentState(newEvent);
349  activeEvents.Add(new ActiveTraitorEvent(traitor, newEvent));
350  newEvent.SetTraitor(traitor);
351  newEvent.SetSecondaryTraitors(secondaryTraitors);
352  eventManager.ActivateEvent(newEvent);
353  SendCurrentState(newEvent);
354  }
355  else
356  {
357  DebugConsole.ThrowError($"Failed to create an instance of the traitor event prefab \"{selectedPrefab.Identifier}\"!",
358  contentPackage: selectedPrefab.ContentPackage);
359  }

384  public void Update(float deltaTime)
385  {
386  if (!Enabled) { return; }
387  if (!started)
388  {
389  if (level?.LevelData is { Type: LevelData.LevelType.LocationConnection })
390  {
391  if (Submarine.MainSub.WorldPosition.X > level.Size.X / 2)
392  {
393  //try starting ASAP if the submarine is already half-way through the level
394  //(brief delay regardless, because otherwise we might retry every frame if finding a suitable event fails below)
395  startTimer = Math.Min(startTimer, 10.0f);
396  }
397  }
398  startTimer -= deltaTime;
399  if (startTimer >= 0.0f) { return; }
400  if (eventManager == null)
401  {
402  throw new InvalidOperationException("EventManager was null. TraitorManager may not have been initialized properly.");
403  }
404  if (level == null)
405  {
406  throw new InvalidOperationException("Level was null. TraitorManager may not have been initialized properly.");
407  }
408  if (TryCreateTraitorEvents(eventManager, level))
409  {
410  started = true;
411  }
412  else
413  {
414  //restart timer, we might be able to start a mission later if more clients join
415  startTimer = Rand.Range(StartDelayMin, StartDelayMax);
416  }
417  }

420  public void EndRound()
421  {
422  Client? votedAsTraitor = GetClientAccusedAsTraitor();
423  foreach (var activeEvent in activeEvents)
424  {
425  if (results != null)
426  {
427  DebugConsole.AddWarning("Multiple traitor events active during the round, only displaying the results for the last one.");
428  }
429  results = new TraitorResults(votedAsTraitor, activeEvent.TraitorEvent);
430  if (results.Value.MoneyPenalty > 0)
431  {
432  GameMain.GameSession?.Campaign?.Bank?.TryDeduct(results.Value.MoneyPenalty);
433  }
434 
435  if (activeEvent.TraitorEvent.CurrentState != TraitorEvent.State.Completed)
436  {
437  activeEvent.TraitorEvent.CurrentState = TraitorEvent.State.Failed;
438  }
439  GameServer.Log(
440  NetworkMember.ClientLogName(activeEvent.Traitor) +
441  (activeEvent.TraitorEvent.CurrentState == TraitorEvent.State.Completed ?
442  " completed their traitor objective successfully." : " failed to complete their traitor objective."),
443  ServerLog.MessageType.Traitors);
444 
445  //consider the event failed if the traitor was identifier correctly,
446  //so the traitor doesn't get rewards or get assigned a follow-up event
447  if (results.Value.VotedCorrectTraitor)
448  {
449  GameServer.Log(
450  NetworkMember.ClientLogName(activeEvent.Traitor) + " was correctly identified as the traitor, and will not receive any rewards.",
451  ServerLog.MessageType.Traitors);
452  activeEvent.TraitorEvent.CurrentState = TraitorEvent.State.Failed;
453  }
454  previousTraitorEvents.Add(new PreviousTraitorEvent(
455  activeEvent.TraitorEvent.Prefab,
456  activeEvent.TraitorEvent.CurrentState,
457  activeEvent.Traitor));
458 
459  if (activeEvent.TraitorEvent.CurrentState == TraitorEvent.State.Completed)
460  {
461  AchievementManager.OnTraitorWin(activeEvent.TraitorEvent.Traitor?.Character);
462  foreach (var secondaryTraitor in activeEvent.TraitorEvent.SecondaryTraitors)
463  {
464  AchievementManager.OnTraitorWin(secondaryTraitor?.Character);
465  }
466  }
467  }
468  if (previousTraitorEvents.Count > MaxPreviousEventHistory)
469  {
470  previousTraitorEvents.RemoveRange(0, previousTraitorEvents.Count - MaxPreviousEventHistory);
471  }
472  activeEvents.Clear();
473  }
