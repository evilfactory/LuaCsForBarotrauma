247  public bool ClientWrite(IWriteMessage msg, VoteType voteType, object data)
248  {
249  msg.WriteByte((byte)voteType);
250 
251  switch (voteType)
252  {
253  case VoteType.Sub:
254  if (data is not SubmarineInfo sub) { return false; }
255  msg.WriteInt32(sub.EqualityCheckVal);
256  if (sub.EqualityCheckVal <= 0)
257  {
258  //sub doesn't exist client-side, use hash to let the server know which one we voted for
259  msg.WriteString(sub.MD5Hash.StringRepresentation);
260  }
261  break;
262  case VoteType.Mode:
263  if (data is not GameModePreset gameMode) { return false; }
264  msg.WriteIdentifier(gameMode.Identifier);
265  break;
266  case VoteType.EndRound:
267  if (data is not bool endRound) { return false; }
268  msg.WriteBoolean(endRound);
269  break;
270  case VoteType.Kick:
271  if (data is not Client votedClient) { return false; }
272 
273  msg.WriteByte(votedClient.SessionId);
274  break;
275  case VoteType.StartRound:
276  if (data is not bool startRound) { return false; }
277  msg.WriteBoolean(startRound);
278  break;
279  case VoteType.PurchaseAndSwitchSub:
280  case VoteType.PurchaseSub:
281  case VoteType.SwitchSub:
282  switch (data)
283  {
284  case (SubmarineInfo voteSub, bool transferItems):
285  //initiate sub vote
286  msg.WriteBoolean(true);
287  msg.WriteString(voteSub.Name);
288  msg.WriteBoolean(transferItems);
289  break;
290  case int vote:
291  // vote
292  msg.WriteBoolean(false);
293  msg.WriteInt32(vote);
294  break;
295  default:
296  return false;
297  }
298  break;
299  case VoteType.TransferMoney:
300  if (data is not int money) { return false; }
301  msg.WriteBoolean(false); //not initiating a vote
302  msg.WriteInt32(money);
303  break;
304  case VoteType.Traitor:
305  //use 0 to indicate we voted for no-one
306  msg.WriteInt32((data as Client)?.SessionId ?? 0);
307  break;
308  }
309 
310  msg.WritePadBits();
311  return true;
312  }

480  }
481

483  if (passed && submarineVoteInfo.SubmarineInfo is { } subInfo)
484  {
