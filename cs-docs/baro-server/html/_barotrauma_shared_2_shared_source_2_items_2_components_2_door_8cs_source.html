437  public override void Update(float deltaTime, Camera cam)
438  {
439  UpdateProjSpecific(deltaTime);
440  toggleCooldownTimer -= deltaTime;
441  damageSoundCooldown -= deltaTime;
442 
443  if (isBroken)
444  {
445  lastBrokenTime = Timing.TotalTime;
446  //the door has to be restored to 50% health before collision detection on the body is re-enabled
447 
448  //multiply by MaxRepairConditionMultiplier so the item gets repaired at 50% of the _default max condition_
449  //otherwise increasing the max condition is arguably harmful, as the door needs to be repaired further to re-enable the collider
450  if (item.ConditionPercentage * Math.Max(item.MaxRepairConditionMultiplier, 1.0f) > 50.0f &&
451  (GameMain.NetworkMember == null || GameMain.NetworkMember.IsServer))
452  {
453  IsBroken = false;
454  }
455  return;
456  }
457 
458  bool isClosing = false;
459  if ((!IsStuck && !IsJammed) || !isOpen)
460  {
461  if (PredictedState == null)
462  {
463  OpenState += deltaTime * (isOpen ? OpeningSpeed : -ClosingSpeed);
464  isClosing = openState > 0.0f && openState < 1.0f && !isOpen;
465  }
466  else
467  {
468  OpenState += deltaTime * ((bool)PredictedState ? OpeningSpeed : -ClosingSpeed);
469  isClosing = openState > 0.0f && openState < 1.0f && !(bool)PredictedState;
470 
471  resetPredictionTimer -= deltaTime;
472  if (resetPredictionTimer <= 0.0f)
473  {
474  PredictedState = null;
475  }
476  }
477  LinkedGap.Open = isBroken ? 1.0f : openState;
478  }
479 
480  if (isClosing)
481  {
482  if (OpenState < 0.9f) { PushCharactersAway(); }
483  if (CheckSubmarinesInDoorWay())
484  {
485  PredictedState = null;
486  isOpen = true;
487  }
488  }
489  else
490  {
491  bool wasEnabled = Body.Enabled;
492  Body.Enabled = Impassable || openState < 1.0f;
493  if (OutsideSubmarineFixture != null)
494  {
495  OutsideSubmarineFixture.CollidesWith = Body.Enabled ? SubmarineBody.CollidesWith : Category.None;
496  }
497  if (wasEnabled && !Body.Enabled && IsHorizontal)
498  {
499  //when opening a hatch, force characters above it to refresh the floor position
500  //(otherwise the character won't fall through the hatch until it moves)
501  foreach (Character c in Character.CharacterList)
502  {
503  if (c.WorldPosition.Y < item.WorldPosition.Y) { continue; }
504  if (c.WorldPosition.X < item.WorldRect.X || c.WorldPosition.X > item.WorldRect.Right) { continue; }
505  c.AnimController?.ForceRefreshFloorY();
506  }
507  }
508  }
509 
510  //don't use the predicted state here, because it might set
511  //other items to an incorrect state if the prediction is wrong
512  item.SendSignal(isOpen ? "1" : "0", "state_out");
513  }

659  private bool CheckSubmarinesInDoorWay()
660  {
661  if (linkedGap != null && linkedGap.IsRoomToRoom) { return false; }
662 
663  Rectangle doorRect = item.WorldRect;
664  if (IsHorizontal)
665  {
666  doorRect.Width = (int)(item.Rect.Width * (1.0f - openState));
667  }
668  else
669  {
670  doorRect.Height = (int)(item.Rect.Height * (1.0f - openState));
671  }
672 
673  foreach (Submarine sub in Submarine.Loaded)
674  {
675  if (sub == item.Submarine || sub.DockedTo.Contains(item.Submarine)) { continue; }
676  Rectangle worldBorders = sub.Borders;
677  worldBorders.Location += sub.WorldPosition.ToPoint();
678  if (!Submarine.RectsOverlap(worldBorders, doorRect)) { continue; }
679  foreach (Hull hull in sub.GetHulls(alsoFromConnectedSubs: false))
680  {
681  if (Submarine.RectsOverlap(hull.WorldRect, doorRect)) { return true; }
682  }
683  }
684  return false;

689  private void PushCharactersAway()
690  {
691  if (!MathUtils.IsValid(item.SimPosition))
692  {
693  if (!itemPosErrorShown)
694  {
695  DebugConsole.ThrowError("Failed to push a character out of a doorway - position of the door is not valid (" + item.SimPosition + ")");
696  GameAnalyticsManager.AddErrorEventOnce("PushCharactersAway:DoorPosInvalid", GameAnalyticsManager.ErrorSeverity.Error,
697  "Failed to push a character out of a doorway - position of the door is not valid (" + item.SimPosition + ").");
698  itemPosErrorShown = true;
699  }
700  return;
701  }
702 
703  Vector2 simPos = ConvertUnits.ToSimUnits(new Vector2(item.Rect.X, item.Rect.Y));
704 
705  Vector2 currSize = IsHorizontal ?
706  new Vector2(item.Rect.Width * (1.0f - openState), doorSprite.size.Y * item.Scale) :
707  new Vector2(doorSprite.size.X * item.Scale, item.Rect.Height * (1.0f - openState));
708  Vector2 simSize = ConvertUnits.ToSimUnits(currSize);
709 
710  foreach (Character c in Character.CharacterList)
711  {
712  if (!c.Enabled) { continue; }
713  if (c.SelectedItem?.GetComponent<Controller>() is { } controller && controller.IsAttachedUser(c)) { continue; }
714  if (!MathUtils.IsValid(c.SimPosition))
715  {
716  if (!characterPosErrorShown.Contains(c))
717  {
718  if (GameSettings.CurrentConfig.VerboseLogging) { DebugConsole.ThrowError("Failed to push a character out of a doorway - position of the character \"" + c.Name + "\" is not valid (" + c.SimPosition + ")"); }
719  GameAnalyticsManager.AddErrorEventOnce("PushCharactersAway:CharacterPosInvalid", GameAnalyticsManager.ErrorSeverity.Error,
720  "Failed to push a character out of a doorway - position of the character \"" + c.SpeciesName + "\" is not valid (" + c.SimPosition + ")." +
721  " Removed: " + c.Removed +
722  " Remoteplayer: " + c.IsRemotePlayer);
723  characterPosErrorShown.Add(c);
724  }
725  continue;
726  }
727  int dir = IsHorizontal ? Math.Sign(c.SimPosition.Y - item.SimPosition.Y) : Math.Sign(c.SimPosition.X - item.SimPosition.X);
728 
729  foreach (Limb limb in c.AnimController.Limbs)
730  {
731  if (limb.IsSevered) { continue; }
732  if (PushBodyOutOfDoorway(c, limb.body, dir, simPos, simSize) && damageSoundCooldown <= 0.0f)
733  {
734 #if CLIENT
735  SoundPlayer.PlayDamageSound("LimbBlunt", 1.0f, limb.body);
736 #endif
737  damageSoundCooldown = 0.5f;
738  }
739  }
740  PushBodyOutOfDoorway(c, c.AnimController.Collider, dir, simPos, simSize);
741  }

744  private bool PushBodyOutOfDoorway(Character c, PhysicsBody body, int dir, Vector2 doorRectSimPos, Vector2 doorRectSimSize)
745  {
746  if (!MathUtils.IsValid(body.SimPosition))
747  {
748  DebugConsole.ThrowError("Failed to push a limb out of a doorway - position of the body (character \"" + c.Name + "\") is not valid (" + body.SimPosition + ")");
749  GameAnalyticsManager.AddErrorEventOnce("PushCharactersAway:LimbPosInvalid", GameAnalyticsManager.ErrorSeverity.Error,
750  "Failed to push a character out of a doorway - position of the character \"" + c.SpeciesName + "\" is not valid (" + body.SimPosition + ")." +
751  " Removed: " + c.Removed +
752  " Remoteplayer: " + c.IsRemotePlayer);
753  return false;
754  }
755 
756  float diff;
757  if (IsHorizontal)
758  {
759  if (body.SimPosition.X < doorRectSimPos.X || body.SimPosition.X > doorRectSimPos.X + doorRectSimSize.X) { return false; }
760  diff = body.SimPosition.Y - item.SimPosition.Y;
761  }
762  else
763  {
764  if (body.SimPosition.Y > doorRectSimPos.Y || body.SimPosition.Y < doorRectSimPos.Y - doorRectSimSize.Y) { return false; }
765  diff = body.SimPosition.X - item.SimPosition.X;
766  }
767 
768  //if the limb is at a different side of the door than the character (collider),
769  //immediately teleport it to the correct side
770  if (Math.Sign(diff) != dir)
771  {
772  if (IsHorizontal)
773  {
774  body.SetTransform(new Vector2(body.SimPosition.X, item.SimPosition.Y + dir * doorRectSimSize.Y * 2.0f), body.Rotation);
775  }
776  else
777  {
778  body.SetTransform(new Vector2(item.SimPosition.X + dir * doorRectSimSize.X * 1.2f, body.SimPosition.Y), body.Rotation);
779  }
780  }
781 
782  //apply an impulse to push the limb further from the door
783  if (IsHorizontal)
784  {
785  if (Math.Abs(body.SimPosition.Y - item.SimPosition.Y) > doorRectSimSize.Y * 0.5f) { return false; }
786  body.ApplyLinearImpulse(new Vector2(isOpen ? 0.0f : 1.0f, dir * 2.0f), maxVelocity: NetConfig.MaxPhysicsBodyVelocity);
787  }
788  else
789  {
790  if (Math.Abs(body.SimPosition.X - item.SimPosition.X) > doorRectSimSize.X * 0.5f) { return false; }
791  body.ApplyLinearImpulse(new Vector2(dir * 2.0f, isOpen ? 0.0f : -1.0f), maxVelocity: NetConfig.MaxPhysicsBodyVelocity);
792  }
793 
794  //don't stun if the door was broken a moment ago
795  //otherwise enabling the door's collider and pushing the character away will interrupt repairing
796  if (lastBrokenTime < Timing.TotalTime - 1.0f)
797  {
798  c.SetStun(0.2f);
799  }
800  return true;
801  }
