242  || user.Removed
243  || !user.IsAnySelectedItem(item)

247  || (UsableIn == UseEnvironment.Air && user.AnimController.InWater))

255  return;
256  }

259  {
260  user.TeleportTo(item.WorldPosition);
261  user.AnimController.Collider.ResetDynamics();
262  foreach (var limb in user.AnimController.Limbs)

267  }
268

270

274 
275  if (user.AnimController.InWater)
276  {
277  if (diff.LengthSquared() > 30.0f * 30.0f)
278  {

280  user.AnimController.TargetDir = diff.X > 0.0f ? Direction.Right : Direction.Left;
281  }
282  else

284  user.AnimController.TargetMovement = Vector2.Zero;
285  UserInCorrectPosition = true;
286  }
287  }
288  else
289  {
290  // Secondary items (like ladders or chairs) will control the character position over primary items
291  // Only control the character position if the character doesn't have another secondary item already controlling it

300  return;
301  }
302  else if (Math.Abs(diff.X) > 0.1f)

308  }
309  }
310  else if (Math.Abs(diff.X) > 10.0f)
311  {
312  user.AnimController.TargetMovement = Vector2.Normalize(diff);

317  }

319  }
320  }

323 
324  if (limbPositions.Count == 0) { return; }

326  user.AnimController.StartUsingItem();
327 
328  if (user.SelectedItem != null)
329  {
330  user.AnimController.ResetPullJoints(l => l.IsLowerBody);
331  }
332  else
333  {
334  user.AnimController.ResetPullJoints();
335  }
336 
337  if (dir != 0) { user.AnimController.TargetDir = dir; }

339  foreach (LimbPos lb in limbPositions)
340  {
341  Limb limb = user.AnimController.GetLimb(lb.LimbType);
342  if (limb == null || !limb.body.Enabled) { continue; }
343  // Don't move lower body limbs if there's another selected secondary item that should control them
344  if (limb.IsLowerBody && user.HasSelectedAnotherSecondaryItem(Item)) { continue; }
345  // Don't move hands if there's a selected primary item that should control them
346  if (!limb.IsLowerBody && Item == user.SelectedSecondaryItem && user.SelectedItem != null) { continue; }
347  if (lb.AllowUsingLimb)
348  {
349  switch (lb.LimbType)
350  {
351  case LimbType.RightHand:
352  case LimbType.RightForearm:
353  case LimbType.RightArm:
354  if (user.Inventory.GetItemInLimbSlot(InvSlotType.RightHand) != null) { continue; }
355  break;
356  case LimbType.LeftHand:
357  case LimbType.LeftForearm:
358  case LimbType.LeftArm:
359  if (user.Inventory.GetItemInLimbSlot(InvSlotType.LeftHand) != null) { continue; }
360  break;
361  }
362  }
363  limb.Disabled = true;
364  Vector2 worldPosition = new Vector2(item.WorldRect.X, item.WorldRect.Y) + lb.Position * item.Scale;
365  Vector2 diff = worldPosition - limb.WorldPosition;
366  limb.PullJointEnabled = true;
