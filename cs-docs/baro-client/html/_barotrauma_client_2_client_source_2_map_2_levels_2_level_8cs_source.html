146  {
147  EventType eventType = (EventType)msg.ReadByte();
148 
149  switch (eventType)
150  {
151  case EventType.GlobalDestructibleWall:
152  {
153  foreach (LevelWall levelWall in ExtraWalls)

155  if (levelWall.Body.BodyType == BodyType.Static) { continue; }
156 
157  Vector2 bodyPos = new Vector2(
158  msg.ReadSingle(),
159  msg.ReadSingle());
160  levelWall.MoveState = msg.ReadRangedSingle(0.0f, MathHelper.TwoPi, 16);
161  if (Vector2.DistanceSquared(bodyPos, levelWall.Body.Position) > 0.5f
162  && !(levelWall is DestructibleLevelWall { Destroyed: true }))
163  {
164  levelWall.Body.SetTransformIgnoreContacts(ref bodyPos, levelWall.Body.Rotation);
165  }
166  }
167  }
168  break;
169  case EventType.SingleDestructibleWall:
170  {
171  int index = msg.ReadUInt16();
172  float damageByte = msg.ReadByte();
173  if (index < ExtraWalls.Count && ExtraWalls[index] is DestructibleLevelWall destructibleWall)
174  {
175  destructibleWall.SetDamage(destructibleWall.MaxHealth * damageByte / 255.0f);
176  }
177  }
178  break;
179  default:
180  throw new Exception($"Malformed incoming level event: {eventType} is not a supported event type");
181  }
