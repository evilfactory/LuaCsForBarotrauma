120  public bool PropEquals(object a, object b)
121  {
122  switch (typeString)
123  {
124  case "float":
125  if (a is not float fa) { return false; }
126  if (b is not float fb) { return false; }
127  return MathUtils.NearlyEqual(fa, fb);
128  case "int":
129  if (a is not int ia) { return false; }
130  if (b is not int ib) { return false; }
131  return ia == ib;
132  case "bool":
133  if (a is not bool ba) { return false; }
134  if (b is not bool bb) { return false; }
135  return ba == bb;
136  case "Enum":
137  if (a is not Enum ea) { return false; }
138  if (b is not Enum eb) { return false; }
139  return ea.Equals(eb);
140  default:
141  return ReferenceEquals(a,b)
142  || string.Equals(a?.ToString(), b?.ToString(), StringComparison.OrdinalIgnoreCase);
143  }

146  public void Read(IReadMessage msg)
147  {
148  int oldPos = msg.BitPosition;
149  UInt32 size = msg.ReadVariableUInt32();
150 
151  float x; float y; float z; float w;
152  byte r; byte g; byte b; byte a;
153  int ix; int iy; int width; int height;
154 
155  switch (typeString)
156  {
157  case "float":
158  if (size != 4) break;
159  property.SetValue(parentObject, msg.ReadSingle());
160  return;
161  case "int":
162  if (size != 4) break;
163  property.SetValue(parentObject, msg.ReadInt32());
164  return;
165  case "vector2":
166  if (size != 8) break;
167  x = msg.ReadSingle();
168  y = msg.ReadSingle();
169  property.SetValue(parentObject, new Vector2(x, y));
170  return;
171  case "vector3":
172  if (size != 12) break;
173  x = msg.ReadSingle();
174  y = msg.ReadSingle();
175  z = msg.ReadSingle();
176  property.SetValue(parentObject, new Vector3(x, y, z));
177  return;
178  case "vector4":
179  if (size != 16) break;
180  x = msg.ReadSingle();
181  y = msg.ReadSingle();
182  z = msg.ReadSingle();
183  w = msg.ReadSingle();
184  property.SetValue(parentObject, new Vector4(x, y, z, w));
185  return;
186  case "color":
187  if (size != 4) break;
188  r = msg.ReadByte();
189  g = msg.ReadByte();
190  b = msg.ReadByte();
191  a = msg.ReadByte();
192  property.SetValue(parentObject, new Color(r, g, b, a));
193  return;
194  case "rectangle":
195  if (size != 16) break;
196  ix = msg.ReadInt32();
197  iy = msg.ReadInt32();
198  width = msg.ReadInt32();
199  height = msg.ReadInt32();
200  property.SetValue(parentObject, new Rectangle(ix, iy, width, height));
201  return;
202  default:
203  msg.BitPosition = oldPos; //reset position to properly read the string
204  string incVal = msg.ReadString();
205  property.TrySetValue(parentObject, incVal);
206  return;
207  }
208 
209  //size didn't match: skip this
