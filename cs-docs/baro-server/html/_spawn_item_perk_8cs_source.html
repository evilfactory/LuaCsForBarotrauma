104  private SuitableContainers FindSuitableContainers(ItemPrefab prefab, Submarine submarine)
105  {

107  HashSet<ItemContainer> primaryContainers = new();
108  HashSet<ItemContainer> secondaryContainers = new();
109 
110  foreach (Item item in submarine.GetItems(alsoFromConnectedSubs: true))
111  {
112  if (item.GetComponent<Fabricator>() != null || item.GetComponent<Deconstructor>() != null) { continue; }
113  if (item.NonInteractable || item.NonPlayerTeamInteractable || item.IsHidden) { continue; }
114

181  static void AllocateContainers(ItemPrefab prefab, ICollection<ItemContainer> containers, ref int remaining, ref Dictionary<ItemContainer, int> containerAllocation)
