196  public void Update(float deltaTime)
197  {
198  if (ActiveVote == null) { return; }
199 
200  ActiveVote.Timer += deltaTime;
201 
202  var inGameClients = GameMain.Server.ConnectedClients.Where(c => c.InGame);
203  if (ActiveVote.Timer >= GameMain.NetworkMember.ServerSettings.VoteTimeout || inGameClients.Count() == 1)
204  {
205  var eligibleClients = inGameClients.Where(c => c != ActiveVote.VoteStarter);
206 
207  // Do not take unanswered into account for total
208  int yes = eligibleClients.Count(c => c.GetVote<int>(ActiveVote.VoteType) == 2);
209  int no = eligibleClients.Count(c => c.GetVote<int>(ActiveVote.VoteType) == 1);
210  int total = yes + no;
211 
212  bool passed = false;
213  //total can be zero if the client who initiated the vote has left
214  if (total > 0)
215  {
216  passed =
217  yes / (float)total >= GameMain.NetworkMember.ServerSettings.VoteRequiredRatio ||
218  inGameClients.Count() == 1;
219  }
220  ActiveVote.Finish(this, passed);
221  }

232  public void ServerRead(IReadMessage inc, Client sender, DoSProtection dosProtection)
233  {
234  if (GameMain.Server == null || sender == null) { return; }
235 
236  byte voteTypeByte = inc.ReadByte();
237  VoteType voteType = VoteType.Unknown;
238  try
239  {
240  voteType = (VoteType)voteTypeByte;
241  }
242  catch (Exception e)
243  {
244  DebugConsole.ThrowError("Failed to cast vote type \"" + voteTypeByte + "\"", e);
245  return;
246  }
247 
248  switch (voteType)
249  {
250  case VoteType.Sub:
251  int equalityCheckVal = inc.ReadInt32();
252  string hash = equalityCheckVal > 0 ? string.Empty : inc.ReadString();
253  SubmarineInfo sub = equalityCheckVal > 0 ?
254  SubmarineInfo.SavedSubmarines.FirstOrDefault(s => s.Type == SubmarineType.Player && s.EqualityCheckVal == equalityCheckVal) :
255  SubmarineInfo.SavedSubmarines.FirstOrDefault(s => s.Type == SubmarineType.Player && s.MD5Hash.StringRepresentation == hash);
256  sender.SetVote(voteType, sub);
257  break;
258  case VoteType.Mode:
259  string modeIdentifier = inc.ReadString();
260  GameModePreset mode = GameModePreset.List.Find(gm => gm.Identifier == modeIdentifier);
261  if (mode == null || !mode.Votable) { break; }
262  var prevHighestVoted = HighestVoted<GameModePreset>(VoteType.Mode, GameMain.Server.ConnectedClients);
263  sender.SetVote(voteType, mode);
264  var newHighestVoted = HighestVoted<GameModePreset>(VoteType.Mode, GameMain.Server.ConnectedClients);
265  if (prevHighestVoted != newHighestVoted)
266  {
267  GameMain.NetLobbyScreen.SelectedModeIdentifier = mode.Identifier;
268  GameMain.NetLobbyScreen.LastUpdateID++;
269  }
270  break;
271  case VoteType.EndRound:
272  if (!sender.HasSpawned) { return; }
273  sender.SetVote(voteType, inc.ReadBoolean());
274  break;
275  case VoteType.Kick:
276  byte kickedClientID = inc.ReadByte();
277  if ((DateTime.Now - sender.JoinTime).TotalSeconds < GameMain.Server.ServerSettings.DisallowKickVoteTime)
278  {
279  GameMain.Server.SendDirectChatMessage($"ServerMessage.kickvotedisallowed", sender);
280  }
281  else
282  {
283  Client kicked = GameMain.Server.ConnectedClients.Find(c => c.SessionId == kickedClientID);
284  if (kicked != null && kicked.Connection != GameMain.Server.OwnerConnection && !kicked.HasKickVoteFrom(sender))
285  {
286  kicked.AddKickVote(sender);
287  Client.UpdateKickVotes(GameMain.Server.ConnectedClients);
288  GameMain.Server.SendChatMessage($"ServerMessage.HasVotedToKick~[initiator]={sender.Name}~[target]={kicked.Name}", ChatMessageType.Server, null);
289  }
290  }
291  break;
292  case VoteType.StartRound:
293  bool ready = inc.ReadBoolean();
294  if (ready != sender.GetVote<bool>(VoteType.StartRound))
295  {
296  sender.SetVote(VoteType.StartRound, ready);
297  GameServer.Log(GameServer.ClientLogName(sender) + (ready ? " is ready to start the game." : " is not ready to start the game."), ServerLog.MessageType.ServerMessage);
298  }
299  break;
300  case VoteType.PurchaseAndSwitchSub:
301  case VoteType.PurchaseSub:
302  case VoteType.SwitchSub:
303  case VoteType.TransferMoney:
304  bool startVote = inc.ReadBoolean();
305  if (startVote)
306  {
307  if (voteType == VoteType.TransferMoney)
308  {
309  int amount = inc.ReadInt32();
310  int fromClientId = inc.ReadByte();
311  int toClientId = inc.ReadByte();
312  if (!ShouldRejectVote(sender, voteType))
313  {
314  pendingVotes.Enqueue(new TransferVote(sender,
315  GameMain.Server.ConnectedClients.Find(c => c.SessionId == fromClientId),
316  amount,
317  GameMain.Server.ConnectedClients.Find(c => c.SessionId == toClientId)));
318  }
319  }
320  else
321  {
322  string subName = inc.ReadString();
323  SubmarineInfo subInfo = GameMain.GameSession.OwnedSubmarines.FirstOrDefault(s => s.Name == subName) ?? SubmarineInfo.SavedSubmarines.FirstOrDefault(s => s.Name == subName);
324  bool transferItems = inc.ReadBoolean();
325  if (!ShouldRejectVote(sender, voteType))
326  {
327  if (GameMain.GameSession?.Campaign is MultiPlayerCampaign campaign &&
328  (campaign.CanPurchaseSub(subInfo, sender) || GameMain.GameSession.IsSubmarineOwned(subInfo)))
329  {
330  StartSubmarineVote(subInfo, transferItems, voteType, sender);
331  }
332  }
333  }
334  }
335  else
336  {
337  sender.SetVote(voteType, (int)inc.ReadByte());
338  }
339  break;
340  case VoteType.Traitor:
341  int clientId = inc.ReadInt32();
342  if (sender.InGame && sender.Character != null)
343  {
344  var client = GameMain.Server.ConnectedClients.FirstOrDefault(c => c.SessionId == clientId);
345  sender.SetVote(voteType, client);
346  if (client?.Character != null)
347  {
348  string msg = TextManager.GetWithVariable("traitor.blamebutton.dialog", "[name]", client.Character.DisplayName).Value;
349  ChatMessage.HandleSpamFilter(sender, msg, out bool flaggedAsSpam);
350  if (!flaggedAsSpam)
351  {
352  GameMain.Server.SendChatMessage(
353  msg,
354  ChatMessageType.Radio, senderClient: sender, senderCharacter: sender.Character);
355  sender.LastSentChatMessages.Add(msg);
356  }
357  }
358  }
359  break;
360  }
361 
362  inc.ReadPadBits();
363 
364  using (dosProtection.Pause(sender))
365  {
366  GameMain.Server.UpdateVoteStatus();
367  }
368  }

370  public void ServerWrite(IWriteMessage msg)
371  {
372  if (GameMain.Server == null) { return; }
373 
374  msg.WriteBoolean(GameMain.Server.ServerSettings.AllowSubVoting);
375  if (GameMain.Server.ServerSettings.AllowSubVoting)
376  {
377  IReadOnlyDictionary<SubmarineInfo, int> voteList = GetVoteCounts<SubmarineInfo>(VoteType.Sub, GameMain.Server.ConnectedClients);
378  msg.WriteByte((byte)voteList.Count);
379  foreach (KeyValuePair<SubmarineInfo, int> vote in voteList)
380  {
381  msg.WriteByte((byte)vote.Value);
382  msg.WriteString(vote.Key.Name);
383  }
384  }
385  msg.WriteBoolean(GameMain.Server.ServerSettings.AllowModeVoting);
386  if (GameMain.Server.ServerSettings.AllowModeVoting)
387  {
388  IReadOnlyDictionary<GameModePreset, int> voteList = GetVoteCounts<GameModePreset>(VoteType.Mode, GameMain.Server.ConnectedClients);
389  msg.WriteByte((byte)voteList.Count);
390  foreach (KeyValuePair<GameModePreset, int> vote in voteList)
391  {
392  msg.WriteByte((byte)vote.Value);
393  msg.WriteIdentifier(vote.Key.Identifier);
394  }
395  }
396  msg.WriteBoolean(GameMain.Server.ServerSettings.AllowEndVoting);
397  if (GameMain.Server.ServerSettings.AllowEndVoting)
398  {
399  msg.WriteByte((byte)GameMain.Server.ConnectedClients.Count(c => c.HasSpawned && c.GetVote<bool>(VoteType.EndRound)));
400  msg.WriteByte((byte)GameMain.Server.ConnectedClients.Count(c => c.HasSpawned));
401  }
402 
403  msg.WriteBoolean(GameMain.Server.ServerSettings.AllowVoteKick);
404 
405  msg.WriteByte((byte)(ActiveVote?.State ?? VoteState.None));
406  if (ActiveVote != null)
407  {
408  msg.WriteByte((byte)ActiveVote.VoteType);
409  if (ActiveVote.State != VoteState.None && ActiveVote.VoteType != VoteType.Unknown)
410  {
411  var eligibleClients = GameMain.Server.ConnectedClients.Where(c => c.InGame && c != ActiveVote.VoteStarter);
412 
413  var yesClients = eligibleClients.Where(c => c.GetVote<int>(ActiveVote.VoteType) == 2);
414  msg.WriteByte((byte)yesClients.Count());
415  foreach (Client c in yesClients)
416  {
417  msg.WriteByte(c.SessionId);
418  }
419 
420  var noClients = eligibleClients.Where(c => c.GetVote<int>(ActiveVote.VoteType) == 1);
421  msg.WriteByte((byte)noClients.Count());
422  foreach (Client c in noClients)
423  {
424  msg.WriteByte(c.SessionId);
425  }
426 
427  msg.WriteByte((byte)eligibleClients.Count());
428 
429  switch (ActiveVote.State)
430  {
431  case VoteState.Started:
432  msg.WriteByte(ActiveVote.VoteStarter.SessionId);
433  msg.WriteByte((byte)GameMain.Server.ServerSettings.VoteTimeout);
434 
435  switch (ActiveVote.VoteType)
436  {
437  case VoteType.PurchaseSub:
438  case VoteType.PurchaseAndSwitchSub:
439  case VoteType.SwitchSub:
440  SubmarineVote vote = ActiveVote as SubmarineVote;
441  msg.WriteString(vote.Sub.Name);
442  msg.WriteBoolean(vote.TransferItems);
443  break;
444  case VoteType.TransferMoney:
445  var transferVote = (ActiveVote as TransferVote);
446  msg.WriteByte(transferVote.From?.SessionId ?? 0);
447  msg.WriteByte(transferVote.To?.SessionId ?? 0);
448  msg.WriteInt32(transferVote.TransferAmount);
449  break;
450  }
451 
452  break;
453  case VoteState.Running:
454  // Nothing specific
455  break;
456  case VoteState.Passed:
457  case VoteState.Failed:
458  msg.WriteBoolean(ActiveVote.State == VoteState.Passed);
459  switch (ActiveVote.VoteType)
460  {
461  case VoteType.PurchaseSub:
462  case VoteType.PurchaseAndSwitchSub:
463  case VoteType.SwitchSub:
464  var subVote = ActiveVote as SubmarineVote;
465  msg.WriteString(subVote.Sub.Name);
466  msg.WriteBoolean(subVote.TransferItems);
467  break;
468  }
469  break;
470  }
471  }
472  }
473 
474  var readyClients = GameMain.Server.ConnectedClients.Where(c => c.GetVote<bool>(VoteType.StartRound));
475  msg.WriteByte((byte)readyClients.Count());
476  foreach (Client c in readyClients)
477  {
478  msg.WriteByte(c.SessionId);
479  }
480 
481  msg.WritePadBits();
482  }
483  }
