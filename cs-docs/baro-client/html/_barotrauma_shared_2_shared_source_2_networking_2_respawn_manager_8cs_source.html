430  public Vector2 FindSpawnPos(Submarine respawnShuttle, Submarine mainSub)
431  {
432  if (Level.Loaded == null || Submarine.MainSub == null) { return Vector2.Zero; }
433 
434  Rectangle dockedBorders = respawnShuttle.GetDockedBorders();
435  Vector2 diffFromDockedBorders =
436  new Vector2(dockedBorders.Center.X, dockedBorders.Y - dockedBorders.Height / 2)
437  - new Vector2(respawnShuttle.Borders.Center.X, respawnShuttle.Borders.Y - respawnShuttle.Borders.Height / 2);
438 
439  int minWidth = Math.Max(dockedBorders.Width, 1000);
440  int minHeight = Math.Max(dockedBorders.Height, 1000);
441 
442  List<Level.InterestingPosition> potentialSpawnPositions = FindValidSpawnPoints(respawnShuttle, minWidth, minHeight, minDistFromSubs: 10000.0f, minDistFromCharacters: 5000.0f);
443  if (potentialSpawnPositions.None())
444  {
445  DebugConsole.NewMessage("Failed to find a shuttle spawn position far away from submarines and characters, attempting to find one closer to to subs and characters...");
446  potentialSpawnPositions = FindValidSpawnPoints(respawnShuttle, minWidth, minHeight, minDistFromSubs: 1000.0f, minDistFromCharacters: 500.0f);
447  if (potentialSpawnPositions.None())
448  {
449  DebugConsole.NewMessage("Failed to find a shuttle spawn position, using the level's start position instead.");
450  return Level.Loaded.StartPosition;
451  }
452  }
453 
454  Vector2 bestSpawnPos = Level.Loaded.StartPosition;
455  float bestSpawnPosValue = 0.0f;
456  foreach (var potentialSpawnPos in potentialSpawnPositions)
457  {
458  //the closer the spawnpos is to the main sub, the better
459  float spawnPosValue = 100000.0f / Math.Max(Vector2.Distance(potentialSpawnPos.Position.ToVector2(), mainSub.WorldPosition), 1.0f);
460 
461  //prefer spawnpoints that are at the left side of the sub (so the shuttle doesn't have to go backwards)
462  if (potentialSpawnPos.Position.X > mainSub.WorldPosition.X)
463  {

465  }
466 
467  if (spawnPosValue > bestSpawnPosValue)
468  {
469  bestSpawnPos = potentialSpawnPos.Position.ToVector2();
470  bestSpawnPosValue = spawnPosValue;
471  }
472  }
473 
474  return bestSpawnPos;

477  private List<Level.InterestingPosition> FindValidSpawnPoints(Submarine respawnShuttle, float minWidth, float minHeight, float minDistFromSubs, float minDistFromCharacters)
478  {
479  List<Level.InterestingPosition> potentialSpawnPositions = new List<Level.InterestingPosition>();
480  foreach (Level.InterestingPosition potentialSpawnPos in Level.Loaded.PositionsOfInterest.Where(p => p.PositionType == Level.PositionType.MainPath))
481  {
482  bool invalid = false;
483  //make sure the shuttle won't overlap with any ruins
484  foreach (var ruin in Level.Loaded.Ruins)
485  {
486  if (Math.Abs(ruin.Area.Center.X - potentialSpawnPos.Position.X) < (minWidth + ruin.Area.Width) / 2) { invalid = true; break; }
487  if (Math.Abs(ruin.Area.Center.Y - potentialSpawnPos.Position.Y) < (minHeight + ruin.Area.Height) / 2) { invalid = true; break; }
488  }
489  if (invalid) { continue; }
490 
491  //make sure there aren't any walls too close
492  var tooCloseCells = Level.Loaded.GetTooCloseCells(potentialSpawnPos.Position.ToVector2(), Math.Max(minWidth, minHeight));
493  if (tooCloseCells.Any()) { continue; }
494 
495  //make sure the spawnpoint is far enough from other subs
496  foreach (Submarine sub in Submarine.Loaded)
497  {
498  if (sub == respawnShuttle || respawnShuttle.DockedTo.Contains(sub)) { continue; }
499  float minDist = Math.Max(Math.Max(minWidth, minHeight) + Math.Max(sub.Borders.Width, sub.Borders.Height), minDistFromSubs);
500  if (Vector2.DistanceSquared(sub.WorldPosition, potentialSpawnPos.Position.ToVector2()) < minDist * minDist)
501  {
502  invalid = true;
503  break;
504  }
505  }
506  if (invalid) { continue; }
507 
508  foreach (Character character in Character.CharacterList)
509  {
510  if (character.IsDead)
511  {
512  //cannot spawn directly over dead bodies
513  if (Math.Abs(character.WorldPosition.X - potentialSpawnPos.Position.X) < minWidth) { invalid = true; break; }
514  if (Math.Abs(character.WorldPosition.Y - potentialSpawnPos.Position.Y) < minHeight) { invalid = true; break; }
515  }
516  else
517  {
518  //cannot spawn near alive characters (to prevent other players from seeing the shuttle
519  //appear out of nowhere, or monsters from immediatelly wrecking the shuttle)
520  if (Vector2.DistanceSquared(character.WorldPosition, potentialSpawnPos.Position.ToVector2()) < minDistFromCharacters * minDistFromCharacters)
521  {
522  invalid = true;
523  break;
524  }
525  }
526  }
527  if (invalid) { continue; }
528 
529  potentialSpawnPositions.Add(potentialSpawnPos);
530  }
531  return potentialSpawnPositions;
532  }
533  }
