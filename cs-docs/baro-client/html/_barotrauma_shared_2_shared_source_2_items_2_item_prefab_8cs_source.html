201  public FabricationRecipe(XElement element, Identifier itemPrefab)
202  {
203  TargetItemPrefabIdentifier = itemPrefab;
204  var displayNameIdentifier = element.GetAttributeIdentifier("displayname", "");
205  displayName = new Lazy<LocalizedString>(() => displayNameIdentifier.IsEmpty
206  ? TargetItem.Name
207  : TextManager.GetWithVariable($"DisplayName.{displayNameIdentifier}", "[itemname]", TargetItem.Name));
208 
209  SuitableFabricatorIdentifiers = element.GetAttributeIdentifierArray("suitablefabricators", Array.Empty<Identifier>()).ToImmutableArray();
210 
211  var requiredSkills = new List<Skill>();
212  RequiredTime = element.GetAttributeFloat("requiredtime", 1.0f);
213  RequiredMoney = element.GetAttributeInt("requiredmoney", 0);
214  OutCondition = element.GetAttributeFloat("outcondition", 1.0f);
215  if (OutCondition > 1.0f)
216  {
217  DebugConsole.AddWarning($"Error in \"{itemPrefab}\"'s fabrication recipe: out condition is above 100% ({OutCondition * 100}).");
218  }
219  var requiredItems = new List<RequiredItem>();
220  RequiresRecipe = element.GetAttributeBool("requiresrecipe", false);
221  Amount = element.GetAttributeInt("amount", 1);
222 
223  int limitDefault = element.GetAttributeInt("fabricationlimit", -1);
224  FabricationLimitMin = element.GetAttributeInt(nameof(FabricationLimitMin), limitDefault);
225  FabricationLimitMax = element.GetAttributeInt(nameof(FabricationLimitMax), limitDefault);
226 
227  if (element.GetAttribute(nameof(Quality)) != null)
228  {
229  Quality = element.GetAttributeInt(nameof(Quality), 0);
230  }
231 
232  foreach (var subElement in element.Elements())
233  {
234  switch (subElement.Name.ToString().ToLowerInvariant())
235  {
236  case "requiredskill":
237  if (subElement.Attribute("name") != null)
238  {
239  DebugConsole.ThrowError("Error in fabricable item " + itemPrefab + "! Use skill identifiers instead of names.");
240  continue;
241  }
242 
243  requiredSkills.Add(new Skill(
244  subElement.GetAttributeIdentifier("identifier", ""),
245  subElement.GetAttributeInt("level", 0)));
246  break;
247  case "item":
248  case "requireditem":
249  Identifier requiredItemIdentifier = subElement.GetAttributeIdentifier("identifier", Identifier.Empty);
250  Identifier requiredItemTag = subElement.GetAttributeIdentifier("tag", Identifier.Empty);
251  if (requiredItemIdentifier == Identifier.Empty && requiredItemTag == Identifier.Empty)
252  {
253  DebugConsole.ThrowError("Error in fabricable item " + itemPrefab + "! One of the required items has no identifier or tag.");
254  continue;
255  }
256 
257  float minCondition = subElement.GetAttributeFloat("mincondition", 1.0f);
258  float maxCondition = subElement.GetAttributeFloat("maxcondition", 1.0f);
259  //Substract mincondition from required item's condition or delete it regardless?
260  bool useCondition = subElement.GetAttributeBool("usecondition", true);
261  int amount = subElement.GetAttributeInt("count", subElement.GetAttributeInt("amount", 1));
262 
263  if (requiredItemIdentifier != Identifier.Empty)
264  {
265  var existing = requiredItems.FindIndex(r =>
266  r is RequiredItemByIdentifier ri &&
267  ri.ItemPrefabIdentifier == requiredItemIdentifier &&
268  MathUtils.NearlyEqual(r.MinCondition, minCondition) &&
269  MathUtils.NearlyEqual(r.MaxCondition, maxCondition));
270  if (existing >= 0)
271  {
272  amount += requiredItems[existing].Amount;
273  requiredItems.RemoveAt(existing);
274  }
275  requiredItems.Add(new RequiredItemByIdentifier(requiredItemIdentifier, amount, minCondition, maxCondition, useCondition));
276  }
277  else
278  {
279  var existing = requiredItems.FindIndex(r =>
280  r is RequiredItemByTag rt &&
281  rt.Tag == requiredItemTag &&
282  MathUtils.NearlyEqual(r.MinCondition, minCondition) &&
283  MathUtils.NearlyEqual(r.MaxCondition, maxCondition));
284  if (existing >= 0)
285  {
286  amount += requiredItems[existing].Amount;
287  requiredItems.RemoveAt(existing);
288  }
289  requiredItems.Add(new RequiredItemByTag(requiredItemTag, amount, minCondition, maxCondition, useCondition));
290  }
291  break;
292  }
293  }
294 
295  this.RequiredSkills = requiredSkills.ToImmutableArray();
296  this.RequiredItems = requiredItems.ToImmutableArray();
297 
298  RecipeHash = GenerateHash();
299  }

301  private uint GenerateHash()
302  {
303  using var md5 = MD5.Create();
304  uint outputId = ToolBox.IdentifierToUint32Hash(TargetItemPrefabIdentifier, md5);
305 
306  var requiredItems = string.Join(':', RequiredItems
307  .Select(i => i.UintIdentifier)
308  .Select(i => string.Join(',', i)));
309 
310  var requiredSkills = string.Join(':', RequiredSkills.Select(s => $"{s.Identifier}:{s.Level}"));
311 
312  uint retVal = ToolBox.StringToUInt32Hash($"{Amount}|{outputId}|{RequiredTime}|{requiredItems}|{requiredSkills}", md5);
313  if (retVal == 0) { retVal = 1; }
314  return retVal;
315  }
