375  public override void AddExtraMissions(LevelData levelData)
376  {
377  if (levelData == null)
378  {
379  throw new ArgumentException("Level data was null.");
380  }
381 
382  extraMissions.Clear();
383 
384  var currentLocation = Map.CurrentLocation;
385  if (currentLocation == null)
386  {
387  throw new InvalidOperationException("Current location was null.");
388  }
389  if (levelData.Type == LevelData.LevelType.Outpost)
390  {
391  //if there's an available mission that takes place in the outpost, select it
392  foreach (var availableMission in currentLocation.AvailableMissions)
393  {
394  if (availableMission.Locations[0] == currentLocation && availableMission.Locations[1] == currentLocation)
395  {
396  currentLocation.SelectMission(availableMission);
397  }
398  }
399  }
400  else
401  {
402  foreach (Mission mission in currentLocation.SelectedMissions.ToList())
403  {
404  //if we had selected a mission that takes place in the outpost, deselect it when leaving the outpost
405  if (mission.Locations[0] == currentLocation &&
406  mission.Locations[1] == currentLocation)
407  {
408  currentLocation.DeselectMission(mission);
409  }
410  }
411  if (levelData.HasBeaconStation && !levelData.IsBeaconActive && Missions.None(m => m.Prefab.Type == Tags.MissionTypeBeacon))
412  {
413  var beaconMissionPrefabs = MissionPrefab.Prefabs.Where(m => m.IsSideObjective && m.Type == Tags.MissionTypeBeacon);
414  if (beaconMissionPrefabs.Any())
415  {
416  var filteredMissions = beaconMissionPrefabs.Where(m => levelData.Difficulty >= m.MinLevelDifficulty && levelData.Difficulty <= m.MaxLevelDifficulty);
417  if (filteredMissions.None())
418  {
419  DebugConsole.AddWarning($"No suitable beacon mission found matching the level difficulty {levelData.Difficulty}. Ignoring the restriction.");
420  }
421  else
422  {
423  beaconMissionPrefabs = filteredMissions;
424  }
425  Random rand = new MTRandom(ToolBox.StringToInt(levelData.Seed));
426  var beaconMissionPrefab = ToolBox.SelectWeightedRandom(beaconMissionPrefabs, p => p.Commonness, rand);
427  extraMissions.Add(beaconMissionPrefab.Instantiate(Map.SelectedConnection.Locations, Submarine.MainSub));
428  }
429  }
430  if (levelData.HasHuntingGrounds)
431  {
432  var huntingGroundsMissionPrefabs = MissionPrefab.Prefabs.Where(m => m.IsSideObjective && m.Tags.Contains("huntinggrounds")).OrderBy(m => m.UintIdentifier);
433  if (!huntingGroundsMissionPrefabs.Any())
434  {
435  DebugConsole.AddWarning("Could not find a hunting grounds mission for the level. No mission with the tag \"huntinggrounds\" found.");
436  }
437  else
438  {
439  Random rand = new MTRandom(ToolBox.StringToInt(levelData.Seed));
440  // Adjust the prefab commonness based on the difficulty tag
441  var prefabs = huntingGroundsMissionPrefabs.ToList();
442  var weights = prefabs.Select(p => (float)Math.Max(p.Commonness, 1)).ToList();
443  for (int i = 0; i < prefabs.Count; i++)
444  {
445  var prefab = prefabs[i];
446  var weight = weights[i];
447  if (prefab.Tags.Contains("easy"))
448  {
449  weight *= MathHelper.Lerp(0.2f, 2f, MathUtils.InverseLerp(80, LevelData.HuntingGroundsDifficultyThreshold, levelData.Difficulty));
450  }
451  else if (prefab.Tags.Contains("hard"))
452  {
453  weight *= MathHelper.Lerp(0.5f, 1.5f, MathUtils.InverseLerp(LevelData.HuntingGroundsDifficultyThreshold + 10, 80, levelData.Difficulty));
454  }
455  weights[i] = weight;
456  }
457  var huntingGroundsMissionPrefab = ToolBox.SelectWeightedRandom(prefabs, weights, rand);
458  if (!Missions.Any(m => m.Prefab.Tags.Contains("huntinggrounds")))
459  {
460  extraMissions.Add(huntingGroundsMissionPrefab.Instantiate(Map.SelectedConnection.Locations, Submarine.MainSub));
461  }
462  }
463  }
464  foreach (Faction faction in factions.OrderBy(f => f.Prefab.MenuOrder))
465  {
466  foreach (var automaticMission in faction.Prefab.AutomaticMissions)
467  {
468  if (faction.Reputation.Value < automaticMission.MinReputation || faction.Reputation.Value > automaticMission.MaxReputation) { continue; }
469 
470  if (automaticMission.DisallowBetweenOtherFactionOutposts && levelData.Type == LevelData.LevelType.LocationConnection)
471  {
472  if (Map.SelectedConnection.Locations.All(l => l.Faction != null && l.Faction != faction))
473  {
474  continue;
475  }
476  }
477  if (automaticMission.MaxDistanceFromFactionOutpost < int.MaxValue)
478  {
479  if (!Map.LocationOrConnectionWithinDistance(
480  currentLocation,
481  automaticMission.MaxDistanceFromFactionOutpost,
482  loc => loc.Faction == faction))
483  {
484  continue;
485  }
486  }
487  Random rand = new MTRandom(ToolBox.StringToInt(levelData.Seed + TotalPassedLevels));
488  if (levelData.Type != automaticMission.LevelType) { continue; }
489  float probability =
490  MathHelper.Lerp(
491  automaticMission.MinProbability,
492  automaticMission.MaxProbability,
493  MathUtils.InverseLerp(automaticMission.MinReputation, automaticMission.MaxReputation, faction.Reputation.Value));
494  if (rand.NextDouble() < probability)
495  {
496  var missionPrefabs = MissionPrefab.Prefabs.Where(m => m.Tags.Any(t => t == automaticMission.MissionTag)).OrderBy(m => m.UintIdentifier);
497  if (missionPrefabs.Any())
498  {
499  var missionPrefab = ToolBox.SelectWeightedRandom(missionPrefabs, p => p.Commonness, rand);
500  if (missionPrefab.Type == Tags.MissionTypePirate && Missions.Any(m => m.Prefab.Type == Tags.MissionTypePirate))
501  {
502  continue;
503  }
504  if (automaticMission.LevelType == LevelData.LevelType.Outpost)
505  {
506  extraMissions.Add(missionPrefab.Instantiate(new Location[] { currentLocation, currentLocation }, Submarine.MainSub));
507  }
508  else
509  {
510  extraMissions.Add(missionPrefab.Instantiate(Map.SelectedConnection.Locations, Submarine.MainSub));
511  }
512  }
513  }
514  }
515  }
516  }
517  if (levelData.Biome.IsEndBiome)
518  {
519  Identifier endMissionTag = Identifier.Empty;
520  if (levelData.Type == LevelData.LevelType.LocationConnection)
521  {
522  int locationIndex = map.EndLocations.IndexOf(map.SelectedLocation);
523  if (locationIndex > -1)
524  {
525  endMissionTag = ("endlevel_locationconnection_" + locationIndex).ToIdentifier();
526  }
527  }
528  else
529  {
530  int locationIndex = map.EndLocations.IndexOf(map.CurrentLocation);
531  if (locationIndex > -1)
532  {
533  endMissionTag = ("endlevel_location_" + locationIndex).ToIdentifier();
534  }
535  }
536  if (!endMissionTag.IsEmpty)
537  {
538  var endLevelMissionPrefabs = MissionPrefab.Prefabs.Where(m => m.Tags.Contains(endMissionTag)).OrderBy(m => m.UintIdentifier);
539  if (endLevelMissionPrefabs.Any())
540  {
541  Random rand = new MTRandom(ToolBox.StringToInt(levelData.Seed));
542  var endLevelMissionPrefab = ToolBox.SelectWeightedRandom(endLevelMissionPrefabs, p => p.Commonness, rand);
543  if (Missions.All(m => m.Prefab.Type != endLevelMissionPrefab.Type))
544  {
545  if (levelData.Type == LevelData.LevelType.LocationConnection)
546  {
547  extraMissions.Add(endLevelMissionPrefab.Instantiate(map.SelectedConnection.Locations, Submarine.MainSub));
548  }
549  else
550  {
551  extraMissions.Add(endLevelMissionPrefab.Instantiate(new Location[] { map.CurrentLocation, map.CurrentLocation }, Submarine.MainSub));
552  }
553  }
554  }
555  }

633  public TransitionType GetAvailableTransition(out LevelData nextLevel, out Submarine leavingSub)

647  }

649  //currently travelling from location to another
