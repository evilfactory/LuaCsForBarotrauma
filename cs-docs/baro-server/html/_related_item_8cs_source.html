99  public bool MatchesItem(Item item)
100  {
101  if (item == null) { return false; }
102  if (ExcludedIdentifiers.Contains(item.Prefab.Identifier)) { return false; }
103  foreach (var excludedIdentifier in ExcludedIdentifiers)
104  {
105  if (item.HasTag(excludedIdentifier)) { return false; }
106  }
107  if (Identifiers.Contains(item.Prefab.Identifier)) { return true; }
108  foreach (var identifier in Identifiers)
109  {
110  if (item.HasTag(identifier)) { return true; }
111  }
112  if (AllowVariants && !item.Prefab.VariantOf.IsEmpty)
113  {
114  if (Identifiers.Contains(item.Prefab.VariantOf)) { return true; }
115  }
116  return false;
117  }

146  public bool CheckRequirements(Character character, Item parentItem)
147  {
148  switch (type)
149  {
150  case RelationType.Contained:
151  if (parentItem == null) { return false; }
152  return CheckContained(parentItem);
153  case RelationType.Container:
154  if (parentItem == null || parentItem.Container == null) { return MatchOnEmpty || RequireEmpty; }
155  return CheckItem(parentItem.Container, this);
156  case RelationType.Equipped:
157  if (character == null) { return false; }
158  var heldItems = character.HeldItems;
159  if ((RequireEmpty || MatchOnEmpty) && heldItems.None()) { return true; }
160  foreach (Item equippedItem in heldItems)
161  {
162  if (equippedItem == null) { continue; }
163  if (CheckItem(equippedItem, this))
164  {
165  if (RequireEmpty && equippedItem.Condition > 0) { return false; }
166  return true;
167  }
168  }
169  break;
170  case RelationType.Picked:
171  if (character == null) { return false; }
172  if (character.Inventory == null) { return MatchOnEmpty || RequireEmpty; }
173  var allItems = character.Inventory.AllItems;
174  if ((RequireEmpty || MatchOnEmpty) && allItems.None()) { return true; }
175  foreach (Item pickedItem in allItems)
176  {
177  if (pickedItem == null) { continue; }
178  if (CheckItem(pickedItem, this))
179  {
180  if (RequireEmpty && pickedItem.Condition > 0) { return false; }
181  return true;
182  }
183  }
184  break;
185  default:
186  return true;
187  }

189  static bool CheckItem(Item i, RelatedItem ri) => (!ri.ExcludeBroken || ri.RequireEmpty || i.Condition > 0.0f) && (!ri.ExcludeFullCondition || !i.IsFullCondition) && ri.MatchesItem(i);
190 
191  return false;
192  }

194  private bool CheckContained(Item parentItem)
195  {
196  if (parentItem.OwnInventory == null) { return false; }
197  bool isEmpty = parentItem.OwnInventory.IsEmpty();
198  if (RequireEmpty && !isEmpty) { return false; }
199  if (MatchOnEmpty && isEmpty) { return true; }
200  foreach (var container in parentItem.GetComponents<Items.Components.ItemContainer>())
201  {
202  foreach (Item contained in container.Inventory.AllItems)
203  {
204  if (TargetSlot > -1 && parentItem.OwnInventory.FindIndex(contained) != TargetSlot) { continue; }
205  if ((!ExcludeBroken || contained.Condition > 0.0f) && (!ExcludeFullCondition || !contained.IsFullCondition) && MatchesItem(contained)) { return true; }
206  if (CheckContained(contained)) { return true; }
207  }
208  }
209  return false;
210  }

244  public static RelatedItem Load(ContentXElement element, bool returnEmpty, string parentDebugName)
245  {
246  Identifier[] identifiers;
247  if (element.GetAttribute("name") != null)
248  {
249  //backwards compatibility + a console warning
250  DebugConsole.ThrowError("Error in RelatedItem config (" + (string.IsNullOrEmpty(parentDebugName) ? element.ToString() : parentDebugName) + ") - use item tags or identifiers instead of names.");
251  Identifier[] itemNames = element.GetAttributeIdentifierArray("name", Array.Empty<Identifier>());
252  //attempt to convert to identifiers and tags
253  List<Identifier> convertedIdentifiers = new List<Identifier>();
254  foreach (Identifier itemName in itemNames)
255  {
256  var matchingItem = ItemPrefab.Prefabs.Find(me => me.Name == itemName.Value);
257  if (matchingItem != null)
258  {
259  convertedIdentifiers.Add(matchingItem.Identifier);
260  }
261  else
262  {
263  //no matching item found, this must be a tag
264  convertedIdentifiers.Add(itemName);
265  }
266  }
267  identifiers = convertedIdentifiers.ToArray();
268  }
269  else
270  {
271  identifiers = element.GetAttributeIdentifierArray("items", null) ?? element.GetAttributeIdentifierArray("item", null);
272  if (identifiers == null)
273  {
274  identifiers = element.GetAttributeIdentifierArray("identifiers", null) ?? element.GetAttributeIdentifierArray("tags", null);
275  if (identifiers == null)
276  {
277  identifiers = element.GetAttributeIdentifierArray("identifier", null) ?? element.GetAttributeIdentifierArray("tag", Array.Empty<Identifier>());
278  }
279  }
280  }
281 
282  Identifier[] excludedIdentifiers = element.GetAttributeIdentifierArray("excludeditems", null) ?? element.GetAttributeIdentifierArray("excludeditem", null);
283  if (excludedIdentifiers == null)
284  {
285  excludedIdentifiers = element.GetAttributeIdentifierArray("excludedidentifiers", null) ?? element.GetAttributeIdentifierArray("excludedtags", null);
286  if (excludedIdentifiers == null)
287  {
288  excludedIdentifiers = element.GetAttributeIdentifierArray("excludedidentifier", null) ?? element.GetAttributeIdentifierArray("excludedtag", Array.Empty<Identifier>());
289  }
290  }
291 
292  if (identifiers.Length == 0 && excludedIdentifiers.Length == 0 && !returnEmpty) { return null; }
293 
294  RelatedItem ri = new RelatedItem(identifiers, excludedIdentifiers)
295  {
296  ExcludeBroken = element.GetAttributeBool("excludebroken", true),
297  RequireEmpty = element.GetAttributeBool("requireempty", false),
298  ExcludeFullCondition = element.GetAttributeBool("excludefullcondition", false),
299  AllowVariants = element.GetAttributeBool("allowvariants", true),
300  Rotation = element.GetAttributeFloat("rotation", 0f),
301  SetActive = element.GetAttributeBool("setactive", false)
302  };
303  if (element.GetAttribute(nameof(Hide)) != null)
304  {
305  ri.Hide = element.GetAttributeBool(nameof(Hide), false);
306  }
307  if (element.GetAttribute(nameof(ItemPos)) != null)
308  {
309  ri.ItemPos = element.GetAttributeVector2(nameof(ItemPos), Vector2.Zero);
310  }
311  string typeStr = element.GetAttributeString("type", "");
312  if (string.IsNullOrEmpty(typeStr))
313  {
314  switch (element.Name.ToString().ToLowerInvariant())
315  {
316  case "containable":
317  typeStr = "Contained";
318  break;
319  case "suitablefertilizer":
320  case "suitableseed":
321  typeStr = "None";
322  break;
323  }
324  }
325  if (!Enum.TryParse(typeStr, true, out ri.type))
326  {
327  DebugConsole.ThrowError("Error in RelatedItem config (" + parentDebugName + ") - \"" + typeStr + "\" is not a valid relation type.");
328  return null;
329  }
330 
331  ri.MsgTag = element.GetAttributeIdentifier("msg", Identifier.Empty);
332  LocalizedString msg = TextManager.Get(ri.MsgTag);
333  if (!msg.Loaded)
334  {
335  ri.Msg = ri.MsgTag.Value;
336  }
337  else
338  {
339 #if CLIENT
340  foreach (InputType inputType in Enum.GetValues(typeof(InputType)))
341  {
342  msg = msg.Replace("[" + inputType.ToString().ToLowerInvariant() + "]", GameSettings.CurrentConfig.KeyMap.KeyBindText(inputType));
343  }
344  ri.Msg = msg;
345 #endif
346  }
347 
348  foreach (var subElement in element.Elements())
349  {
350  if (!subElement.Name.ToString().Equals("statuseffect", StringComparison.OrdinalIgnoreCase)) { continue; }
351  ri.statusEffects.Add(StatusEffect.Load(subElement, parentDebugName));
352  }
353 
354  ri.IsOptional = element.GetAttributeBool("optional", false);
355  ri.IgnoreInEditor = element.GetAttributeBool("ignoreineditor", false);
356  ri.MatchOnEmpty = element.GetAttributeBool("matchonempty", false);
357  ri.TargetSlot = element.GetAttributeInt("targetslot", -1);
358 
359  return ri;
360  }
