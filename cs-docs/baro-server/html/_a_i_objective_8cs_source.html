212  public void SortSubObjectives()
213  {
214  if (!AllowSubObjectiveSorting) { return; }
215  if (subObjectives.None()) { return; }
216  var previousSubObjective = subObjectives.First();
217  subObjectives.ForEach(so => so.GetPriority());
218  subObjectives.Sort((x, y) => y.Priority.CompareTo(x.Priority));
219  if (ConcurrentObjectives)
220  {
221  subObjectives.ForEach(so => so.SortSubObjectives());
222  }
223  else
224  {
225  var currentSubObjective = subObjectives.First();
226  if (previousSubObjective != currentSubObjective)
227  {
228  previousSubObjective.OnDeselected();
229  currentSubObjective.OnSelected();
230  }
231  currentSubObjective.SortSubObjectives();
232  }

307  {
308  float yDist = Math.Abs(selfPos.Y - targetWorldPos.Y);
309  yDist = yDist > 100 ? yDist * verticalDistanceMultiplier : 0;
310  float distance = Math.Abs(selfPos.X - targetWorldPos.X) + yDist;

547  protected static bool CanPutInInventory(Character character, Item item, bool allowWearing)
548  {
549  if (item == null) { return false; }
550  bool canEquip = false;
551  if (item.AllowedSlots.Contains(InvSlotType.Any))
552  {
553  if (character.Inventory.IsAnySlotAvailable(item))
554  {
555  canEquip = true;
556  }
557  }
558  if (!canEquip)
559  {
560  var inv = character.Inventory;
561  foreach (var allowedSlot in item.AllowedSlots)
562  {
563  if (!allowWearing)
564  {
565  if (!allowedSlot.HasFlag(InvSlotType.RightHand) && !allowedSlot.HasFlag(InvSlotType.LeftHand))
566  {
567  continue;
568  }
569  }
570  foreach (var slotType in inv.SlotTypes)
571  {
572  if (!allowedSlot.HasFlag(slotType)) { continue; }
573  for (int i = 0; i < inv.Capacity; i++)
574  {
575  canEquip = true;
576  if (allowedSlot.HasFlag(inv.SlotTypes[i]) && inv.GetItemAt(i) != null)
577  {
578  canEquip = false;
579  break;
580  }
581  }
582  }
583  }
584  }
585  return canEquip && character.Inventory.CanBePut(item);
586  }
