177  public static bool IsValidName(string name, ServerSettings serverSettings)
178  {
179  if (string.IsNullOrWhiteSpace(name)) { return false; }
180 
181  char[] disallowedChars =
182  {
183  //',', //previously disallowed because of the ban list format
184 
185  ';',
186  '<',
187  '>',
188 
189  '/', //disallowed because of server messages using forward slash as a delimiter (TODO: implement escaping)
190 
191  '\\',
192  '[',
193  ']',
194  '"',
195  '?'
196  };
197  if (name.Any(c => disallowedChars.Contains(c))) { return false; }
198 
199  foreach (char character in name)
200  {
201  if (!serverSettings.AllowedClientNameChars.Any(charRange => (int)character >= charRange.Start && (int)character <= charRange.End)) { return false; }
202  }
203 
204  return true;
205  }

291  public bool TryTakeOverBot(Character botCharacter)
292  {
293  if (GameMain.Server == null)
294  {
295  DebugConsole.ThrowError($"TryTakeOverBot: Client {Name} requested to take over a bot but GameMain.Server is null!");
296  return false;
297  }
298  if (GameMain.NetworkMember is not { ServerSettings.RespawnMode: RespawnMode.Permadeath })
299  {
300  DebugConsole.ThrowError($"Client {Name} requested to take over a bot but Permadeath is not enabled!");
301  GameMain.Server.SendConsoleMessage($"Permadeath mode is not enabled, cannot take over a bot.", this, Color.Red);
302  return false;
303  }
304  if (CharacterInfo == null)
305  {
306  DebugConsole.ThrowError($"Permadeath: Client {Name} requested to take over a bot, but they don't seem to have a character at all yet.");
307  GameMain.Server.SendConsoleMessage($"Permadeath: Taking over a bot requires having a character that died first.", this, Color.Red);
308  return false;
309  }
310  if (CharacterInfo is not { PermanentlyDead: true })
311  {
312  DebugConsole.ThrowError($"Permadeath: Client {Name} requested to take over a bot, but their character has not been permanently killed.");
313  GameMain.Server.SendConsoleMessage($"Permadeath: Could not take over the bot, previous character not permanently killed.", this, Color.Red);
314  return false;
315  }
316  if (!botCharacter.IsBot)
317  {
318  DebugConsole.ThrowError($"Permadeath: {Name} requested to take over a bot character, but the target character is not a bot!");
319  GameMain.Server.SendConsoleMessage($"Permadeath: Could not take over the target character because it is not a bot.", this, Color.Red);
320  return false;
321  }
322 
323  if (botCharacter.Info != null)
324  {
325  botCharacter.Info.RenamingEnabled = true; // Grant one opportunity to rename a taken over bot
326  }
327 
328  // Now that the old permanently killed character will be replaced, we can fully discard it
329  var mpCampaign = GameMain.GameSession?.Campaign as MultiPlayerCampaign;
330  mpCampaign?.DiscardClientCharacterData(this);
331  GameMain.Server.SetClientCharacter(this, botCharacter);
332  if (mpCampaign?.SetClientCharacterData(this) is CharacterCampaignData characterData)
333  {
334  //the bot has spawned, but the new CharacterCampaignData technically hasn't, because we just created it
335  characterData.HasSpawned = true;
336  }
337 
338  SpectateOnly = false;
339  return true;
340  }
