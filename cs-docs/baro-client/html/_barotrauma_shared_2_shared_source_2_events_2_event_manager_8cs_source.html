511  private void CreateEvents(EventSet eventSet)
512  {
513  selectedEvents.Remove(eventSet);
514  if (level == null) { return; }
515  if (level.LevelData.HasHuntingGrounds && eventSet.DisableInHuntingGrounds) { return; }
516  if (eventSet.Exhaustible && level.LevelData.EventsExhausted) { return; }
517 
518  DebugConsole.NewMessage($"Loading event set {eventSet.Identifier}", Color.LightBlue, debugOnly: true);
519 
520  int applyCount = 1;
521  List<Func<Level.InterestingPosition, bool>> spawnPosFilter = new List<Func<Level.InterestingPosition, bool>>();
522  if (eventSet.PerRuin)
523  {
524  applyCount = level.Ruins.Count;
525  foreach (var ruin in level.Ruins)
526  {
527  spawnPosFilter.Add(pos => pos.Ruin == ruin);
528  }
529  }
530  else if (eventSet.PerCave)
531  {
532  applyCount = level.Caves.Count;
533  foreach (var cave in level.Caves)
534  {
535  spawnPosFilter.Add(pos => pos.Cave == cave);
536  }
537  }
538  else if (eventSet.PerWreck)
539  {
540  var wrecks = Submarine.Loaded.Where(s => s.Info.IsWreck && (s.WreckAI == null || !s.WreckAI.IsAlive));
541  applyCount = wrecks.Count();
542  foreach (var wreck in wrecks)
543  {
544  spawnPosFilter.Add(pos => pos.Submarine == wreck);
545  }
546  }
547 
548  foreach (var subEventPrefab in eventSet.EventPrefabs)
549  {
550  foreach (Identifier missingId in subEventPrefab.GetMissingIdentifiers())
551  {
552  DebugConsole.ThrowError($"Error in event set \"{eventSet.Identifier}\" ({eventSet.ContentFile?.ContentPackage?.Name ?? "null"}) - could not find an event prefab with the identifier \"{missingId}\".",
553  contentPackage: eventSet.ContentPackage);
554  }
555  }
556 
557  var suitablePrefabSubsets = eventSet.EventPrefabs.Where(
558  e => IsFactionSuitable(e.Faction, level) && e.EventPrefabs.Any(ep => IsSuitable(ep, level))).ToArray();
559 
560  for (int i = 0; i < applyCount; i++)
561  {
562  if (eventSet.ChooseRandom)
563  {
564  if (suitablePrefabSubsets.Any())
565  {
566  var unusedEvents = suitablePrefabSubsets.ToList();
567  int eventCount = eventSet.GetEventCount(level);
568  for (int j = 0; j < eventCount; j++)
569  {
570  if (unusedEvents.All(e => e.EventPrefabs.All(p => CalculateCommonness(p, e.Commonness) <= 0.0f))) { break; }
571  EventSet.SubEventPrefab subEventPrefab = ToolBox.SelectWeightedRandom(unusedEvents, e => e.EventPrefabs.Max(p => CalculateCommonness(p, e.Commonness)), random);
572  (IEnumerable<EventPrefab> eventPrefabs, float commonness, float probability) = subEventPrefab;
573  if (eventPrefabs != null && random.NextDouble() <= probability)
574  {
575  var eventPrefab = ToolBox.SelectWeightedRandom(eventPrefabs.Where(e => IsSuitable(e, level)), e => e.Commonness, random);
576  var newEvent = eventPrefab.CreateInstance(RandomSeed);
577  if (newEvent == null) { continue; }
578  if (i < spawnPosFilter.Count) { newEvent.SpawnPosFilter = spawnPosFilter[i]; }
579  DebugConsole.NewMessage($"Initialized event {newEvent}", debugOnly: true);
580  if (!selectedEvents.ContainsKey(eventSet))
581  {
582  selectedEvents.Add(eventSet, new List<Event>());
583  }
584  selectedEvents[eventSet].Add(newEvent);
585  unusedEvents.Remove(subEventPrefab);
586  }
587  }
588  }
589  if (eventSet.ChildSets.Any())
590  {
591  int setCount = eventSet.SubSetCount;
592  if (setCount > 1)
593  {
594  var unusedSets = eventSet.ChildSets.ToList();
595  for (int j = 0; j < setCount; j++)
596  {
597  var newEventSet = SelectRandomEvents(unusedSets, random: random);
598  if (newEventSet == null) { break; }
599  unusedSets.Remove(newEventSet);
600  CreateEvents(newEventSet);
601  }
602  }
603  else
604  {
605  var newEventSet = SelectRandomEvents(eventSet.ChildSets, random: random);
606  if (newEventSet != null)
607  {
608  CreateEvents(newEventSet);
609  }
610  }
611  }
612  }
613  else
614  {
615  foreach ((IEnumerable<EventPrefab> eventPrefabs, float commonness, float probability) in suitablePrefabSubsets)
616  {
617  if (random.NextDouble() > probability) { continue; }
618 
619  var eventPrefab = ToolBox.SelectWeightedRandom(eventPrefabs.Where(e => IsSuitable(e, level)), e => e.Commonness, random);
620  var newEvent = eventPrefab.CreateInstance(RandomSeed);
621  if (newEvent == null) { continue; }
622  if (i < spawnPosFilter.Count) { newEvent.SpawnPosFilter = spawnPosFilter[i]; }
623  if (!selectedEvents.ContainsKey(eventSet))
624  {
625  selectedEvents.Add(eventSet, new List<Event>());
626  }
627  selectedEvents[eventSet].Add(newEvent);
628  }
629 
630  var location = GetEventLocation();
631  foreach (EventSet childEventSet in eventSet.ChildSets)
632  {
633  if (!IsValidForLevel(childEventSet, level)) { continue; }
634  if (!IsValidForLocation(childEventSet, location)) { continue; }
635  CreateEvents(childEventSet);
636  }
637  }
638  }

641  private IEnumerable<EventSet> GetAllowedEventSets(IReadOnlyList<EventSet> eventSets, bool? requireCampaignSet = null)
642  {
643  if (level == null) { return Enumerable.Empty<EventSet>(); }
644 
645  var allowedEventSets =
646  eventSets.Where(set => IsValidForLevel(set, level));
647 
648  if (requireCampaignSet.HasValue)
649  {
650  if (requireCampaignSet.Value)
651  {
652  if (allowedEventSets.Any(es => es.IsCampaignSet))
653  {
654  allowedEventSets =
655  allowedEventSets.Where(es => es.IsCampaignSet);
656  }
657  else
658  {
659  DebugConsole.AddWarning("No campaign event sets available. Using a non-campaign-specific set instead.");
660  }
661  }
662  else
663  {
664  allowedEventSets =
665  allowedEventSets.Where(es => !es.IsCampaignSet);
666  }
667  }
668 
669  var location = GetEventLocation();
670  allowedEventSets = allowedEventSets.Where(set => IsValidForLocation(set, location));
671 
672  allowedEventSets = allowedEventSets.Where(set => !set.CampaignTutorialOnly ||
673  (GameMain.IsSingleplayer && GameMain.GameSession?.Campaign?.Settings is { TutorialEnabled: true }));
674 
675  int? discoveryIndex = GameMain.GameSession?.Map?.GetDiscoveryIndex(location);
676  int? visitIndex = GameMain.GameSession?.Map?.GetVisitIndex(location);
677  if (discoveryIndex is not null && discoveryIndex >= 0 && allowedEventSets.Any(set => set.ForceAtDiscoveredNr == discoveryIndex))
678  {
679  allowedEventSets = allowedEventSets.Where(set => set.ForceAtDiscoveredNr == discoveryIndex);
680  }
681  else if (visitIndex is not null && visitIndex >= 0 && allowedEventSets.Any(set => set.ForceAtVisitedNr == visitIndex))
682  {
683  allowedEventSets = allowedEventSets.Where(set => set.ForceAtVisitedNr == visitIndex);
684  }
685  else
686  {
687  // When there are no forced sets, only allow sets that aren't forced at any specific location
688  allowedEventSets = allowedEventSets.Where(set => set.ForceAtDiscoveredNr < 0 && set.ForceAtVisitedNr < 0);
689  }
690  return allowedEventSets;
691  }

693  private EventSet SelectRandomEvents(IReadOnlyList<EventSet> eventSets, bool? requireCampaignSet = null, Random random = null)
694  {
695  var allowedEventSets = GetAllowedEventSets(eventSets, requireCampaignSet);
696 
697  if (allowedEventSets.Count() == 1)
698  {
699  // When there's only a single set available, just select it directly
700  return allowedEventSets.First();
701  }
702 
703  Random rand = random ?? new MTRandom(ToolBox.StringToInt(level.Seed));
704  float totalCommonness = allowedEventSets.Sum(e => e.GetCommonness(level));
705  float randomNumber = (float)rand.NextDouble();
706  randomNumber *= totalCommonness;
707  foreach (EventSet eventSet in allowedEventSets)
708  {
709  float commonness = eventSet.GetCommonness(level);
710  if (randomNumber <= commonness)

961  private void CalculateCurrentIntensity(float deltaTime)
962  {
963  intensityUpdateTimer -= deltaTime;
964  if (intensityUpdateTimer > 0.0f) { return; }
965  intensityUpdateTimer = IntensityUpdateInterval;
966 
967  // crew health --------------------------------------------------------
968 
969  avgCrewHealth = 0.0f;
970  int characterCount = 0;
971  foreach (Character character in Character.CharacterList)
972  {
973  if (character.IsDead || character.TeamID == CharacterTeamType.FriendlyNPC) { continue; }
974  if (character.AIController is HumanAIController || character.IsRemotePlayer)
975  {
976  avgCrewHealth += character.Vitality / character.MaxVitality * (character.IsUnconscious ? 0.5f : 1.0f);
977  characterCount++;
978  }
979  }
980  if (characterCount > 0)
981  {
982  avgCrewHealth /= characterCount;
983  }
984  else
985  {
986  avgCrewHealth = 0.5f;
987  }
988 
989  // enemy amount --------------------------------------------------------
990 
991  enemyDanger = 0.0f;
992  monsterStrength = 0;
993  foreach (Character character in Character.CharacterList)
994  {
995  if (character.IsIncapacitated || character.IsHandcuffed || !character.Enabled || character.IsPet) { continue; }
996 
997  if (character.AIController is EnemyAIController enemyAI)
998  {
999  if (!enemyAI.AIParams.StayInAbyss)
1000  {
1001  // Ignore abyss monsters because they can stay active for quite great distances. They'll be taken into account when they target the sub.
1002  monsterStrength += enemyAI.CombatStrength;
1003  }
1004 
1005  if (Submarine.MainSub != null &&
1006  character.CurrentHull?.Submarine.Info is { Type: SubmarineType.Player } &&
1007  (character.CurrentHull.Submarine == Submarine.MainSub || Submarine.MainSub.DockedTo.Contains(character.CurrentHull.Submarine)))
1008  {
1009  // Enemy onboard -> Crawler inside the sub adds 0.2 to enemy danger, Mudraptor 0.42
1010  enemyDanger += enemyAI.CombatStrength / 500.0f;
1011  }
1012  else if (enemyAI.SelectedAiTarget?.Entity?.Submarine != null)
1013  {
1014  // Enemy outside targeting the sub or something in it
1015  // -> One Crawler adds 0.02, a Mudraptor 0.042, a Hammerhead 0.1, and a Moloch 0.25.
1016  enemyDanger += enemyAI.CombatStrength / 5000.0f;
1017  }
1018  }
1019  else if (character.AIController is HumanAIController humanAi && !character.IsOnFriendlyTeam(CharacterTeamType.Team1))
1020  {
1021  if (character.Submarine != null &&
1022  character.Submarine.PhysicsBody is { BodyType: BodyType.Dynamic } &&
1023  Vector2.DistanceSquared(character.Submarine.WorldPosition, Submarine.MainSub.WorldPosition) < Sonar.DefaultSonarRange * Sonar.DefaultSonarRange)
1024  {
1025  //we have no easy way to define the strength of a human enemy (depends more on the sub and it's state than the character),
1026  //so let's just go with a fixed value.
1027  //5 living enemy characters in an enemy sub in sonar range is enough to bump the intensity to max
1028  enemyDanger += 0.2f;
1029  }
1030  }
1031  }
1032 
1033  // Add a portion of the total strength of active monsters to the enemy danger so that we don't spawn too many monsters around the sub.
1034  // On top of the existing value, so if 10 crawlers are targeting the sub simultaneously from outside, the final value would be: 0.02 x 10 + 0.2 = 0.4.
1035  // And if they get inside, we add 0.1 per crawler on that.
1036  // So, in practice the danger per enemy that is attacking the sub is half of what it would be when the enemy is not targeting the sub.
1037  // 10 Crawlers -> +0.2 (0.4 in total if all target the sub from outside).
1038  // 5 Mudraptors -> +0.21 (0.42 in total, before they get inside).
1039  // 3 Hammerheads -> +0.3 (0.6 in total, if they all target the sub).
1040  // 2 Molochs -> +0.5 (1.0 in total, if both target the sub).
1041  enemyDanger += monsterStrength / 5000f;
1042  enemyDanger = MathHelper.Clamp(enemyDanger, 0.0f, 1.0f);
1043 
1044  // The definitions above aim for that we never spawn more monsters that the player (and the performance) can handle.
1045  // Some examples that result in the max intensity even when the creatures would just idle around.
1046  // The values are theoretical, because in practice many of the monsters are targeting the sub, which will double the danger of those monster and effectively halve the max monster count.
1047  // In practice we don't use the max intensity. For example on level 50 we use max intensity 50, which would mean that we'd halve the numbers below.
1048  // There's no hard cap for the monster count, but if the amount of monsters is higher than this, we don't spawn more monsters from the events:
1049  // 50 Crawlers (We shouldn't actually ever spawn that many. 12 is the max per event, but theoretically 25 crawlers would result in max intensity).
1050  // 25 Tigerthreshers (Max 9 per event. 12 targeting the sub at the same time results in max intensity).
1051  // 10 Hammerheads (Max 3 per event. 5 targeting the sub at the same time results in max intensity).
1052  // 4 Molochs (Max 2 per event and 2 targeting the sub at the same time results in max intensity).
1053 
1054  // hull status (gaps, flooding, fire) --------------------------------------------------------
1055 
1056  float holeCount = 0.0f;
1057  float waterAmount = 0.0f;
1058  float dryHullVolume = 0.0f;
1059  foreach (Hull hull in Hull.HullList)
1060  {
1061  if (hull.Submarine == null || hull.Submarine.Info.Type != SubmarineType.Player) { continue; }
1062  if (GameMain.GameSession?.GameMode is PvPMode)
1063  {
1064  if (hull.Submarine.TeamID != CharacterTeamType.Team1 && hull.Submarine.TeamID != CharacterTeamType.Team2) { continue; }
1065  }
1066  else
1067  {
1068  if (hull.Submarine.TeamID != CharacterTeamType.Team1) { continue; }
1069  }
1070  fireAmount += hull.FireSources.Sum(fs => fs.Size.X);
1071  if (hull.IsWetRoom) { continue; }
1072  foreach (Gap gap in hull.ConnectedGaps)
1073  {
1074  if (!gap.IsRoomToRoom)
1075  {
1076  holeCount += gap.Open;
1077  }
1078  }
1079  waterAmount += hull.WaterVolume;
1080  dryHullVolume += hull.Volume;
1081  }
1082  if (dryHullVolume > 0)
1083  {
1084  floodingAmount = waterAmount / dryHullVolume;
1085  }
1086 
1087  //hull integrity at 0.0 if there are 10 or more wide-open holes
1088  avgHullIntegrity = MathHelper.Clamp(1.0f - holeCount / 10.0f, 0.0f, 1.0f);
1089 
1090  //a fire of any size bumps up the fire amount to 20%
1091  //if the total width of the fires is 1000 or more, the fire amount is considered to be at 100%
1092  fireAmount = MathHelper.Clamp(fireAmount / 1000.0f, fireAmount > 0.0f ? 0.2f : 0.0f, 1.0f);
1093 
1094  //flooding less than 10% of the sub is ignored
1095  //to prevent ballast tanks from affecting the intensity
1096  if (floodingAmount < 0.1f)
1097  {
1098  floodingAmount = 0.0f;
1099  }
1100  else
1101  {
1102  floodingAmount *= 1.5f;
1103  }
1104 
1105  // calculate final intensity --------------------------------------------------------
1106 
1107  targetIntensity =
1108  ((1.0f - avgCrewHealth) + (1.0f - avgHullIntegrity) + floodingAmount) / 3.0f;
1109  targetIntensity += fireAmount * 0.5f;
1110  targetIntensity += enemyDanger;
1111  targetIntensity = MathHelper.Clamp(targetIntensity, 0.0f, 1.0f);
1112 
1113  if (targetIntensity > currentIntensity)
1114  {
1115  //25 seconds for intensity to go from 0.0 to 1.0
1116  currentIntensity = Math.Min(currentIntensity + 0.04f * IntensityUpdateInterval, targetIntensity);
1117  //20 seconds for intensity to go from 0.0 to 1.0
1118  musicIntensity = Math.Min(musicIntensity + 0.05f * IntensityUpdateInterval, targetIntensity);
1119  }
1120  else
1121  {
1122  //400 seconds for intensity to go from 1.0 to 0.0
1123  currentIntensity = Math.Max(currentIntensity - 0.0025f * IntensityUpdateInterval, targetIntensity);
1124  //20 seconds for intensity to go from 1.0 to 0.0
1125  musicIntensity = Math.Max(musicIntensity - 0.05f * IntensityUpdateInterval, targetIntensity);
1126  }
