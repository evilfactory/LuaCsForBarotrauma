60  void AddTraitorExpSteal(Character traitorCharacter, float expSteal)
61  {
62  if (traitorCharacter == null) { return; }
63  if (!traitorExpSteal.ContainsKey(traitorCharacter))
64  {
65  traitorExpSteal.Add(traitorCharacter, 0.0f);
66  }
67  traitorExpSteal[traitorCharacter] += expSteal;
68  }
69  }
70  totalExpSteal = traitorExpSteal.Values.Sum();
71  //if exp to steal exceeds 100%, normalize to get it back to 100%
72  //(e.g. two traitors who both steal 75%, they'll share 50% of all the exp gains)
73  if (totalExpSteal > 100.0f)
74  {
75  foreach (Character traitor in traitorExpSteal.Keys)
76  {
77  traitorExpSteal[traitor] /= totalExpSteal;
78  }
79  totalExpSteal = 100.0f;
80  }
81  if (totalExpSteal > 0)
82  {
83  GameServer.Log($"Traitors stole {(int)totalExpSteal}% of the total experience.", ServerLog.MessageType.Traitors);
84  }
85 
86  int nonTraitorCount = GameSession.GetSessionCrewCharacters(CharacterType.Both).Count(c => !traitorExpSteal.ContainsKey(c));
87  foreach (Networking.Client c in GameMain.Server.ConnectedClients)
88  {
89  //give the experience to the stored characterinfo if the client isn't currently controlling a character
90  GiveMissionExperience(c.Character?.Info ?? c.CharacterInfo);
91  }
92  foreach (Character bot in GameSession.GetSessionCrewCharacters(CharacterType.Bot))
93  {

98  {

100  var experienceGainMultiplierIndividual = new AbilityMissionExperienceGainMultiplier(this, 1f, info.Character);

103  {
104  if (c == info.Character) { continue; }

108 
109  int finalExperienceGain = (int)(experienceGain * experienceGainMultiplierIndividual.Value);
110  if (info.Character != null && traitorExpSteal.TryGetValue(info.Character, out float expToSteal))
111  {
112  int stealAmount = (int)(experienceGain * nonTraitorCount * expToSteal / 100.0f);
113  GameServer.Log($"Traitor {info.Character} stole {stealAmount} ({(int)expToSteal}%) of the total experience.", ServerLog.MessageType.Traitors);
114  finalExperienceGain += stealAmount;
115  }
116  else
117  {
118  GameServer.Log($"{(int)(finalExperienceGain * totalExpSteal / 100.0f)} ({(int)totalExpSteal}%) was stolen from {info.Name}.", ServerLog.MessageType.Traitors);
119  finalExperienceGain -= (int)(finalExperienceGain * totalExpSteal / 100.0f);
120  }
121  info.GiveExperience(finalExperienceGain);
