498  protected override void UpdateMissionSpecific(float deltaTime)
499  {
500  if (state >= 2 || enemySub == null) { return; }
501 
502  float sqrSonarRange = MathUtils.Pow2(Sonar.DefaultSonarRange);
503  outsideOfSonarRange = Vector2.DistanceSquared(enemySub.WorldPosition, Submarine.MainSub.WorldPosition) > sqrSonarRange;
504 
505  if (CheckWinState())
506  {
507  State = 2;
508  }
509  else
510  {
511  switch (State)
512  {
513  case 0:
514  for (int i = patrolPositions.Count - 1; i >= 0; i--)
515  {
516  if (Vector2.DistanceSquared(patrolPositions[i], Submarine.MainSub.WorldPosition) < sqrSonarRange)
517  {
518  patrolPositions.RemoveAt(i);
519  }
520  }
521  if (!outsideOfSonarRange || patrolPositions.None())
522  {
523  State = 1;
524  }
525  break;
526  case 1:
527  if (outsideOfSonarRange)
528  {
529  if (lastSighting.HasValue && Vector2.DistanceSquared(lastSighting.Value, Submarine.MainSub.WorldPosition) < sqrSonarRange)
530  {
531  lastSighting = null;
532  }
533  pirateSightingUpdateTimer -= deltaTime;
534  if (pirateSightingUpdateTimer < 0)
535  {
536  pirateSightingUpdateTimer = pirateSightingUpdateFrequency;
537  lastSighting = enemySub.WorldPosition;
538  }
539  }
540  else
541  {
542  lastSighting = enemySub.WorldPosition;
543  pirateSightingUpdateTimer = 0;
544  }
545  break;
546  }
547  }
