64  public static bool TalentTreeMeetsRequirements(TalentTree tree, TalentSubTree targetTree, IReadOnlyCollection<Identifier> selectedTalents)
65  {
66  IEnumerable<TalentSubTree> blockingSubTrees = tree.TalentSubTrees.Where(tst => tst.BlockedTrees.Contains(targetTree.Identifier)),
67  requiredSubTrees = tree.TalentSubTrees.Where(tst => targetTree.RequiredTrees.Contains(tst.Identifier));

75  public static TalentStages GetTalentOptionStageState(Character character, Identifier subTreeIdentifier, int index, IReadOnlyCollection<Identifier> selectedTalents)
76  {
77  if (character?.Info?.Job.Prefab is null) { return TalentStages.Invalid; }
78 
79  if (!JobTalentTrees.TryGet(character.Info.Job.Prefab.Identifier, out TalentTree talentTree)) { return TalentStages.Invalid; }
80 
81  TalentSubTree subTree = talentTree!.TalentSubTrees.FirstOrDefault(tst => tst.Identifier == subTreeIdentifier);
82  if (subTree is null) { return TalentStages.Invalid; }
83 
84  if (!TalentTreeMeetsRequirements(talentTree, subTree, selectedTalents))
85  {
86  return TalentStages.Locked;
87  }
88 
89  TalentOption targetTalentOption = subTree.TalentOptionStages[index];
90 
91  if (targetTalentOption.HasEnoughTalents(character.Info))
92  {
93  return TalentStages.Unlocked;
94  }
95 
96  if (targetTalentOption.HasSelectedTalent(selectedTalents))
97  {
98  return TalentStages.Highlighted;
99  }
100 
101  bool hasTalentInLastTier = true;
102  bool isLastTalentPurchased = true;
103 
104  int lastindex = index - 1;
105  if (lastindex >= 0)
106  {
107  TalentOption lastLatentOption = subTree.TalentOptionStages[lastindex];
108  hasTalentInLastTier = lastLatentOption.HasEnoughTalents(selectedTalents);
109  isLastTalentPurchased = lastLatentOption.HasEnoughTalents(character.Info);
110  }
111 
112  if (!hasTalentInLastTier)
113  {
114  return TalentStages.Locked;
115  }
116 
117  bool hasPointsForNewTalent = character.Info.GetTotalTalentPoints() - selectedTalents.Count > 0;
118 
119  if (hasPointsForNewTalent)
120  {
121  return isLastTalentPurchased ? TalentStages.Highlighted : TalentStages.Available;
122  }
123 
124  return TalentStages.Locked;
125  }

128  public static bool IsViableTalentForCharacter(Character character, Identifier talentIdentifier, IReadOnlyCollection<Identifier> selectedTalents)
129  {
130  if (character?.Info?.Job.Prefab == null) { return false; }
131  if (character.Info.GetTotalTalentPoints() - selectedTalents.Count <= 0) { return false; }
132  if (!JobTalentTrees.TryGet(character.Info.Job.Prefab.Identifier, out TalentTree talentTree)) { return false; }
133 
134  if (IsTalentLocked(talentIdentifier)) { return false; }
135 
136  foreach (var subTree in talentTree!.TalentSubTrees)
137  {
138  if (subTree.AllTalentIdentifiers.Contains(talentIdentifier) && subTree.HasMaxTalents(selectedTalents)) { return false; }
139 
140  foreach (var talentOptionStage in subTree.TalentOptionStages)
141  {
142  if (talentOptionStage.TalentIdentifiers.Contains(talentIdentifier))
143  {
144  return !talentOptionStage.HasMaxTalents(selectedTalents) && TalentTreeMeetsRequirements(talentTree, subTree, selectedTalents);
145  }
146  bool optionStageCompleted = talentOptionStage.HasEnoughTalents(selectedTalents);
147  if (!optionStageCompleted)
148  {
149  break;
150  }
151  }
152  }
153 
154  return false;
155  }

169  public static List<Identifier> CheckTalentSelection(Character controlledCharacter, IEnumerable<Identifier> selectedTalents)
170  {
171  List<Identifier> viableTalents = new List<Identifier>();
172  bool canStillUnlock = true;
173  // keep trying to unlock talents until none of the talents are unlockable
174  while (canStillUnlock && selectedTalents.Any())
175  {
176  canStillUnlock = false;
177  foreach (Identifier talent in selectedTalents)
178  {
179  if (!viableTalents.Contains(talent) && IsViableTalentForCharacter(controlledCharacter, talent, viableTalents))

183  }

316  showCaseTalentIdentifiers.Add(identifier);
