376  public void Shoot(Character user, Vector2 weaponPos, Vector2 spawnPos, float rotation, List<Body> ignoredBodies, bool createNetworkEvent, float damageMultiplier = 1f, float launchImpulseModifier = 0f)
377  {
378  //add the limbs of the shooter to the list of bodies to be ignored
379  //so that the player can't shoot himself
380  IgnoredBodies = ignoredBodies;
381  Vector2 projectilePos = weaponPos;
382  //make sure there's no obstacles between the base of the weapon (or the shoulder of the character) and the end of the barrel
383  if (Submarine.PickBody(weaponPos, spawnPos, IgnoredBodies, Physics.CollisionWall | Physics.CollisionLevel | Physics.CollisionItemBlocking,
384  customPredicate: (Fixture f) => { return IgnoredBodies == null || !IgnoredBodies.Contains(f.Body); }) == null)
385  {
386  //no obstacles -> we can spawn the projectile at the barrel
387  projectilePos = spawnPos;
388  }
389  else if ((weaponPos - spawnPos).LengthSquared() > 0.0001f)
390  {
391  //spawn the projectile body.GetMaxExtent() away from the position where the raycast hit the obstacle
392  Vector2 newPos = weaponPos - Vector2.Normalize(spawnPos - projectilePos) * Math.Max(Item.body.GetMaxExtent(), 0.1f);
393  if (MathUtils.IsValid(newPos))
394  {
395  projectilePos = newPos;
396  }
397  }
398  Launch(user, projectilePos, rotation, damageMultiplier, launchImpulseModifier);
399  if (createNetworkEvent && !Item.Removed && GameMain.NetworkMember != null && GameMain.NetworkMember.IsServer)
400  {
401 #if SERVER
402  launchRot = rotation;
403  Item.CreateServerEvent(this, new EventData(launch: true, spreadCounter: (byte)(spreadIndex - 1)));

408  public bool Use(Character character = null, float launchImpulseModifier = 0f)
409  {
410  if (character != null && !characterUsable) { return false; }
411  if (item.body == null) { return false; }
412  //can't launch if already launched
413  if (StickTarget != null || IsActive) { return false; }
414 
415  float initialRotation = item.body.Rotation;
416  for (int i = 0; i < HitScanCount; i++)
417  {
418  float launchAngle;
419  if (StaticSpread)
420  {
421  launchAngle = initialRotation + MathHelper.ToRadians(i - ((float)(HitScanCount - 1) / 2)) * Spread;
422  }
423  else
424  {
425  launchAngle = initialRotation + MathHelper.ToRadians(Spread * GetSpreadFromPool());
426  }
427  spreadIndex++;
428 
429  Vector2 launchDir = new Vector2((float)Math.Cos(launchAngle), (float)Math.Sin(launchAngle));
430  if (Hitscan)
431  {
432  Vector2 prevSimpos = item.SimPosition;
433  item.body.SetTransformIgnoreContacts(item.body.SimPosition, launchAngle);
434  DoHitscan(launchDir);
435  if (i < HitScanCount - 1)
436  {
437  item.SetTransform(prevSimpos, item.body.Rotation);
438  }
439  }
440  else
441  {
442  item.body.SetTransform(item.body.SimPosition, launchAngle);
443  float modifiedLaunchImpulse = (LaunchImpulse + launchImpulseModifier) * (1 + Rand.Range(-ImpulseSpread, ImpulseSpread));
444  DoLaunch(launchDir * modifiedLaunchImpulse);
445  }
446  }
447  User = character;
448  ApplyStatusEffects(ActionType.OnUse, 1.0f, User, user: User);
449  return true;
450  }

455  {
456  hits.Clear();

458  Vector2 prevVelocity = item.body.LinearVelocity;
459 
460  if (item.AiTarget != null)

464  }
465 
466  //do not create a network event about dropping the projectile at this point,
467  //otherwise clients would fail to launch the correct projectile when firing the weapon
468  var prevInventory = item.ParentInventory;
469  if (prevInventory != null && GameMain.NetworkMember is { IsServer: true })
470  {
471  //update the state of the inventory after a delay,
472  //in case a client failed to launch the projectile for some reason
473  CoroutineManager.Invoke(() =>
474  {
475  if (item.Removed) { return; }
476  prevInventory.CreateNetworkEvent();
477  }, delay: CorrectionDelay / 2);
478  }
479  item.Drop(null, createNetworkEvent: false);
480 
481  Item.WaterDragCoefficient = WaterDragCoefficient;
482 
483  launchPos = item.SimPosition;
484  LaunchSub = item.Submarine;
485 
486  item.body.Enabled = true;
487  if (item.body.BodyType == BodyType.Kinematic)
488  {
489  item.body.LinearVelocity = impulse;
490  }
491  else if (impulse.LengthSquared() > 0.001f)
492  {
493  impulse *= item.body.Mass;
494  item.body.ApplyLinearImpulse(impulse, maxVelocity: NetConfig.MaxPhysicsBodyVelocity * 0.95f);
495  }
496  else
497  {
498  //if no impulse is defined, maintain the projectile's original velocity (if any)
499  //can be used to make throwable items behave as projectiles
500  item.body.LinearVelocity = prevVelocity;
501  }
502 
503  item.body.FarseerBody.OnCollision += OnProjectileCollision;
504  item.body.FarseerBody.IsBullet = true;
505 
506  EnableProjectileCollisions();
507 
508  IsActive = true;
509 
510  if (stickJoint == null) { return; }
511 
512  StickTarget = null;
513  GameMain.World.Remove(stickJoint);
514  stickJoint = null;

517  private void DoHitscan(Vector2 dir)
518  {
519  float rotation = item.body.Rotation;
520  Vector2 simPositon = item.SimPosition;
521  Vector2 rayStartWorld = item.WorldPosition;
522  item.Drop(null, createNetworkEvent: false);
523  Item.WaterDragCoefficient = WaterDragCoefficient;
524 
525  item.body.Enabled = true;
526  //set the velocity of the body because the OnProjectileCollision method
527  //uses it to determine the direction from which the projectile hit
528  item.body.LinearVelocity = dir;
529  IsActive = true;
530 
531  Vector2 rayStart = simPositon;
532  Vector2 rayEnd = rayStart + dir * 500.0f;
533 
534  float worldDist = 1000.0f;
535 #if CLIENT
536  worldDist = Screen.Selected?.Cam?.WorldView.Width ?? GameMain.GraphicsWidth;
537 #endif
538  Vector2 rayEndWorld = rayStartWorld + dir * worldDist;
539 
540  List<HitscanResult> hits = new List<HitscanResult>();
541  hits.AddRange(DoRayCast(rayStart, rayEnd, submarine: item.Submarine));
542 
543  if (item.Submarine != null)
544  {
545  //shooting indoors, do a hitscan outside as well
546  hits.AddRange(DoRayCast(rayStart + item.Submarine.SimPosition, rayEnd + item.Submarine.SimPosition, submarine: null));
547  //do a hitscan in other subs' coordinate spaces
548  RayCastInOtherSubs(rayStart + item.Submarine.SimPosition, rayEnd + item.Submarine.SimPosition);

555  void RayCastInOtherSubs(Vector2 rayStart, Vector2 rayEnd)
556  {
557  //shooting outdoors, see if we can hit anything inside a sub
558  foreach (Submarine submarine in Submarine.Loaded)
559  {
560  if (submarine == item.Submarine) { continue; }
561  var inSubHits = DoRayCast(rayStart - submarine.SimPosition, rayEnd - submarine.SimPosition, submarine);
562  //transform back to world coordinates
563  for (int i = 0; i < inSubHits.Count; i++)
564  {
565  inSubHits[i] = new HitscanResult(
566  inSubHits[i].Fixture,
567  inSubHits[i].Point + submarine.SimPosition,
568  inSubHits[i].Normal,
569  inSubHits[i].Fraction);
570  }
571  hits.AddRange(inSubHits);
572  }
573  }
574 
575  int hitCount = 0;
576  Vector2 lastHitPos = item.WorldPosition;
577  hits = hits.OrderBy(h => h.Fraction).ToList();
578  for (int i = 0; i < hits.Count; i++)
579  {
580  var h = hits[i];
581  item.SetTransform(h.Point, rotation);
582  item.UpdateTransform();
583  if (HandleProjectileCollision(h.Fixture, h.Normal, Vector2.Zero))
584  {
585  hitCount++;
586  if (hitCount >= MaxTargetsToHit || i == hits.Count - 1)
587  {
588  LaunchProjSpecific(rayStartWorld, item.WorldPosition);
589  break;
590  }
591  }
592  }
593  //the raycast didn't hit anything (or didn't hit enough targets to stop the projectile) -> the projectile flew somewhere outside the level and is permanently lost
594  if (hitCount < MaxTargetsToHit)
595  {
596  item.body.SetTransformIgnoreContacts(item.body.SimPosition, rotation);
597  LaunchProjSpecific(rayStartWorld, rayEndWorld);
598  if (Entity.Spawner == null)
599  {
600  item.Remove();
601  }
602  else
603  {
604  if (GameMain.NetworkMember != null && GameMain.NetworkMember.IsClient)
605  {
606  //clients aren't allowed to remove items by themselves, so lets hide the projectile until the server tells us to remove it
607  item.HiddenInGame = Hitscan;
608  }
609  else
610  {
611  Entity.Spawner.AddItemToRemoveQueue(item);
612  }
613  }
614  }

617  private List<HitscanResult> DoRayCast(Vector2 rayStart, Vector2 rayEnd, Submarine submarine)
618  {
619  List<HitscanResult> hits = new List<HitscanResult>();
620 
621  Vector2 dir = rayEnd - rayStart;
622  dir = dir.LengthSquared() < 0.00001f ? Vector2.UnitY : Vector2.Normalize(dir);
623 
624  //do an AABB query first to see if the start of the ray is inside a fixture
625  var aabb = new FarseerPhysics.Collision.AABB(rayStart - Vector2.One * 0.001f, rayStart + Vector2.One * 0.001f);
626  GameMain.World.QueryAABB((fixture) =>
627  {
628  if (fixture?.Body.UserData is LevelObject levelObj)
629  {
630  if (!levelObj.Prefab.TakeLevelWallDamage) { return true; }
631  }
632  else if (fixture?.Body == null || fixture.IsSensor)
633  {
634  //ignore sensors
635  return true;
636  }
637  if (fixture.Body.UserData is VineTile) { return true; }
638  if (fixture.CollidesWith == Category.None) { return true; }
639 
640  if (fixture.Body.UserData as string == "ruinroom" || fixture.Body.UserData is Hull || fixture.UserData is Hull) { return true; }
641 
642  //if doing the raycast in a submarine's coordinate space, ignore anything that's not in that sub
643  if (submarine != null)
644  {
645  if (fixture.Body.UserData is VoronoiCell) { return true; }
646  if (fixture.Body.UserData is Entity entity && entity.Submarine != submarine) { return true; }
647  }
648 
649  if (fixture.Body.UserData is VoronoiCell && (this.item.Submarine != null || submarine != null)) { return true; }
650 
651  if (fixture.Body.UserData is Item item)
652  {
653  if (item == Item) { return true; }
654  if (item.Condition <= 0) { return true; }
655  if (!item.Prefab.DamagedByProjectiles && item.GetComponent<Door>() == null) { return true; }
656  }
657  else if (fixture.Body.UserData is Holdable { CanPush: false })
658  {
659  // Ignore holdables that can't push -> shouldn't block
660  return true;
661  }
662  else
663  {
664  // TODO: This might make us ignore something we don't want to ignore?

666  if (!fixture.CollisionCategories.HasFlag(Physics.CollisionCharacter) &&
667  !fixture.CollisionCategories.HasFlag(Physics.CollisionWall) &&
668  !fixture.CollisionCategories.HasFlag(Physics.CollisionLevel)) { return true; }
669  }
670 
671  fixture.Body.GetTransform(out FarseerPhysics.Common.Transform transform);
672  if (!fixture.Shape.TestPoint(ref transform, ref rayStart)) { return true; }
673 
674  hits.Add(new HitscanResult(fixture, rayStart, -dir, 0.0f));
675  return true;
676  }, ref aabb);

679  {

698

702  if (fixture.Body.UserData is VoronoiCell) { return -1; }
703  if (fixture.Body.UserData is Entity entity && entity.Submarine != submarine) { return -1; }
