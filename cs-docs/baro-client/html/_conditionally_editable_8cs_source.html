34  public bool IsEditable(ISerializableEntity entity)
35  {
36  if (onlyInEditors && Screen.Selected is { IsEditor: false }) { return false; }
37 
38  return conditionType switch
39  {
40  ConditionType.AllowLinkingWifiToChat
41  => GameMain.NetworkMember is not { ServerSettings.AllowLinkingWifiToChat: false },
42  ConditionType.IsSwappableItem
43  => entity is Item item && item.Prefab.SwappableItem != null,
44  ConditionType.AllowRotating
45  => (entity is Item { body: null } item && item.Prefab.AllowRotatingInEditor)
46  || (entity is Structure structure && structure.Prefab.AllowRotatingInEditor),
47  ConditionType.Attachable
48  => GetComponent<Holdable>(entity) is Holdable { Attachable: true },
49  ConditionType.HasBody
50  => entity is Structure { HasBody: true } or Item { body: not null },
51  ConditionType.Pickable
52  => entity is Item item && item.GetComponent<Pickable>() != null,
53  ConditionType.OnlyByStatusEffectsAndNetwork
54  => GameMain.NetworkMember is { IsServer: true },
55  ConditionType.HasIntegratedButtons
56  => GetComponent<Door>(entity) is { HasIntegratedButtons: true },
57  ConditionType.IsToggleableController
58  => GetComponent<Controller>(entity) is Controller { IsToggle: true } controller &&
59  controller.Item.GetComponent<ConnectionPanel>() != null,
60  ConditionType.HasConnectionPanel
61  => GetComponent<ConnectionPanel>(entity) != null,
62  _
63  => false
64  };
