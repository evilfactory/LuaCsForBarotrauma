250  if (SnapOnCollision)

253  if (raycastTimer > RayCastInterval)
254  {
255  if (Submarine.PickBody(ConvertUnits.ToSimUnits(source.WorldPosition), ConvertUnits.ToSimUnits(target.WorldPosition),
256  collisionCategory: Physics.CollisionLevel | Physics.CollisionWall,
257  customPredicate: (Fixture f) =>
258  {

260  {
261  Submarine alreadyHitSub = null;
262  if (body.UserData is Structure hitStructure)
263  {
264  alreadyHitSub = hitStructure.Submarine;
265  }
266  else if (body.UserData is Submarine hitSub)
267  {
268  alreadyHitSub = hitSub;
269  }
270  if (alreadyHitSub != null)

272  if (f.Body?.UserData is MapEntity me && me.Submarine == alreadyHitSub) { return false; }
273  if (f.Body?.UserData as Submarine == alreadyHitSub) { return false; }
274  }
275  }

277

281  {
282  return false;
283  }
284  }
285  else if (f.Body?.UserData is Submarine sub)
286  {
287  if (sub == targetSub || sub == source.Submarine)
288  {
289  return false;
290  }
291  }
292  return true;
293  }) != null)

296  return;
297  }

299  }

302  Vector2 forceDir = diff;
303  currentRopeLength = diff.Length();
304  if (currentRopeLength > 0.001f)
305  {
306  forceDir = Vector2.Normalize(forceDir);
307  }
308 
309  if (Math.Abs(ProjectilePullForce) > 0.001f)
310  {
311  projectile.Item?.body?.ApplyForce(-forceDir * ProjectilePullForce);
312  }
313 
314  if (projectile.StickTarget != null)
315  {
316  float targetMass = float.MaxValue;
317  Character targetCharacter = null;
318  switch (projectile.StickTarget.UserData)
319  {
320  case Limb targetLimb:
321  targetCharacter = targetLimb.character;
322  targetMass = targetLimb.ragdoll.Mass;
323  break;
324  case Character character:
325  targetCharacter = character;
326  targetMass = character.Mass;
327  break;
328  case Item _:
329  targetMass = projectile.StickTarget.Mass;
330  break;
331  }
332  if (projectile.StickTarget.BodyType != BodyType.Dynamic)
333  {
334  targetMass = float.MaxValue;
335  }
336  // Currently can only apply pull forces to the source, when it's a character, not e.g. when the item would be auto-operated by an AI. Might have to change this.
337  if (user != null)
338  {
339  if (!snapped &&
340  //user can only hold on to the rope if it was launched from a holdable item, or by something else than an item (limb?)
341  (projectile.Launcher == null || projectile.Launcher.GetComponent<Holdable>() != null))
342  {
343  user.AnimController.HoldToRope();
344  if (targetCharacter != null)
345  {
346  targetCharacter.AnimController.DragWithRope();
347  }
348  if (user.InWater)
349  {
350  user.AnimController.HangWithRope();
351  }
352  }
353  if (Math.Abs(SourcePullForce) > 0.001f && targetMass > TargetMinMass)
354  {
355  // This should be the main collider.
356  var sourceBody = GetBodyToPull(source);
357  if (sourceBody != null)
358  {
359  PhysicsBody targetBody = GetBodyToPull(target);
360  if (sourceBody.UserData is Character)
361  {
362  isReelingIn = user.InWater && user.IsRagdolled || !user.InWater && targetCharacter is { IsIncapacitated: false };
363  if (isReelingIn)
364  {
365  float pullForce = SourcePullForce;
366  if (!user.InWater)
367  {
368  // Apply a tiny amount to the character holding the rope, so that the connection "feels" more real.
369  pullForce *= 0.1f;
370  }
371  float lengthFactor = MathUtils.InverseLerp(0, MaxLength / 2, currentRopeLength);
372  float force = LerpForces ? MathHelper.Lerp(0, pullForce, lengthFactor) : pullForce;
373  sourceBody.ApplyForce(forceDir * force);
374  // Take the target velocity into account.
375  if (targetBody != null)
376  {
377  if (targetCharacter != null)
378  {
379  if (targetBody.LinearVelocity != Vector2.Zero && sourceBody.LinearVelocity != Vector2.Zero)
380  {
381  Vector2 targetDir = Vector2.Normalize(targetBody.LinearVelocity);
382  float movementDot = Vector2.Dot(Vector2.Normalize(sourceBody.LinearVelocity), targetDir);
383  if (movementDot < 0)
384  {
385  // Pushing to a different dir -> add some counter force
386  const float multiplier = 5;
387  float inverseLengthFactor = MathHelper.Lerp(1, 0, lengthFactor);
388  sourceBody.ApplyForce(targetBody.LinearVelocity * Math.Min(targetBody.Mass * multiplier, 250) * sourceBody.Mass * -movementDot * inverseLengthFactor);
389  }
390  float forceDot = Vector2.Dot(forceDir, targetDir);
391  if (forceDot > 0)
392  {
393  // Pulling to the same dir -> add extra force
394  float targetSpeed = targetBody.LinearVelocity.Length();
395  const float multiplier = 25;
396  sourceBody.ApplyForce(forceDir * targetSpeed * sourceBody.Mass * multiplier * forceDot * lengthFactor);
397  }
398  float colliderMainLimbDistance = Vector2.Distance(sourceBody.SimPosition, user.AnimController.MainLimb.SimPosition);
399  const float minDist = 1;
400  const float maxDist = 10;
401  if (colliderMainLimbDistance > minDist && sourceBody.UserData is not Submarine)
402  {
403  // Move the ragdoll closer to the collider, if it's too far (the correction force in HumanAnimController is not enough -> the ragdoll would lag behind and get teleported).
404  float correctionForce = MathHelper.Lerp(10.0f, NetConfig.MaxPhysicsBodyVelocity, MathUtils.InverseLerp(minDist, maxDist, colliderMainLimbDistance));
405  Vector2 targetPos = sourceBody.SimPosition + new Vector2((float)Math.Sin(-sourceBody.Rotation), (float)Math.Cos(-sourceBody.Rotation)) * 0.4f;
406  user.AnimController.MainLimb.MoveToPos(targetPos, correctionForce);
407  }
408  }
409  }
410  else
411  {
412  sourceBody.ApplyForce(targetBody.LinearVelocity * sourceBody.Mass);
413  }
414  }
415  }
416  }
417  else
418  {
419  float distance = Vector2.Distance(source.WorldPosition, target.WorldPosition);
420  float force = LerpForces ? MathHelper.Lerp(0, SourcePullForce, MathUtils.InverseLerp(0, MaxLength / 2, distance)) : SourcePullForce;
421  sourceBody.ApplyForce(forceDir * force);
422  }
423  }
424  }
425  }
426 
427  if (Math.Abs(TargetPullForce) > 0.001f && user is not { IsRagdolled: true})
428  {
429  PhysicsBody targetBody = GetBodyToPull(target);
430  if (targetBody == null) { return; }
431  bool lerpForces = LerpForces;
432  float maxVelocity = NetConfig.MaxPhysicsBodyVelocity * 0.25f;
433  // The distance where we start pulling with max force.
434  float maxPullDistance = MaxLength / 3;
435  float minPullDistance = MinPullDistance;
436  const float absoluteMinPullDistance = 50;
437  if (targetCharacter != null)
438  {
439  if (targetCharacter.IsRagdolled || targetCharacter.IsUnconscious)
440  {
441  if (!targetCharacter.InWater)
442  {
443  // Limits the velocity of ragdolled characters on ground/air, because otherwise they tend to move with too high forces.
444  maxVelocity = NetConfig.MaxPhysicsBodyVelocity * 0.075f;
445  }
446  }
447  else
448  {
449  // Target alive and kicking -> Use the absolute min pull distance and full forces to pull.
450  // Keep some lerping, because it results into smoothing when the target is close by.
451  minPullDistance = absoluteMinPullDistance;
452  maxPullDistance = 200;
453  }
454  }
455  minPullDistance = MathHelper.Max(minPullDistance, absoluteMinPullDistance);
456  if (currentRopeLength < minPullDistance) { return; }
457  maxPullDistance = MathHelper.Max(minPullDistance * 2, maxPullDistance);
458  float force = lerpForces
459  ? MathHelper.Lerp(0, TargetPullForce, MathUtils.InverseLerp(minPullDistance, maxPullDistance, currentRopeLength))
460  : TargetPullForce;
461  targetBody.ApplyForce(-forceDir * force, maxVelocity);
462  AnimController targetRagdoll = targetCharacter?.AnimController;
463  if (targetRagdoll?.Collider != null)
464  {
465  isReelingIn = true;
466  if (targetRagdoll.InWater || targetRagdoll.OnGround)
467  {
468  float forceMultiplier = 1;
469  if (!targetCharacter.IsRagdolled && !targetCharacter.IsIncapacitated && IncreaseForceForEscapingTargets)
470  {
471  // Pulling the main collider requires higher forces when the target is trying to move away.
472  Vector2 targetMovement = targetCharacter.AnimController.TargetMovement;
473  float dot = Vector2.Dot(Vector2.Normalize(targetMovement), forceDir);
474  if (dot > 0)
475  {
476  const float constMultiplier = 2.5f;
477  float targetVelocity = targetMovement.Length();
478  float massFactor = Math.Max((float)Math.Log(targetCharacter.Mass / 10), 1);
479  forceMultiplier = Math.Max(targetVelocity * massFactor * constMultiplier * dot, 1);
480  }
481  }
482  targetRagdoll.Collider.ApplyForce(-forceDir * force * forceMultiplier, maxVelocity);
483  }
484  }
485  }
