127  private void OnP2PData(P2PEndpoint senderEndpoint, IReadMessage inc)
128  {
129  if (!isActive) { return; }
130 
131  receivedBytes += inc.LengthBytes;
132 
133  if (senderEndpoint != ServerEndpoint) { return; }
134 
135  timeout = Screen.Selected == GameMain.GameScreen
136  ? NetworkConnection.TimeoutThresholdInGame
137  : NetworkConnection.TimeoutThreshold;
138 
139  var (_, packetHeader, initialization) = INetSerializableStruct.Read<PeerPacketHeaders>(inc);
140 
141  if (!packetHeader.IsServerMessage()) { return; }
142 
143  if (packetHeader.IsConnectionInitializationStep())
144  {
145  if (!initialization.HasValue) { return; }
146 
147  var relayPacket = INetSerializableStruct.Read<P2PInitializationRelayPacket>(inc);
148 
149  if (initializationStep != ConnectionInitialization.Success)
150  {
151  incomingInitializationMessages.Add(new IncomingInitializationMessage
152  {
153  InitializationStep = initialization.Value,
154  Message = relayPacket.Message.GetReadMessageUncompressed()
155  });
156  }
157  }
158  else if (packetHeader.IsDataFragment())
159  {
160  var completeMessageOption = defragmenter.ProcessIncomingFragment(INetSerializableStruct.Read<MessageFragment>(inc));
161  if (!completeMessageOption.TryUnwrap(out var completeMessage)) { return; }
162 
163  int completeMessageLengthBits = completeMessage.Length * 8;
164  incomingDataMessages.Add(new ReadWriteMessage(completeMessage.ToArray(), 0, completeMessageLengthBits, copyBuf: false));
165  }
166  else if (packetHeader.IsHeartbeatMessage())
167  {
168  return; //TODO: implement heartbeats
169  }
170  else if (packetHeader.IsDisconnectMessage())
171  {
172  PeerDisconnectPacket packet = INetSerializableStruct.Read<PeerDisconnectPacket>(inc);
173  Close(packet);
174  }
175  else
176  {
177  var packet = INetSerializableStruct.Read<PeerPacketMessage>(inc);
178  incomingDataMessages.Add(packet.GetReadMessage(packetHeader.IsCompressed(), ServerConnection!));
179  }

369  private void ForwardToRemotePeer(IWriteMessage msg, DeliveryMethod deliveryMethod)
370  {
371  if (!isActive) { return; }
372  if (socket is null) { return; }
373 
374  heartbeatTimer = 5.0;
375 
376  int length = msg.LengthBytes;
377 
378  if (length + 4 >= MsgConstants.MTU)
379  {
380  DebugConsole.Log($"WARNING: message length comes close to exceeding MTU, forcing reliable send ({length} bytes)");
381  deliveryMethod = DeliveryMethod.Reliable;
382  }
383 
384  bool success = socket.SendMessage(ServerEndpoint, msg, deliveryMethod);
385 
386  sentBytes += length;
387 
388  if (success) { return; }
389 
390  if (deliveryMethod is DeliveryMethod.Unreliable)
391  {
392  DebugConsole.Log($"WARNING: message couldn't be sent unreliably, forcing reliable send ({length} bytes)");
393  success = socket.SendMessage(ServerEndpoint, msg, DeliveryMethod.Reliable);
394  sentBytes += length;
395  }
396 
397  if (!success)
398  {
399  DebugConsole.AddWarning($"Failed to send message to remote peer! ({length} bytes)");
400  }
401  }
