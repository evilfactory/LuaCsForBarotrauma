247  {
248  //refresh the character data of clients who are still in the server
249  foreach (Client c in GameMain.Server.ConnectedClients)
250  {
251  //ignore if the character is controlling a monster
252  //(we'll just use the previously saved campaign data if there's any)
253  if (c.Character != null && c.Character.Info == null)
254  {
255  c.Character = null;
256  }
257  //use the info of the character the client is currently controlling
258  // or the previously saved info if not (e.g. if the client has been spectating or died)
259  var characterInfo = c.Character?.Info;
260  var matchingCharacterData = characterData.Find(d => d.MatchesClient(c));
261  if (matchingCharacterData != null)
262  {
263  //hasn't spawned this round -> don't touch the data
264  if (!matchingCharacterData.HasSpawned) { continue; }
265  characterInfo ??= matchingCharacterData.CharacterInfo;
266  }
267  if (characterInfo == null || characterInfo.Discarded) { continue; }
268  //reduce skills if the character has died
269  if (characterInfo.CauseOfDeath != null && characterInfo.CauseOfDeath.Type != CauseOfDeathType.Disconnected)
270  {
271  characterInfo.ApplyDeathEffects();
272  }
273  c.CharacterInfo = characterInfo;
274  SetClientCharacterData(c);
275  }
276 
277  //refresh the character data of clients who aren't in the server anymore
278  List<CharacterCampaignData> prevCharacterData = new List<CharacterCampaignData>(characterData);
279  foreach (CharacterCampaignData data in prevCharacterData)
280  {
281  if (data.HasSpawned && !GameMain.Server.ConnectedClients.Any(c => data.MatchesClient(c)))
282  {
283  var character = Character.CharacterList.Find(c => c.Info == data.CharacterInfo && !c.IsHusk);
284  if (character != null &&
285  (!character.IsDead || character.CauseOfDeath?.Type == CauseOfDeathType.Disconnected))
286  {
287  //character still alive (or killed by Disconnect) -> save it as-is
288  characterData.RemoveAll(cd => cd.IsDuplicate(data));
289  data.Refresh(character, refreshHealthData: character.CauseOfDeath?.Type != CauseOfDeathType.Disconnected);
290  characterData.Add(data);
291  }
292  else
293  {
294  //character dead or removed -> reduce skills, remove items, health data, etc
295  data.CharacterInfo.ApplyDeathEffects();
296  data.Reset();
297  }
298  }
299  }
300 
301  MoveDiscardedCharacterBalancesToBank();
302 
303  characterData.ForEach(cd => cd.HasSpawned = false);
304  foreach (var cd in characterData)
305  {
306  //remove from crewmanager - we don't need to save the data there if it's been saved as CharacterCampaignData
307  //(e.g. if a client has taken over a bot, we need to do this to prevent it being saved twice)
308  CrewManager.RemoveCharacterInfo(cd.CharacterInfo);
309  }
310 
311  SavePets();
312 
313  //remove all items that are in someone's inventory
314  foreach (Character c in Character.CharacterList)
315  {
316  if (c.Inventory == null) { continue; }
317  if (Level.Loaded.Type == LevelData.LevelType.Outpost && c.Submarine != Level.Loaded.StartOutpost)
318  {
319  Map.CurrentLocation.RegisterTakenItems(c.Inventory.AllItems.Where(it => it.SpawnedInCurrentOutpost && it.OriginalModuleIndex > 0));
320  }
321 
322  if (c.Info != null && c.IsBot)
323  {
324  if (c.IsDead && c.CauseOfDeath?.Type != CauseOfDeathType.Disconnected) { CrewManager.RemoveCharacterInfo(c.Info); }
325  c.Info.HealthData = new XElement("health");
326  c.CharacterHealth.Save(c.Info.HealthData);
327  c.Info.InventoryData = new XElement("inventory");

333  }

353  protected override IEnumerable<CoroutineStatus> DoLevelTransition(TransitionType transitionType, LevelData newLevel, Submarine leavingSub, bool mirror)
354  {
355  IncrementAllLastUpdateIds();
356 
357  switch (transitionType)
358  {
359  case TransitionType.None:
360  throw new InvalidOperationException("Level transition failed (no transitions available).");
361  case TransitionType.ReturnToPreviousLocation:
362  //deselect destination on map
363  map.SelectLocation(-1);
364  break;
365  case TransitionType.ProgressToNextLocation:
366  Map.MoveToNextLocation();
367  TotalPassedLevels++;
368  break;
369  case TransitionType.End:
370  EndCampaign();
371  IsFirstRound = true;
372  break;
373  case TransitionType.ProgressToNextEmptyLocation:
374  Map.Visit(Map.CurrentLocation);
375  TotalPassedLevels++;
376  break;
377  }
378 
379  Map.ProgressWorld(this, transitionType, GameMain.GameSession.RoundDuration);
380 
381  bool success = GameMain.Server.ConnectedClients.Any(c => c.InGame && c.Character != null && !c.Character.IsDead);
382  if (success)
383  {
384  foreach (Client c in GameMain.Server.ConnectedClients)
385  {
386  if (c.Character?.HasAbilityFlag(AbilityFlags.RetainExperienceForNewCharacter) ?? false)
387  {
388  (GameMain.GameSession?.GameMode as MultiPlayerCampaign)?.SaveExperiencePoints(c);
389  }
390  }
391  // Event history must be registered before ending the round or it will be cleared
392  GameMain.GameSession.EventManager.RegisterEventHistory();
393  }
394 
395  //store the currently active missions at this point so we can communicate their states to clients, they're cleared in EndRound
396  List<Mission> missions = GameMain.GameSession.Missions.ToList();
397 
398  GameMain.GameSession.EndRound("", transitionType);
399 
400  //--------------------------------------
401 
402  if (success)
403  {
404  SavePlayers();
405  yield return CoroutineStatus.Running;
406  LeaveUnconnectedSubs(leavingSub);
407  NextLevel = newLevel;
408  GameMain.GameSession.SubmarineInfo = new SubmarineInfo(GameMain.GameSession.Submarine);
409  SaveUtil.SaveGame(GameMain.GameSession.SavePath);
410  }
411  else
412  {
413  PendingSubmarineSwitch = null;
414  GameMain.Server.EndGame(TransitionType.None, wasSaved: false);
415  LoadCampaign(GameMain.GameSession.SavePath, client: null);
416  LastSaveID++;
417  IncrementAllLastUpdateIds();
418  yield return CoroutineStatus.Success;
419  }
420 
421  CrewManager?.ClearCurrentOrders();
422 
423  //--------------------------------------
424 
425  GameMain.Server.EndGame(transitionType, wasSaved: true, missions);
426 
427  ForceMapUI = false;
428 
429  NextLevel = newLevel;
430  MirrorLevel = mirror;
431 
432 
433  yield return new WaitForSeconds(EndTransitionDuration * 0.5f);
434 
435  //don't start the next round automatically if we just finished the campaign
436  if (transitionType != TransitionType.End)
437  {
438  GameMain.Server.TryStartGame();
439  }
440 
441  yield return CoroutineStatus.Success;
442  }

655  msg.WriteBoolean(ForceMapUI);

658  msg.WriteUInt16(map.SelectedLocationIndex == -1 ? UInt16.MaxValue : (UInt16)map.SelectedLocationIndex);

666  if (mission.Locations[0] == mission.Locations[1])
667  {
668  msg.WriteByte((byte)255);
669  }
