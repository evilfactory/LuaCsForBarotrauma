284  {
285  // Does not auto-equip the item if specified and no suitable any slot found (for example handcuffs are not auto-equipped)
286  if (item.AllowedSlots.Contains(InvSlotType.Any))
287  {
288  var wearable = item.GetComponent<Wearable>();
289  if (wearable != null && !wearable.AutoEquipWhenFull && !IsAnySlotAvailable(item))
290  {
291  return false;
292  }
293  }
294 
295  if (allowedSlots != null && allowedSlots.Any() && !allowedSlots.Contains(InvSlotType.Any))
296  {
297  bool allSlotsTaken = true;
298  foreach (var allowedSlot in allowedSlots)
299  {
300  if (allowedSlot == (InvSlotType.RightHand | InvSlotType.LeftHand))
301  {
302  int rightHandSlot = FindLimbSlot(InvSlotType.RightHand);
303  int leftHandSlot = FindLimbSlot(InvSlotType.LeftHand);
304  if (rightHandSlot > -1 && slots[rightHandSlot].CanBePut(item) &&
305  leftHandSlot > -1 && slots[leftHandSlot].CanBePut(item))
306  {
307  allSlotsTaken = false;
308  break;
309  }
310  }
311  else
312  {
313  int slot = FindLimbSlot(allowedSlot);
314  if (slot > -1 && slots[slot].CanBePut(item))
315  {
316  allSlotsTaken = false;
317  break;
318  }
319  }
320 
321  }
322  if (allSlotsTaken)
323  {
324  int slot = FindLimbSlot(allowedSlots.First());
325  if (slot > -1 && slots[slot].Items.Any(it => it != item) && slots[slot].First().AllowDroppingOnSwapWith(item))
326  {
327  foreach (Item existingItem in slots[slot].Items.ToList())
328  {
329  existingItem.Drop(user);
330  if (existingItem.ParentInventory != null) { existingItem.ParentInventory.RemoveItem(existingItem); }
331  }
332  }
333  }

342  public override bool TryPutItem(Item item, Character user, IEnumerable<InvSlotType> allowedSlots = null, bool createNetworkEvent = true, bool ignoreCondition = false)
343  {
344  if (allowedSlots == null || !allowedSlots.Any()) { return false; }
345  if (item == null)
346  {
347 #if DEBUG
348  throw new Exception("item null");
349 #else
350  return false;
351 #endif
352  }
353  if (item.Removed)
354  {
355 #if DEBUG
356  throw new Exception("Tried to put a removed item (" + item.Name + ") in an inventory");
357 #else
358  DebugConsole.ThrowError("Tried to put a removed item (" + item.Name + ") in an inventory.\n" + Environment.StackTrace.CleanupStackTrace());
359  return false;
360 #endif
361  }
362 
363  if (item.GetComponent<Pickable>() == null || item.AllowedSlots.None()) { return false; }
364 
365  int currentSlot = -1;
366  bool inWrongSlot = false;
367  bool inSuitableSlot = false;
368 
369  // verify item's current placement
370  for (int slotIndex = 0; slotIndex < capacity; slotIndex++)
371  {
372  if (!slots[slotIndex].Contains(item)) { continue; }
373 
374  // item is at least in this slot, can be in many
375  currentSlot = slotIndex;
376 
377  var firstMatchingSlotType = allowedSlots.FirstOrDefault(slot => slot.HasFlag(SlotTypes[slotIndex]));
378 
379  if (firstMatchingSlotType == default) // if (firstMatchingSlotType == InvSlotType.None)
380  {
381  inWrongSlot = true;
382  break;
383  }
384 
385  inSuitableSlot = true;
386 
387  // can have more than one flag, such as InvSlotType.InnerClothes | InvSlotType.OuterClothes
388  var individualFlags = EnumExtensions.GetIndividualFlags(firstMatchingSlotType);
389 
390  // if item is not in ALL required slot types
391  foreach (var flag in individualFlags)
392  {
393  if (flag == InvSlotType.None) { continue; }
394  if (!IsInLimbSlot(item, flag))
395  {
396  inSuitableSlot = false;
397  break;
398  }
399  }
400  }
401 
402  //all good
403  if (inSuitableSlot && !inWrongSlot) { return true; }
404 
405  //try to place the item in a LimbSlot.Any slot if that's allowed
406  if (allowedSlots.Contains(InvSlotType.Any) && item.AllowedSlots.Contains(InvSlotType.Any))
407  {
408  int freeIndex = GetFreeAnySlot(item, inWrongSlot);
409  if (freeIndex > -1)
410  {
411  PutItem(item, freeIndex, user, true, createNetworkEvent);
412  item.Unequip(character);
413  return true;
414  }
415  }
416 
417  int placedInSlot = -1;
418  foreach (InvSlotType allowedSlot in allowedSlots)
419  {
420  if (allowedSlot.HasFlag(InvSlotType.RightHand) && character.AnimController.GetLimb(LimbType.RightHand) == null) { continue; }
421  if (allowedSlot.HasFlag(InvSlotType.LeftHand) && character.AnimController.GetLimb(LimbType.LeftHand) == null) { continue; }
422 
423  //check if all the required slots are free
424  bool free = true;
425  for (int i = 0; i < capacity; i++)
426  {
427  if (allowedSlot.HasFlag(SlotTypes[i]) && item.AllowedSlots.Any(s => s.HasFlag(SlotTypes[i])) && slots[i].Items.Any(it => it != item))
428  {
429  if (!slots[i].First().AllowedSlots.Contains(InvSlotType.Any) || !TryPutItem(slots[i].FirstOrDefault(), character, new List<InvSlotType> { InvSlotType.Any }, true, ignoreCondition))
430  {
431  free = false;
432 #if CLIENT
433  for (int j = 0; j < capacity; j++)
434  {
435  if (visualSlots != null && slots[j] == slots[i]) { visualSlots[j].ShowBorderHighlight(GUIStyle.Red, 0.1f, 0.9f); }
436  }
437 #endif
438  }
439  }
440  }
441 
442  if (!free) { continue; }
443 
444  for (int i = 0; i < capacity; i++)
445  {
446  if (allowedSlot.HasFlag(SlotTypes[i]) && item.GetComponents<Pickable>().Any(p => p.AllowedSlots.Any(s => s.HasFlag(SlotTypes[i]))) && slots[i].Empty())
447  {
448  bool removeFromOtherSlots = item.ParentInventory != this;
449  if (placedInSlot == -1 && inWrongSlot)
450  {
451  if (!slots[i].HideIfEmpty || SlotTypes[currentSlot] != InvSlotType.Any) { removeFromOtherSlots = true; }
452  }
453 
454  PutItem(item, i, user, removeFromOtherSlots, createNetworkEvent);
455  item.Equip(character);
456  placedInSlot = i;
457  }
458  }
459  if (placedInSlot > -1) { break; }
460  }
461 
462  return placedInSlot > -1;
463  }

469  private int GetFreeAnySlot(Item item, bool inWrongSlot)
470  {
471  //attempt to stack first
472  for (int i = 0; i < capacity; i++)
473  {
474  if (SlotTypes[i] != InvSlotType.Any) { continue; }
475  if (!slots[i].Empty() && CanBePutInSlot(item, i))
476  {
477  return i;
478  }
479  }
480  for (int i = 0; i < capacity; i++)
481  {
482  if (SlotTypes[i] != InvSlotType.Any) { continue; }
483  if (slots[i].Contains(item))
484  {
485  return i;
486  }
487  }
488  for (int i = 0; i < capacity; i++)
489  {
490  if (SlotTypes[i] != InvSlotType.Any) { continue; }
491  if (CanBePutInSlot(item, i))
492  {
493  return i;
494  }
495  }
496  for (int i = 0; i < capacity; i++)
497  {
498  if (SlotTypes[i] != InvSlotType.Any) { continue; }
499  if (inWrongSlot)
500  {
501  //another item already in the slot
502  if (slots[i].Any() && slots[i].Items.Any(it => it != item)) { continue; }
503  }
504  else
505  {
506  if (!CanBePutInSlot(item, i)) { continue; }
507  }
508  return i;
509  }
510  return -1;
511  }

513  public override bool TryPutItem(Item item, int index, bool allowSwapping, bool allowCombine, Character user, bool createNetworkEvent = true, bool ignoreCondition = false)
514  {
515  if (index < 0 || index >= slots.Length)
516  {
517  string errorMsg = "CharacterInventory.TryPutItem failed: index was out of range(" + index + ").\n" + Environment.StackTrace.CleanupStackTrace();
518  GameAnalyticsManager.AddErrorEventOnce("CharacterInventory.TryPutItem:IndexOutOfRange", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
519  return false;
520  }
521  //there's already an item in the slot
522  if (slots[index].Any())
523  {
524  if (slots[index].Contains(item)) { return false; }
525  return base.TryPutItem(item, index, allowSwapping, allowCombine, user, createNetworkEvent, ignoreCondition);
526  }
527 
528  if (SlotTypes[index] == InvSlotType.Any)
529  {
530  if (!item.GetComponents<Pickable>().Any(p => p.AllowedSlots.Contains(InvSlotType.Any))) { return false; }
531  if (slots[index].Any()) { return slots[index].Contains(item); }
532  PutItem(item, index, user, true, createNetworkEvent);
533  return true;
534  }
535 
536  InvSlotType placeToSlots = InvSlotType.None;
537 
538  bool slotsFree = true;
539  foreach (Pickable pickable in item.GetComponents<Pickable>())
540  {
541  foreach (InvSlotType allowedSlot in pickable.AllowedSlots)
542  {
543  if (!allowedSlot.HasFlag(SlotTypes[index])) { continue; }
544  for (int i = 0; i < capacity; i++)
545  {
546  if (allowedSlot.HasFlag(SlotTypes[i]) && slots[i].Any() && !slots[i].Contains(item))
547  {
548  slotsFree = false;
549  break;
550  }
551  placeToSlots = allowedSlot;
552  }
553  }
554  }
555 
556  if (!slotsFree) { return false; }
557 
558  return TryPutItem(item, user, new List<InvSlotType>() { placeToSlots }, createNetworkEvent, ignoreCondition);
559  }
