280  public ItemContainer(Item item, ContentXElement element)
281  : base(item, element)
282  {
283  int totalCapacity = capacity;
284 
285  foreach (var subElement in element.Elements())
286  {
287  switch (subElement.Name.ToString().ToLowerInvariant())
288  {
289  case "containable":
290  RelatedItem containable = RelatedItem.Load(subElement, returnEmpty: false, parentDebugName: item.Name);
291  if (containable == null)
292  {
293  DebugConsole.ThrowError("Error in item config \"" + item.ConfigFilePath + "\" - containable with no identifiers.",
294  contentPackage: element.ContentPackage);
295  continue;
296  }
297  ContainableItems ??= new List<RelatedItem>();
298  ContainableItems.Add(containable);
299  break;
300  case "subcontainer":
301  totalCapacity += subElement.GetAttributeInt("capacity", 1);
302  HasSubContainers = true;
303  break;
304  }
305  }
306  Inventory = new ItemInventory(item, this, totalCapacity, SlotsPerRow);
307 
308  // we have to assign this here because the fields are serialized before the inventory is created otherwise
309  ExtraStackSize = element.GetAttributeInt(nameof(ExtraStackSize), 0);
310 
311  List<SlotRestrictions> newSlotRestrictions = new List<SlotRestrictions>(totalCapacity);
312  for (int i = 0; i < capacity; i++)
313  {
314  newSlotRestrictions.Add(new SlotRestrictions(maxStackSize, ContainableItems, autoInject: false));
315  }
316 
317  int subContainerIndex = capacity;
318  foreach (var subElement in element.Elements())
319  {
320  if (subElement.Name.ToString().ToLowerInvariant() != "subcontainer") { continue; }
321 
322  int subCapacity = subElement.GetAttributeInt("capacity", 1);
323  int subMaxStackSize = subElement.GetAttributeInt("maxstacksize", maxStackSize);
324  bool autoInject = subElement.GetAttributeBool("autoinject", false);
325 
326  var subContainableItems = new List<RelatedItem>();
327  foreach (var subSubElement in subElement.Elements())
328  {
329  if (subSubElement.Name.ToString().ToLowerInvariant() != "containable") { continue; }
330 
331  RelatedItem containable = RelatedItem.Load(subSubElement, returnEmpty: false, parentDebugName: item.Name);
332  if (containable == null)
333  {
334  DebugConsole.ThrowError("Error in item config \"" + item.ConfigFilePath + "\" - containable with no identifiers.",
335  contentPackage: element.ContentPackage);
336  continue;
337  }
338  subContainableItems.Add(containable);
339  AllSubContainableItems ??= new List<RelatedItem>();
340  AllSubContainableItems.Add(containable);
341  }
342 
343  for (int i = subContainerIndex; i < subContainerIndex + subCapacity; i++)
344  {
345  newSlotRestrictions.Add(new SlotRestrictions(subMaxStackSize, subContainableItems, autoInject));
346  }
347  subContainerIndex += subCapacity;
348  }

398  public void OnItemContained(Item containedItem)
399  {
400  int index = Inventory.FindIndex(containedItem);
401  RelatedItem relatedItem = null;
402  if (index >= 0 && index < slotRestrictions.Length)
403  {
404  if (slotRestrictions[index].ContainableItems != null)
405  {
406  activeContainedItems.RemoveAll(i => i.Item == containedItem);
407  foreach (var containableItem in slotRestrictions[index].ContainableItems)
408  {
409  if (!containableItem.MatchesItem(containedItem)) { continue; }
410  //the 1st matching ContainableItem of the slot determines the hiding, position and rotation of the item
411  relatedItem ??= containableItem;
412  foreach (StatusEffect effect in containableItem.StatusEffects)
413  {
414  activeContainedItems.Add(new ActiveContainedItem(containedItem, effect, containableItem.ExcludeBroken, containableItem.ExcludeFullCondition));
415  }
416  }
417  }
418  }
419 
420  var containedItemInfo = new ContainedItem(containedItem,
421  Hide: relatedItem?.Hide ?? false,
422  ItemPos: relatedItem?.ItemPos,
423  Rotation: relatedItem?.Rotation ?? 0.0f);
424  containedItems.RemoveAll(d => d.Item == containedItem);
425 
426  if (hideItems)
427  {
428  //if the items aren't visible, the draw order doesn't matter and we can skip the sorting
429  containedItems.Add(containedItemInfo);
430  }
431  else
432  {
433  int containedIndex = 0;
434  while (containedIndex < containedItems.Count)
435  {
436  if (index <= Inventory.FindIndex(containedItems[containedIndex].Item))
437  {
438  break;
439  }
440  containedIndex++;
441  }
442  //sort drawables by their order in the inventory
443  containedItems.Insert(containedIndex, containedItemInfo);
444  }
445 
446  if (item.GetComponent<Planter>() != null)
447  {
448  GameAnalyticsManager.AddDesignEvent("MicroInteraction:" + (GameMain.GameSession?.GameMode?.Preset.Identifier.Value ?? "null") + ":GardeningPlanted:" + containedItem.Prefab.Identifier);
449  }
450 
451  //no need to Update() if this item has no statuseffects and no physics body
452  IsActive = activeContainedItems.Count > 0 || Inventory.AllItems.Any(it => it.body != null);
453 
454  if (IsActive && item.GetRootInventoryOwner() is Character owner &&
455  owner.HasEquippedItem(item, predicate: slot => slot.HasFlag(InvSlotType.LeftHand) || slot.HasFlag(InvSlotType.RightHand)))
456  {
457  // Set the contained items active if there's an item inserted inside the container. Enables e.g. the rifle flashlight when it's attached to the rifle (put inside of it).
458  SetContainedActive(true);
459  }
460  if (containedItem.FlippedX != item.FlippedX)
461  {
462  containedItem.FlipX(relativeToSub: false);
463  }
464  if (containedItem.FlippedY != item.FlippedY)
465  {
466  containedItem.FlipY(relativeToSub: false);
467  }
468  item.SetContainedItemPositions();
469  CharacterHUD.RecreateHudTextsIfFocused(item, containedItem);
470  OnContainedItemsChanged.Invoke(this);
471  }

540  public override void Update(float deltaTime, Camera cam)
541  {
542  if (!string.IsNullOrEmpty(SpawnWithId) && !alwaysContainedItemsSpawned)
543  {
544  SpawnAlwaysContainedItems();
545  alwaysContainedItemsSpawned = true;
546  }
547 
548  if (item.ParentInventory is CharacterInventory ownerInventory)
549  {
550  SetContainedItemPositionsIfNeeded();
551 
552  if (AutoInject || slotRestrictions.Any(s => s.AutoInject))
553  {
554  //normally autoinjection should delete the (medical) item, so it only gets applied once
555  //but in multiplayer clients aren't allowed to remove items themselves, so they may be able to trigger this dozens of times
556  //before the server notifies them of the item being removed, leading to a sharp lag spike.
557  //this can also happen with mods, if there's a way to autoinject something that doesn't get removed On Use.
558  //so let's ensure the item is only applied once per second at most.
559 
560  autoInjectCooldown -= deltaTime;
561  if (autoInjectCooldown <= 0.0f &&
562  ownerInventory?.Owner is Character ownerCharacter &&
563  ownerCharacter.HealthPercentage / 100f <= AutoInjectThreshold &&
564  ownerCharacter.HasEquippedItem(item))
565  {
566  if (AutoInject)
567  {
568  Inventory.AllItemsMod.ForEach(i => Inject(i));
569  }
570  else
571  {
572  for (int i = 0; i < slotRestrictions.Length; i++)
573  {
574  if (slotRestrictions[i].AutoInject)
575  {
576  Inventory.GetItemsAt(i).ForEachMod(i => Inject(i));
577  }
578  }
579  }
580  void Inject(Item item)
581  {
582  item.ApplyStatusEffects(ActionType.OnSuccess, 1.0f, ownerCharacter, useTarget: ownerCharacter);
583  item.ApplyStatusEffects(ActionType.OnUse, 1.0f, ownerCharacter, useTarget: ownerCharacter);
584  item.GetComponent<GeneticMaterial>()?.Equip(ownerCharacter);
585  }
586  autoInjectCooldown = AutoInjectInterval;
587  }
588  }
589 
590  }
591  else if (item.body != null && item.body.Enabled)
592  {
593  if (item.body.FarseerBody.Awake)
594  {
595  SetContainedItemPositionsIfNeeded();
596  }
597  }
598  else if (activeContainedItems.Count == 0)
599  {
600  IsActive = false;
601  return;
602  }
603 
604  foreach (var activeContainedItem in activeContainedItems)
605  {
606  Item contained = activeContainedItem.Item;
607 
608  if (activeContainedItem.ExcludeBroken && contained.Condition <= 0.0f) { continue; }
609  if (activeContainedItem.ExcludeFullCondition && contained.IsFullCondition) { continue; }
610  StatusEffect effect = activeContainedItem.StatusEffect;
611 
612  targets.Clear();
613  bool wearing = item.GetComponent<Wearable>() is Wearable { IsActive: true };
614  if (effect.HasTargetType(StatusEffect.TargetType.This))
615  {
616  targets.AddRange(item.AllPropertyObjects);
617  }
618  if (effect.HasTargetType(StatusEffect.TargetType.Contained))
619  {
620  targets.AddRange(contained.AllPropertyObjects);
621  }
622  if (effect.HasTargetType(StatusEffect.TargetType.Character) && item.ParentInventory?.Owner is Character character)
623  {
624  targets.Add(character);
625  }
626  if (effect.HasTargetType(StatusEffect.TargetType.NearbyItems) ||
627  effect.HasTargetType(StatusEffect.TargetType.NearbyCharacters))
628  {
629  effect.AddNearbyTargets(item.WorldPosition, targets);
630  }
631  effect.Apply(ActionType.OnActive, deltaTime, item, targets);
632  effect.Apply(ActionType.OnContaining, deltaTime, item, targets);
633  if (wearing) { effect.Apply(ActionType.OnWearing, deltaTime, item, targets); }
634  }

670  {
