67  // Otherwise they may have an item in their inventory they shouldn't have been able to pick up,
68  // and receiving an event for that inventory later will cause the item to be dropped
69  CreateNetworkEvent();

84  Dictionary<Item, bool> GetItemAccessibility()
85  {
86  Dictionary<Item, bool> itemAccessibility = new Dictionary<Item, bool>();
87 
88  for (int i = 0; i < capacity; i++)
89  {
90  // for every item that the new inventory state contains
91  foreach (ushort itemId in receivedItemIdsFromClient[i])
92  {
93  // if there is no such item, skip
94  if (Entity.FindEntityByID(itemId) is not Item item) { continue; }
95  // add entry: can the sender access the item?
96  itemAccessibility[item] = item.CanClientAccess(sender);
97  }
98  }
99 
100  // we now have accessibility for every item in the new inventory state
101  // but not for the items that were in the inventory before and perhaps dropped, so let's add those as well
102  foreach (var item in prevItems)
103  {
104  if (!itemAccessibility.ContainsKey(item))
105  {
106  itemAccessibility[item] = item.CanClientAccess(sender);
107  }
108  }
109 
110  return itemAccessibility;

113  void HandleRemovedItems()
114  {
115  for (int slotIndex = 0; slotIndex < capacity; slotIndex++)
116  {
117  foreach (Item item in slots[slotIndex].Items.ToList())
118  {
119  bool shouldBeRemoved = !receivedItemIdsFromClient[slotIndex].Contains(item.ID) &&
120  item.IsInteractable(sender.Character); // item is interactable to sender: not hidden and player team
121  if (shouldBeRemoved)
122  {
123  bool itemAccessDenied = prevItems.Contains(item) && // if the item was in the inventory before
124  !itemAccessibility[item] && // and the sender is not allowed to access it
125  (item.PreviousParentInventory == null || // and either the item has no previous inventory
126  !sender.Character.CanAccessInventory(item.PreviousParentInventory)); // or the sender can't access the previous inventory
127 
128  if (itemAccessDenied)
129  {
130 #if DEBUG || UNSTABLE
131  DebugConsole.NewMessage($"Client {sender.Name} failed to drop item \"{item}\" (parent inventory: {item.ParentInventory?.Owner.ToString() ?? "null"}). No access.", Color.Yellow);
132 #endif
133  continue;
134  }
135 
136  Item droppedItem = item;
137  Entity prevOwner = Owner;
138  Inventory previousInventory = droppedItem.ParentInventory;
139  droppedItem.Drop(null);
140  droppedItem.PreviousParentInventory = previousInventory;
141 
142  var previousCharacterInventory = prevOwner switch
143  {
144  Item itemInventory => itemInventory.FindParentInventory(inventory => inventory is CharacterInventory) as CharacterInventory,
145  Character character => character.Inventory,
146  _ => null
147  };
148 
149  if (previousCharacterInventory != null && previousCharacterInventory != sender.Character?.Inventory)
150  {
151  GameMain.Server?.KarmaManager.OnItemTakenFromPlayer(previousCharacterInventory, sender, droppedItem);
152  }
153 
154  if (droppedItem.body != null && prevOwner != null)
155  {
156  droppedItem.body.SetTransform(prevOwner.SimPosition, 0.0f);
157  }
158  }
159  }
160 
161  foreach (ushort id in receivedItemIdsFromClient[slotIndex])
162  {
163  Item newItem = id == 0 ? null : Entity.FindEntityByID(id) as Item;
164  prevItemInventories.Add(newItem?.ParentInventory);

169  void HandleAddedItems()

172  {

176

178  (pickable.IsAttached && !pickable.PickingDone) || item.AllowedSlots.None() || !item.IsInteractable(sender.Character))
179  {
180  DebugConsole.AddWarning($"Client {sender.Name} tried to pick up a non-pickable item \"{item}\" (parent inventory: {item.ParentInventory?.Owner.ToString() ?? "null"})",
181  item.Prefab.ContentPackage);
182  continue;
183  }
184 
185  if (GameMain.Server != null)
186  {
187  var holdable = item.GetComponent<Holdable>();
188  if (holdable != null && !holdable.CanBeDeattached()) { continue; }
189 
190  bool itemAccessDenied = !prevItems.Contains(item) && !itemAccessibility[item] &&
191  (sender.Character == null || item.PreviousParentInventory == null || !sender.Character.CanAccessInventory(item.PreviousParentInventory));
192 
193  if (itemAccessDenied)
194  {
195 #if DEBUG || UNSTABLE
196  DebugConsole.NewMessage($"Client {sender.Name} failed to pick up item \"{item}\" (parent inventory: {item.ParentInventory?.Owner.ToString() ?? "null"}). No access.", Color.Yellow);
197 #endif
198  if (item.body != null && !sender.PendingPositionUpdates.Contains(item))
199  {
200  sender.PendingPositionUpdates.Enqueue(item);
201  }
202  item.PositionUpdateInterval = 0.0f;
203  continue;
204  }
205  }

207  for (int j = 0; j < capacity; j++)

262  private void EnsureItemsInBothHands(Character character)
263  {
264  if (this is not CharacterInventory charInv) { return; }
265 
266  int leftHandSlot = charInv.FindLimbSlot(InvSlotType.LeftHand),
267  rightHandSlot = charInv.FindLimbSlot(InvSlotType.RightHand);
268 
269  if (IsSlotIndexOutOfBound(leftHandSlot) || IsSlotIndexOutOfBound(rightHandSlot)) { return; }
270 
271  TryPutInOppositeHandSlot(rightHandSlot, leftHandSlot);
272  TryPutInOppositeHandSlot(leftHandSlot, rightHandSlot);
273 
274  void TryPutInOppositeHandSlot(int originalSlot, int otherHandSlot)
275  {
276  const InvSlotType bothHandSlot = InvSlotType.LeftHand | InvSlotType.RightHand;
277 
278  foreach (Item it in slots[originalSlot].Items)
279  {
280  if (it.AllowedSlots.None(static s => s.HasFlag(bothHandSlot)) || slots[otherHandSlot].Contains(it)) { continue; }
281 
282  TryPutItem(it, otherHandSlot, true, true, character, false);
283  }
284  }
285 
286  bool IsSlotIndexOutOfBound(int index) => index < 0 || index >= slots.Length;
287  }
