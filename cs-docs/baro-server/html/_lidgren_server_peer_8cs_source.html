98  public override void Update(float deltaTime)
99  {
100  if (netServer is null) { return; }
101 
102  ToolBox.ThrowIfNull(incomingLidgrenMessages);
103 
104  netServer.ReadMessages(incomingLidgrenMessages);
105 
106  //process incoming connections first
107  foreach (NetIncomingMessage inc in incomingLidgrenMessages.Where(m => m.MessageType == NetIncomingMessageType.ConnectionApproval))
108  {
109  HandleConnection(inc);
110  }
111 
112  try
113  {
114  //after processing connections, go ahead with the rest of the messages
115  foreach (NetIncomingMessage inc in incomingLidgrenMessages.Where(m => m.MessageType != NetIncomingMessageType.ConnectionApproval))
116  {
117  switch (inc.MessageType)
118  {
119  case NetIncomingMessageType.Data:
120  HandleDataMessage(inc);
121  break;
122  case NetIncomingMessageType.StatusChanged:
123  HandleStatusChanged(inc);
124  break;
125  }
126  }
127  }
128 
129  catch (Exception e)
130  {
131  string errorMsg = "Server failed to read an incoming message. {" + e + "}\n" + e.StackTrace.CleanupStackTrace();
132  GameAnalyticsManager.AddErrorEventOnce($"LidgrenServerPeer.Update:ClientReadException{e.TargetSite}", GameAnalyticsManager.ErrorSeverity.Error, errorMsg);
133 #if DEBUG
134  DebugConsole.ThrowError(errorMsg);
135 #else
136  if (GameSettings.CurrentConfig.VerboseLogging) { DebugConsole.ThrowError(errorMsg); }
137 #endif
138  }
139 
140  for (int i = 0; i < pendingClients.Count; i++)
141  {
142  PendingClient pendingClient = pendingClients[i];
143 
144  LidgrenConnection connection = (LidgrenConnection)pendingClient.Connection;
145 
146  if (connection.NetConnection.Status == NetConnectionStatus.InitiatedConnect ||
147  connection.NetConnection.Status == NetConnectionStatus.ReceivedInitiation ||
148  connection.NetConnection.Status == NetConnectionStatus.RespondedAwaitingApproval ||
149  connection.NetConnection.Status == NetConnectionStatus.RespondedConnect)
150  {
151  continue;
152  }
153 
154  UpdatePendingClient(pendingClient);
155  if (i >= pendingClients.Count || pendingClients[i] != pendingClient) { i--; }
156  }
157 
158  incomingLidgrenMessages.Clear();
159  }

185  private void HandleConnection(NetIncomingMessage inc)
186  {
187  if (netServer == null) { return; }
188 
189  var skipDeny = false;
190  {
191  var result = GameMain.LuaCs.Hook.Call<bool?>("lidgren.handleConnection", inc);
192  if (result != null) {
193  if (result.Value) skipDeny = true;
194  else return;
195  }
196  }
197 
198  if (!skipDeny && connectedClients.Count >= serverSettings.MaxPlayers)
199  {
200  inc.SenderConnection.Deny(PeerDisconnectPacket.WithReason(DisconnectReason.ServerFull).ToLidgrenStringRepresentation());
201  return;
202  }
203 
204  if (serverSettings.BanList.IsBanned(new LidgrenEndpoint(inc.SenderConnection.RemoteEndPoint), out string banReason))
205  {
206  //IP banned: deny immediately
207  inc.SenderConnection.Deny(PeerDisconnectPacket.Banned(banReason).ToLidgrenStringRepresentation());
208  return;
209  }
210 
211  PendingClient? pendingClient = pendingClients.Find(c => c.Connection is LidgrenConnection l && l.NetConnection == inc.SenderConnection);
212 
213  if (pendingClient is null)
214  {
215  pendingClient = new PendingClient(new LidgrenConnection(inc.SenderConnection));
216  pendingClients.Add(pendingClient);
217  }
218 
219  inc.SenderConnection.Approve();
220  }

222  private void HandleDataMessage(NetIncomingMessage lidgrenMsg)
223  {
224  if (netServer == null) { return; }
225 
226  PendingClient? pendingClient = pendingClients.Find(c => c.Connection is LidgrenConnection l && l.NetConnection == lidgrenMsg.SenderConnection);
227 
228  IReadMessage inc = lidgrenMsg.ToReadMessage();
229 
230  var (_, packetHeader, initialization) = INetSerializableStruct.Read<PeerPacketHeaders>(inc);
231 
232  if (packetHeader.IsConnectionInitializationStep() && pendingClient != null && initialization.HasValue)
233  {
234  ReadConnectionInitializationStep(pendingClient, inc, initialization.Value);
235  }
236  else if (!packetHeader.IsConnectionInitializationStep())
237  {
238  if (connectedClients.Find(c => c is LidgrenConnection l && l.NetConnection == lidgrenMsg.SenderConnection) is not LidgrenConnection conn)
239  {
240  if (pendingClient != null)
241  {
242  RemovePendingClient(pendingClient, PeerDisconnectPacket.WithReason(DisconnectReason.AuthenticationRequired));
243  }
244  else if (lidgrenMsg.SenderConnection.Status != NetConnectionStatus.Disconnected &&
245  lidgrenMsg.SenderConnection.Status != NetConnectionStatus.Disconnecting)
246  {
247  lidgrenMsg.SenderConnection.Disconnect(PeerDisconnectPacket.WithReason(DisconnectReason.AuthenticationRequired).ToLidgrenStringRepresentation());
248  }
249 
250  return;
251  }
252 
253  if (pendingClient != null) { pendingClients.Remove(pendingClient); }
254 
255  if (serverSettings.BanList.IsBanned(conn.Endpoint, out string banReason)
256  || (conn.AccountInfo.AccountId.TryUnwrap(out var accountId) && serverSettings.BanList.IsBanned(accountId, out banReason))
257  || conn.AccountInfo.OtherMatchingIds.Any(id => serverSettings.BanList.IsBanned(id, out banReason)))
258  {
259  Disconnect(conn, PeerDisconnectPacket.Banned(banReason));
260  return;
261  }
262 
263  var packet = INetSerializableStruct.Read<PeerPacketMessage>(inc);
264  callbacks.OnMessageReceived.Invoke(conn, packet.GetReadMessage(packetHeader.IsCompressed(), conn));
265  }
266  }

268  private void HandleStatusChanged(NetIncomingMessage inc)
269  {
270  if (netServer == null) { return; }
271 
272  switch (inc.SenderConnection.Status)
273  {
274  case NetConnectionStatus.Disconnected:
275  LidgrenConnection? conn = connectedClients.Cast<LidgrenConnection>().FirstOrDefault(c => c.NetConnection == inc.SenderConnection);
276  if (conn != null)
277  {
278  if (conn == OwnerConnection)
279  {
280  DebugConsole.NewMessage("Owner disconnected: closing the server...");
281  GameServer.Log("Owner disconnected: closing the server...", ServerLog.MessageType.ServerMessage);
282  Close();
283  }
284  else
285  {
286  Disconnect(conn, PeerDisconnectPacket.WithReason(DisconnectReason.Disconnected));
287  }
288  }
289  else
290  {
291  PendingClient? pendingClient = pendingClients.Find(c => c.Connection is LidgrenConnection l && l.NetConnection == inc.SenderConnection);
292  if (pendingClient != null)
293  {
294  RemovePendingClient(pendingClient, PeerDisconnectPacket.WithReason(DisconnectReason.Disconnected));
295  }
296  }
297 
298  break;

307  private void OnAuthChange(Steamworks.SteamId steamId, Steamworks.SteamId ownerId, Steamworks.AuthResponse status)
308  {
309  if (netServer == null) { return; }
310 
311  PendingClient? pendingClient = pendingClients.Find(c => c.AccountInfo.AccountId is Some<AccountId> { Value: SteamId id } && id.Value == steamId);
312  DebugConsole.Log($"{steamId} validation: {status}, {(pendingClient != null)}");
313 
314  if (pendingClient is null)
315  {
316  if (status == Steamworks.AuthResponse.OK) { return; }
317 
318  if (connectedClients.Find(c
319  => c.AccountInfo.AccountId is Some<AccountId> { Value: SteamId id } && id.Value == steamId)
320  is LidgrenConnection connection)
321  {
322  Disconnect(connection, PeerDisconnectPacket.SteamAuthError(status));
323  }
324 
325  return;
326  }
327 
328  LidgrenConnection pendingConnection = (LidgrenConnection)pendingClient.Connection;
329  if (serverSettings.BanList.IsBanned(pendingConnection.Endpoint, out string banReason)
330  || serverSettings.BanList.IsBanned(new SteamId(steamId), out banReason)
331  || serverSettings.BanList.IsBanned(new SteamId(ownerId), out banReason))
332  {
333  RemovePendingClient(pendingClient, PeerDisconnectPacket.Banned(banReason));
334  return;
335  }
336 
337  if (status == Steamworks.AuthResponse.OK)
338  {
339  pendingClient.Connection.SetAccountInfo(new AccountInfo(new SteamId(steamId), new SteamId(ownerId)));
340  pendingClient.InitializationStep = serverSettings.HasPassword ? ConnectionInitialization.Password : ConnectionInitialization.ContentPackageOrder;
341  pendingClient.UpdateTime = Timing.TotalTime;
342  }
343  else
344  {
345  RemovePendingClient(pendingClient, PeerDisconnectPacket.SteamAuthError(status));
346  }

425  protected override void ProcessAuthTicket(ClientSteamTicketAndVersionPacket packet, PendingClient pendingClient)
426  {
427  if (pendingClient.AccountInfo.AccountId.IsNone())
428  {
429  bool requireSteamAuth = GameSettings.CurrentConfig.RequireSteamAuthentication;
430 #if DEBUG
431  requireSteamAuth = false;
432 #endif
433  bool hasSteamAuth = packet.SteamAuthTicket.TryUnwrap(out var ticket);
434 
435  //steam auth cannot be done (SteamManager not initialized or no ticket given),
436  //but it's not required either -> let the client join without auth
437  if ((!SteamManager.IsInitialized || !hasSteamAuth) && !requireSteamAuth)
438  {
439  pendingClient.Name = packet.Name;
440  pendingClient.OwnerKey = packet.OwnerKey;
441  pendingClient.InitializationStep = serverSettings.HasPassword ? ConnectionInitialization.Password : ConnectionInitialization.ContentPackageOrder;
442  }
443  else
444  {
445  if (!packet.SteamId.TryUnwrap(out var id) || id is not SteamId steamId)
446  {
447  if (requireSteamAuth)
448  {
449  RemovePendingClient(pendingClient, PeerDisconnectPacket.WithReason(DisconnectReason.SteamAuthenticationFailed));
450  return;
451  }
452  }
453  else
454  {
455  Steamworks.BeginAuthResult authSessionStartState = SteamManager.StartAuthSession(ticket, steamId);
456  if (authSessionStartState != Steamworks.BeginAuthResult.OK)
457  {
458  if (requireSteamAuth)
459  {
460  RemovePendingClient(pendingClient, PeerDisconnectPacket.SteamAuthError(authSessionStartState));
461  }
462  else
463  {
464  packet.SteamId = Option<AccountId>.None();
465  pendingClient.InitializationStep = serverSettings.HasPassword ? ConnectionInitialization.Password : ConnectionInitialization.ContentPackageOrder;
466  }
467  }
468  }
469 
470  pendingClient.Connection.SetAccountInfo(new AccountInfo(packet.SteamId.Select(uid => (AccountId)uid)));
471  pendingClient.Name = packet.Name;
472  pendingClient.OwnerKey = packet.OwnerKey;
473  pendingClient.AuthSessionStarted = true;
474  }
475  }
476  else
477  {
478  if (pendingClient.AccountInfo.AccountId != packet.SteamId.Select(uid => (AccountId)uid))
479  {
480  RemovePendingClient(pendingClient, PeerDisconnectPacket.WithReason(DisconnectReason.SteamAuthenticationFailed));
481  }
482  }
