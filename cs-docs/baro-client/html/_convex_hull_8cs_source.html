207  private void MergeLosVertices(ConvexHull ch, bool refreshOtherOverlappingHulls = true)
208  {
209  if (ch == this) { return; }
210 
211  //merge dist in the direction parallel to the segment
212  //(e.g. how far up/down we can stretch a vertical segment)
213  float mergeDistParallel = MathHelper.Clamp(ch.thickness * 0.65f, 16, 512);
214  if (MaxMergeLosVerticesDist.HasValue)
215  {
216  mergeDistParallel = Math.Max(mergeDistParallel, MaxMergeLosVerticesDist.Value);
217  }
218  else
219  {
220  Rectangle inflatedAABB = ch.BoundingBox;
221  inflatedAABB.Inflate(2, 2);
222  //if this los segment isn't touching the other's bounding box,
223  //don't extend the segment by more than 50% of it's length
224  if (!inflatedAABB.Contains(losVertices[0].Pos) &&
225  !inflatedAABB.Contains(losVertices[1].Pos))
226  {
227  mergeDistParallel = Math.Min(mergeDistParallel, Vector2.Distance(losVertices[0].Pos, losVertices[1].Pos) * 0.5f);
228  }
229  }
230  //merge dist in the direction perpendicular to the segment
231  //(e.g. how far right/left we can stretch a vertical segment)
232  //do not allow more than ~half of the thickness, because that'd make the segment go outside the convex hull
233  float mergeDistPerpendicular = Math.Min(mergeDistParallel, thickness * 0.35f);
234 
235  Vector2 center = (losVertices[0].Pos + losVertices[1].Pos) / 2;
236 
237  bool changed = false;
238  for (int i = 0; i < losVertices.Length; i++)
239  {
240  Vector2 segmentDir = Vector2.Normalize(losVertices[i].Pos - center);
241  //check if the closest point on the other convex hull segment is close enough, disregarding any offsets
242  //otherwise we might end up moving the vertex too much if we stretch it to an already-offset segment
243  if (!isCloseEnough(
244  MathUtils.GetClosestPointOnLineSegment(ch.losVertices[0].Pos, ch.losVertices[1].Pos, losVertices[i].Pos),
245  losVertices[i].Pos))
246  {
247  continue;
248  }
249 
250  //check the offset position of the segment next
251  Vector2 closest = MathUtils.GetClosestPointOnLineSegment(
252  ch.losVertices[0].Pos + ch.losOffsets[0],
253  ch.losVertices[1].Pos + ch.losOffsets[1],
254  losVertices[i].Pos);
255  if (!isCloseEnough(closest, losVertices[i].Pos)) { continue; }
256 
257  //find where the segments would intersect if they had infinite length
258  // if it's close to the closest point, let's use that instead to keep
259  // the direction of the segment unchanged (i.e. vertical segment stays vertical)
260  if (MathUtils.GetLineIntersection(
261  ch.losVertices[0].Pos + ch.losOffsets[0], ch.losVertices[1].Pos + ch.losOffsets[1],
262  losVertices[0].Pos, losVertices[1].Pos,
263  areLinesInfinite: true, out Vector2 intersection) &&
264  //the intersection needs to be outwards from the vertex we're checking
265  Vector2.Dot(segmentDir, intersection - losVertices[i].Pos) > 0 &&
266  //the intersection needs to be close enough to the default position of the vertex and the closest point
267  //(we don't want to merge the segments somewhere close to infinity!)
268  (Vector2.DistanceSquared(intersection, losVertices[i].Pos) < mergeDistParallel * mergeDistParallel ||
269  Vector2.DistanceSquared(intersection, closest) < 16.0f * 16.0f))
270  {
271  closest = intersection;
272  }
273 
274  //don't move the vertices of the segment too close to each other
275  if (Vector2.DistanceSquared(losVertices[1 - i].Pos + losOffsets[1 - i], closest) < mergeDistPerpendicular * mergeDistPerpendicular)
276  {
277  continue;
278  }
279 
280  losOffsets[i] = closest - losVertices[i].Pos;
281  overlappingHulls.Add(ch);
282  ch.overlappingHulls.Add(this);
283  changed = true;
284 
285  bool isCloseEnough(Vector2 closest, Vector2 vertex)
286  {
287  float dist = Vector2.Distance(closest, vertex);
288  if (dist < 0.001f) { return true; }
289  if (dist > mergeDistParallel) { return false; }
290 
291  Vector2 closestDir = (closest - vertex) / dist;
292 
293  float dot = Math.Abs(Vector2.Dot(segmentDir, closestDir));
294  float distAlongAxis = dist * dot;
295  if (distAlongAxis > mergeDistParallel) { return false; }
296 
297  float distPerpendicular = dist * (1.0f - dot);
298  if (distPerpendicular > mergeDistPerpendicular) { return false; }
299 
300  return true;
301  }
302  }
303  if (changed && refreshOtherOverlappingHulls)
304  {
305  foreach (var overlapping in overlappingHulls)
306  {
307  overlapping.MergeLosVertices(this, refreshOtherOverlappingHulls: false);
308  }
309  }
310  }

570  private static bool IsSegmentFacing(Vector2 segmentPos1, Vector2 segmentPos2, Vector2 viewPosition)
571  {
572  Vector2 segmentMid = (segmentPos1 + segmentPos2) / 2;
573  Vector2 segmentDiff = segmentPos2 - segmentPos1;
574  Vector2 segmentNormal = new Vector2(-segmentDiff.Y, segmentDiff.X);
575 
576  Vector2 viewDirection = viewPosition - segmentMid;
577  return Vector2.Dot(segmentNormal, viewDirection) > 0;

690  public static List<ConvexHull> GetHullsInRange(Vector2 position, float range, Submarine ParentSub)
691  {
692  List<ConvexHull> list = new List<ConvexHull>();
693 
694  foreach (ConvexHullList chList in HullLists)
695  {
696  Vector2 lightPos = position;
697  if (ParentSub == null)
698  {
699  //light and the convexhull are both outside
700  if (chList.Submarine == null)
701  {
702  list.AddRange(chList.List.FindAll(ch => MathUtils.CircleIntersectsRectangle(lightPos, range, ch.BoundingBox)));
703  }
704  //light is outside, convexhull inside a sub
705  else
706  {
707  Rectangle subBorders = chList.Submarine.Borders;
708  subBorders.Y -= chList.Submarine.Borders.Height;
709  if (!MathUtils.CircleIntersectsRectangle(lightPos - chList.Submarine.WorldPosition, range, subBorders)) { continue; }
710 
711  lightPos -= chList.Submarine.WorldPosition - chList.Submarine.HiddenSubPosition;
712 
713  list.AddRange(chList.List.FindAll(ch => MathUtils.CircleIntersectsRectangle(lightPos, range, ch.BoundingBox)));
714  }
715  }
716  else
717  {
718  //light is inside, convexhull outside
719  if (chList.Submarine == null)
720  {
721  continue;
722  }
723  //light and convexhull are both inside the same sub
724  if (chList.Submarine == ParentSub)
725  {
726  list.AddRange(chList.List.FindAll(ch => MathUtils.CircleIntersectsRectangle(lightPos, range, ch.BoundingBox)));
727  }
728  //light and convexhull are inside different subs
729  else
730  {
731  lightPos -= (chList.Submarine.Position - ParentSub.Position);
732 
733  Rectangle subBorders = chList.Submarine.Borders;
734  subBorders.Location += chList.Submarine.HiddenSubPosition.ToPoint() - new Point(0, chList.Submarine.Borders.Height);
735 
736  if (!MathUtils.CircleIntersectsRectangle(lightPos, range, subBorders)) continue;
737 
738  list.AddRange(chList.List.FindAll(ch => MathUtils.CircleIntersectsRectangle(lightPos, range, ch.BoundingBox)));
739  }
740  }
741  }
742 
743  return list;
