25  public static void RefreshSelfAccountIds(Action? onRefreshComplete = null)
26  {
27  SelfAccountIds = ImmutableHashSet<AccountId>.Empty;
28  var selfPuids = EosInterface.IdQueries.GetLoggedInPuids();
29 
30  if (selfPuids.Length == 0)
31  {
32  onRefreshComplete?.Invoke();
33  return;
34  }
35 
36  var collectedIds = new Option<ImmutableArray<AccountId>>[selfPuids.Length];
37 
38  Action<Task> taskDoneHandler(int index)
39  {
40  void countDoneTask(Task t)
41  {
42  try
43  {
44  if (!t.TryGetResult(out Result<ImmutableArray<AccountId>, EosInterface.IdQueries.GetSelfExternalIdError>? result)) { return; }
45  if (!result.TryUnwrapSuccess(out var ids)) { return; }
46  collectedIds[index] = Option.Some(ids);
47  }
48  finally
49  {
50  // If we failed to get IDs from this query, fill in the relevant slot in the collectedIds array
51  // to indicate that the task is done anyway
52  collectedIds[index] = Option.Some(collectedIds[index].Fallback(ImmutableArray<AccountId>.Empty));
53 
54  // If all of the tasks are done, merge all of the collected IDs into the hashset
55  if (collectedIds.All(o => o.IsSome()))
56  {
57  SelfAccountIds = collectedIds.NotNone().SelectMany(a => a).ToImmutableHashSet();
58  onRefreshComplete?.Invoke();
59  }
60  }
61  }
62 
63  return countDoneTask;
64  }
65 
66  for (int i = 0; i < selfPuids.Length; i++)
67  {
68  TaskPool.Add($"SelfPlayerRowWithExternalAccountIds{i}",
69  EosInterface.IdQueries.GetSelfExternalAccountIds(selfPuids[i]),
70  taskDoneHandler(i));
