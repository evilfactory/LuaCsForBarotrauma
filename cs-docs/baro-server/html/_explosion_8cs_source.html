271  {
272  Hull hull = Hull.FindHull(worldPosition);
273  ExplodeProjSpecific(worldPosition, hull);
274 
275  if (hull != null && !string.IsNullOrWhiteSpace(decal) && decalSize > 0.0f)
276  {
277  hull.AddDecal(decal, worldPosition, decalSize, isNetworkEvent: false);
278  }
279 
280  Attack.DamageMultiplier = 1.0f;
281  float displayRange = Attack.Range;
282  if (damageSource is Item sourceItem)
283  {
284  var launcher = sourceItem.GetComponent<Projectile>()?.Launcher;
285  displayRange *=
286  1.0f
287  + sourceItem.GetQualityModifier(Quality.StatType.ExplosionRadius)
288  + (launcher?.GetQualityModifier(Quality.StatType.ExplosionRadius) ?? 0);
289  Attack.DamageMultiplier *=
290  1.0f
291  + sourceItem.GetQualityModifier(Quality.StatType.ExplosionDamage)
292  + (launcher?.GetQualityModifier(Quality.StatType.ExplosionDamage) ?? 0);
293  Attack.SourceItem ??= sourceItem;
294  }
295 
296  if (attacker is not null)
297  {
298  displayRange *= 1f + attacker.GetStatValue(StatTypes.ExplosionRadiusMultiplier);
299  Attack.DamageMultiplier *= 1f + attacker.GetStatValue(StatTypes.ExplosionDamageMultiplier);
300  }
301 
302  Vector2 cameraPos = GameMain.GameScreen.Cam.Position;
303  float cameraDist = Vector2.Distance(cameraPos, worldPosition) / 2.0f;
304  GameMain.GameScreen.Cam.Shake = cameraShake * Math.Max((cameraShakeRange - cameraDist) / cameraShakeRange, 0.0f);
305 #if CLIENT
306  if (screenColor != Color.Transparent)
307  {
308  Color flashColor = Color.Lerp(Color.Transparent, screenColor, Math.Max((screenColorRange - cameraDist) / screenColorRange, 0.0f));
309  Screen.Selected.ColorFade(flashColor, Color.Transparent, screenColorDuration);
310  }
311  foreach (Item item in Item.ItemList)
312  {
313  item.GetComponent<Sonar>()?.RegisterExplosion(this, worldPosition);
314  }
315 #endif
316 
317  if (displayRange < 0.1f) { return; }
318 
319  if (!MathUtils.NearlyEqual(Attack.GetStructureDamage(1.0f), 0.0f) || !MathUtils.NearlyEqual(Attack.GetLevelWallDamage(1.0f), 0.0f))
320  {
321  RangedStructureDamage(worldPosition, displayRange, Attack.GetStructureDamage(1.0f), Attack.GetLevelWallDamage(1.0f), attacker,
322  IgnoredSubmarines,
323  Attack.EmitStructureDamageParticles,
324  DistanceFalloff);
325  }
326 
327  if (BallastFloraDamage > 0.0f)
328  {
329  RangedBallastFloraDamage(worldPosition, displayRange, BallastFloraDamage, attacker, DistanceFalloff);
330  }
331 
332  if (EmpStrength > 0.0f)
333  {
334  float displayRangeSqr = displayRange * displayRange;
335  foreach (Item item in Item.ItemList)
336  {
337  float distSqr = Vector2.DistanceSquared(item.WorldPosition, worldPosition);
338  if (distSqr > displayRangeSqr) { continue; }
339  float distFactor = DistanceFalloff ? CalculateDistanceFactor(distSqr, displayRange) : 1.0f;
340 
341  //damage repairable power-consuming items
342  var powered = item.GetComponent<Powered>();
343  if (powered == null || !powered.VulnerableToEMP) { continue; }
344  if (item.Repairables.Any())
345  {
346  item.Condition -= item.MaxCondition * EmpStrength * distFactor;
347  }
348 
349  var lightComponent = item.GetComponent<LightComponent>();
350  if (lightComponent != null)
351  {
352  //multiply by 10 to make the effect more noticeable
353  //(a strength of 1 is already enough to kill power and shut down the lights, but we want weaker EMPs to make the lights flicker noticeably)
354  lightComponent.TemporaryFlickerTimer = Math.Min(EmpStrength * distFactor * 10.0f, 10.0f);
355  }
356 
357  //discharge batteries
358  var powerContainer = item.GetComponent<PowerContainer>();
359  if (powerContainer != null)
360  {
361  powerContainer.Charge -= powerContainer.GetCapacity() * EmpStrength * distFactor;
362  }
363  }
364  static float CalculateDistanceFactor(float distSqr, float displayRange) => 1.0f - MathF.Sqrt(distSqr) / displayRange;
365  }
366 
367  if (itemRepairStrength > 0.0f)
368  {
369  float displayRangeSqr = displayRange * displayRange;
370  foreach (Item item in Item.ItemList)
371  {
372  float distSqr = Vector2.DistanceSquared(item.WorldPosition, worldPosition);
373  if (distSqr > displayRangeSqr) { continue; }
374 
375  float distFactor =
376  DistanceFalloff ?
377  1.0f - (float)Math.Sqrt(distSqr) / displayRange :
378  1.0f;
379  //repair repairable items
380  if (item.Repairables.Any())
381  {
382  item.Condition += itemRepairStrength * distFactor;
383  }
384  }
385  }
386 
387  if (MathUtils.NearlyEqual(force, 0.0f) && MathUtils.NearlyEqual(Attack.Stun, 0.0f) && Attack.Afflictions.None())
388  {
389  return;
390  }
391 
392  DamageCharacters(worldPosition, Attack, force, damageSource, attacker);
393 
394  if (GameMain.NetworkMember == null || !GameMain.NetworkMember.IsClient)
395  {
396  foreach (Item item in Item.ItemList)
397  {
398  if (item.Condition <= 0.0f) { continue; }
399  float dist = Vector2.Distance(item.WorldPosition, worldPosition);
400  float itemRadius = item.body == null ? 0.0f : item.body.GetMaxExtent();
401  dist = Math.Max(0.0f, dist - ConvertUnits.ToDisplayUnits(itemRadius));
402  if (dist > displayRange) { continue; }
403 
404  if (dist < displayRange * 0.5f && applyFireEffects && !item.FireProof && ignoreFireEffectsForTags.None(t => item.HasTag(t)))
405  {
406  //don't apply OnFire effects if the item is inside a fireproof container
407  //(or if it's inside a container that's inside a fireproof container, etc)
408  Item container = item.Container;
409  bool fireProof = false;
410  while (container != null)
411  {
412  if (container.FireProof)
413  {
414  fireProof = true;
415  break;
416  }
417  container = container.Container;
418  }
419  if (!fireProof)
420  {
421  item.ApplyStatusEffects(ActionType.OnFire, 1.0f);
422  if (item.Condition <= 0.0f && GameMain.NetworkMember is { IsServer: true })
423  {
424  GameMain.NetworkMember.CreateEntityEvent(item, new Item.ApplyStatusEffectEventData(ActionType.OnFire));
425  }
426  }
427  }
428 
429  if (item.Prefab.DamagedByExplosions && !item.Indestructible)
430  {
431  float distFactor =
432  DistanceFalloff ?
433  1.0f - dist / displayRange :
434  1.0f;
435  float damageAmount = Attack.GetItemDamage(1.0f, item.Prefab.ExplosionDamageMultiplier);
436 
437  Vector2 explosionPos = worldPosition;
438  if (item.Submarine != null) { explosionPos -= item.Submarine.Position; }
439 
440  damageAmount *= GetObstacleDamageMultiplier(ConvertUnits.ToSimUnits(explosionPos), worldPosition, item.SimPosition, IgnoredCover);
441  item.Condition -= damageAmount * distFactor;
442  }
443  }
444  }

449  private void DamageCharacters(Vector2 worldPosition, Attack attack, float force, Entity damageSource, Character attacker)
450  {
451  if (attack.Range <= 0.0f) { return; }
452 
453  //long range for the broad distance check, because large characters may still be in range even if their collider isn't
454  float broadRange = Math.Max(attack.Range * 10.0f, 10000.0f);
455 
456  foreach (Character c in Character.CharacterList)
457  {
458  if (IgnoredCharacters.Contains(c)) { continue; }
459 
460  if (!c.Enabled ||
461  Math.Abs(c.WorldPosition.X - worldPosition.X) > broadRange ||
462  Math.Abs(c.WorldPosition.Y - worldPosition.Y) > broadRange)
463  {
464  continue;
465  }
466 
467  if (OnlyInside && c.Submarine == null)
468  {
469  continue;
470  }
471  else if (OnlyOutside && c.Submarine != null)
472  {
473  continue;
474  }
475 
476  Vector2 explosionPos = worldPosition;
477  if (c.Submarine != null) { explosionPos -= c.Submarine.Position; }
478 
479  Hull hull = Hull.FindHull(explosionPos, null, false);
480  bool underWater = hull == null || explosionPos.Y < hull.Surface;
481 
482  explosionPos = ConvertUnits.ToSimUnits(explosionPos);
483 
484  Dictionary<Limb, float> distFactors = new Dictionary<Limb, float>();
485  Dictionary<Limb, float> damages = new Dictionary<Limb, float>();
486  List<Affliction> modifiedAfflictions = new List<Affliction>();
487 
488  foreach (Limb limb in c.AnimController.Limbs)
489  {
490  if (limb.IsSevered || limb.IgnoreCollisions || !limb.body.Enabled) { continue; }
491 
492  float dist = Vector2.Distance(limb.WorldPosition, worldPosition);
493 
494  //calculate distance from the "outer surface" of the physics body
495  //doesn't take the rotation of the limb into account, but should be accurate enough for this purpose
496  float limbRadius = limb.body.GetMaxExtent();
497  dist = Math.Max(0.0f, dist - ConvertUnits.ToDisplayUnits(limbRadius));
498 
499  if (dist > attack.Range) { continue; }
500 
501  float distFactor =
502  DistanceFalloff ?
503  1.0f - dist / attack.Range :
504  1.0f;
505 
506  //solid obstacles between the explosion and the limb reduce the effect of the explosion
507  if (!IgnoreCover)
508  {
509  distFactor *= GetObstacleDamageMultiplier(explosionPos, worldPosition, limb.SimPosition, IgnoredCover);
510  }
511  if (distFactor > 0)
512  {
513  distFactors.Add(limb, distFactor);
514  }
515  }
516 
517  foreach (Limb limb in distFactors.Keys)
518  {
519  if (!distFactors.TryGetValue(limb, out float distFactor)) { continue; }
520  modifiedAfflictions.Clear();
521  foreach (Affliction affliction in attack.Afflictions.Keys)
522  {
523  float dmgMultiplier = distFactor;
524  if (affliction.DivideByLimbCount)
525  {
526  float limbCountFactor = distFactors.Count;
527  if (affliction.Prefab.LimbSpecific && affliction.Prefab.AfflictionType == AfflictionPrefab.DamageType)
528  {
529  // Shouldn't go above 15, or the damage can be unexpectedly low -> doesn't break armor
530  // Effectively this makes large explosions more effective against large creatures (because more limbs are affected), but I don't think that's necessarily a bad thing.
531  limbCountFactor = Math.Min(distFactors.Count, 15);
532  }
533  dmgMultiplier /= limbCountFactor;
534  }
535  modifiedAfflictions.Add(affliction.CreateMultiplied(dmgMultiplier, affliction));
536  }
537  c.LastDamageSource = damageSource;
538  if (attacker == null)
539  {
540  if (damageSource is Item item)
541  {
542  attacker = item.GetComponent<Projectile>()?.User;
543  attacker ??= item.GetComponent<MeleeWeapon>()?.User;
544  }
545  }
546 
547  if (attack.Afflictions.Any() || attack.Stun > 0.0f)
548  {
549  if (!attack.OnlyHumans || c.IsHuman)
550  {
551  AbilityAttackData attackData = new AbilityAttackData(Attack, c, attacker);
552  if (attackData.Afflictions != null)
553  {
554  modifiedAfflictions.AddRange(attackData.Afflictions);
555  }
556 
557  //use a position slightly from the limb's position towards the explosion
558  //ensures that the attack hits the correct limb and that the direction of the hit can be determined correctly in the AddDamage methods
559  Vector2 dir = worldPosition - limb.WorldPosition;
560  Vector2 hitPos = limb.WorldPosition + (dir.LengthSquared() <= 0.001f ? Rand.Vector(1.0f) : Vector2.Normalize(dir)) * 0.01f;
561  AttackResult attackResult = c.AddDamage(hitPos, modifiedAfflictions, attack.Stun * distFactor, false, attacker: attacker, damageMultiplier: attack.DamageMultiplier * attackData.DamageMultiplier);
562  damages.Add(limb, attackResult.Damage);
563  }
564  }
565 
566  if (attack.StatusEffects != null && attack.StatusEffects.Any())
567  {
568  attack.SetUser(attacker);
569  var statusEffectTargets = new List<ISerializableEntity>();
570  foreach (StatusEffect statusEffect in attack.StatusEffects)
571  {
572  statusEffectTargets.Clear();
573  if (statusEffect.HasTargetType(StatusEffect.TargetType.Character)) { statusEffectTargets.Add(c); }
574  if (statusEffect.HasTargetType(StatusEffect.TargetType.Limb)) { statusEffectTargets.Add(limb); }
575  statusEffect.Apply(ActionType.OnUse, 1.0f, damageSource, statusEffectTargets);
576  statusEffect.Apply(ActionType.Always, 1.0f, damageSource, statusEffectTargets);
577  statusEffect.Apply(underWater ? ActionType.InWater : ActionType.NotInWater, 1.0f, damageSource, statusEffectTargets);
578  }
579  }
580 
581  if (limb.WorldPosition != worldPosition && !MathUtils.NearlyEqual(force, 0.0f))
582  {
583  Vector2 limbDiff = Vector2.Normalize(limb.WorldPosition - worldPosition);
584  if (!MathUtils.IsValid(limbDiff)) { limbDiff = Rand.Vector(1.0f); }
585  Vector2 impulse = limbDiff * distFactor * force;
586  Vector2 impulsePoint = limb.SimPosition - limbDiff * limb.body.GetMaxExtent();
587  limb.body.ApplyLinearImpulse(impulse, impulsePoint, maxVelocity: NetConfig.MaxPhysicsBodyVelocity * 0.2f);
588  }
589  }
590 
591  if (c == Character.Controlled && !c.IsDead && playTinnitus)
592  {
593  Limb head = c.AnimController.GetLimb(LimbType.Head);
594  if (head != null && damages.TryGetValue(head, out float headDamage) && headDamage > 0.0f && distFactors.TryGetValue(head, out float headFactor))
595  {
596  PlayTinnitusProjSpecific(headFactor);
597  }
598  }
599 
600  //sever joints
601  if (attack.SeverLimbsProbability > 0.0f)
602  {
603  foreach (Limb limb in c.AnimController.Limbs)
604  {
605  if (limb.character.Removed || limb.Removed) { continue; }
606  if (limb.IsSevered) { continue; }
607  if (!c.IsDead && !limb.CanBeSeveredAlive) { continue; }
608  if (distFactors.TryGetValue(limb, out float distFactor))
609  {
610  if (damages.TryGetValue(limb, out float damage))
611  {
612  c.TrySeverLimbJoints(limb, attack.SeverLimbsProbability * distFactor, damage, allowBeheading: true, attacker: attacker);
613  }
614  }
615  }
616  }
617  }

620  private static readonly Dictionary<Structure, float> damagedStructures = new Dictionary<Structure, float>();
624  public static Dictionary<Structure, float> RangedStructureDamage(Vector2 worldPosition, float worldRange, float damage, float levelWallDamage, Character attacker = null, IEnumerable<Submarine> ignoredSubmarines = null,
625  bool emitWallDamageParticles = true, bool distanceFalloff = true)
626  {
627  float dist = 600.0f;
628  damagedStructures.Clear();
629  foreach (Structure structure in Structure.WallList)
630  {
631  if (ignoredSubmarines != null && structure.Submarine != null && ignoredSubmarines.Contains(structure.Submarine)) { continue; }
632 
633  if (structure.HasBody &&
634  !structure.IsPlatform &&
635  Vector2.Distance(structure.WorldPosition, worldPosition) < dist * 3.0f)
636  {
637  for (int i = 0; i < structure.SectionCount; i++)
638  {
639  float distFactor =
640  distanceFalloff ?
641  1.0f - (Vector2.Distance(structure.SectionPosition(i, true), worldPosition) / worldRange) :
642  1.0f;
643  if (distFactor <= 0.0f) { continue; }
644 
645  structure.AddDamage(i, damage * distFactor, attacker, emitParticles: emitWallDamageParticles);
646 
647  if (damagedStructures.ContainsKey(structure))
648  {
649  damagedStructures[structure] += damage * distFactor;
650  }
651  else
652  {
653  damagedStructures.Add(structure, damage * distFactor);
654  }
655  }
656  }
657  }
658 
659  if (Level.Loaded != null && !MathUtils.NearlyEqual(levelWallDamage, 0.0f))
660  {
661  if (Level.Loaded?.LevelObjectManager != null)
662  {
663  foreach (var levelObject in Level.Loaded.LevelObjectManager.GetAllObjects(worldPosition, worldRange))
664  {
665  if (levelObject.Prefab.TakeLevelWallDamage)
666  {
667  float distFactor = 1.0f - (Vector2.Distance(levelObject.WorldPosition, worldPosition) / worldRange);
668  if (distFactor <= 0.0f) { continue; }
669  levelObject.AddDamage(levelWallDamage * distFactor, 1.0f, null);
670  }
671  }
672  }
673 
674  for (int i = Level.Loaded.ExtraWalls.Count - 1; i >= 0; i--)
675  {
676  if (Level.Loaded.ExtraWalls[i] is not DestructibleLevelWall destructibleWall) { continue; }
677 
678  bool inRange = false;
679  foreach (var cell in destructibleWall.Cells)
680  {
681  if (cell.IsPointInside(worldPosition))
682  {
683  inRange = true;
684  break;
685  }
686  foreach (var edge in cell.Edges)
687  {
688  if (MathUtils.LineSegmentToPointDistanceSquared((edge.Point1 + cell.Translation).ToPoint(), (edge.Point2 + cell.Translation).ToPoint(), worldPosition.ToPoint()) < worldRange * worldRange)
689  {
690  inRange = true;
691  break;
692  }
693  }
694  if (inRange) { break; }
695  }
696  if (inRange)
697  {
698  destructibleWall.AddDamage(levelWallDamage, worldPosition);
699  }
700  }
701  }
702 
703  return damagedStructures;
704  }

706  public static void RangedBallastFloraDamage(Vector2 worldPosition, float worldRange, float damage, Character attacker = null, bool distanceFalloff = true)
707  {
708  List<BallastFloraBehavior> ballastFlorae = new List<BallastFloraBehavior>();
709 
710  foreach (Hull hull in Hull.HullList)
711  {
712  if (hull.BallastFlora != null) { ballastFlorae.Add(hull.BallastFlora); }
713  }
714 
715  foreach (BallastFloraBehavior ballastFlora in ballastFlorae)
716  {
717  float resistanceMuliplier = ballastFlora.HasBrokenThrough ? 1f : 1f - ballastFlora.ExplosionResistance;
718  ballastFlora.Branches.ForEachMod(branch =>
719  {
720  Vector2 branchWorldPos = ballastFlora.GetWorldPosition() + branch.Position;
721  float branchDist = Vector2.Distance(branchWorldPos, worldPosition);
722  if (branchDist < worldRange)
723  {
724  float distFactor =
725  distanceFalloff ?
726  1.0f - (branchDist / worldRange) :
727  1.0f;
728  if (distFactor <= 0.0f) { return; }
729 
730  Vector2 explosionPos = worldPosition;
731  Vector2 branchPos = branchWorldPos;
732  if (ballastFlora.Parent?.Submarine != null)
733  {
734  explosionPos -= ballastFlora.Parent.Submarine.Position;
735  branchPos -= ballastFlora.Parent.Submarine.Position;
736  }
737  distFactor *= GetObstacleDamageMultiplier(ConvertUnits.ToSimUnits(explosionPos), worldPosition, ConvertUnits.ToSimUnits(branchPos));
738  ballastFlora.DamageBranch(branch, damage * distFactor * resistanceMuliplier, BallastFloraBehavior.AttackType.Explosives, attacker);
739  }
740  });
741  }
742  }

744  private static float GetObstacleDamageMultiplier(Vector2 explosionSimPos, Vector2 explosionWorldPos, Vector2 targetSimPos, IEnumerable<Structure> ignoredCover = null)
745  {
746  float damageMultiplier = 1.0f;
