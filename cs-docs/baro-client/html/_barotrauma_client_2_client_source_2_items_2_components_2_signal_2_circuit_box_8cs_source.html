378  public void ClientEventRead(IReadMessage msg, float sendingTime)
379  {
380  var header = (CircuitBoxOpcode)msg.ReadByte();
381 
382  switch (header)
383  {
384  case CircuitBoxOpcode.AddComponent:
385  {
386  var data = INetSerializableStruct.Read<CircuitBoxServerCreateComponentEvent>(msg);
387  AddComponentFromData(data);
388  break;
389  }
390  case CircuitBoxOpcode.DeleteComponent:
391  {
392  var data = INetSerializableStruct.Read<CircuitBoxRemoveComponentEvent>(msg);
393  RemoveComponentInternal(data.TargetIDs);
394  break;
395  }
396  case CircuitBoxOpcode.MoveComponent:
397  {
398  var data = INetSerializableStruct.Read<CircuitBoxMoveComponentEvent>(msg);
399  MoveNodesInternal(data.TargetIDs, data.IOs, data.MoveAmount);
400  break;
401  }
402  case CircuitBoxOpcode.UpdateSelection:
403  {
404  var data = INetSerializableStruct.Read<CircuitBoxServerUpdateSelection>(msg);
405 
406  var nodeDict = data.ComponentIds.ToImmutableDictionary(static s => s.ID, static s => s.SelectedBy);
407  var wireDict = data.WireIds.ToImmutableDictionary(static s => s.ID, static s => s.SelectedBy);
408  var ioDict = data.InputOutputs.ToImmutableDictionary(static s => s.Type, static s => s.SelectedBy);
409 
410  UpdateSelections(nodeDict, wireDict, ioDict);
411  break;
412  }
413  case CircuitBoxOpcode.AddWire:
414  {
415  var data = INetSerializableStruct.Read<CircuitBoxServerCreateWireEvent>(msg);
416  AddWireFromData(data);
417  break;
418  }
419  case CircuitBoxOpcode.RemoveWire:
420  {
421  var data = INetSerializableStruct.Read<CircuitBoxRemoveWireEvent>(msg);
422  RemoveWireInternal(data.TargetIDs);
423  break;
424  }
425  case CircuitBoxOpcode.ServerInitialize:
426  {
427  Components.Clear();
428  Wires.Clear();
429 
430  var data = INetSerializableStruct.Read<CircuitBoxInitializeStateFromServerEvent>(msg);
431  foreach (var compData in data.Components) { AddComponentFromData(compData); }
432  foreach (var wireData in data.Wires) { AddWireFromData(wireData); }
433 
434  foreach (var node in InputOutputNodes)
435  {
436  node.Position = node.NodeType switch
437  {
438  CircuitBoxInputOutputNode.Type.Input => data.InputPos,
439  CircuitBoxInputOutputNode.Type.Output => data.OutputPos,
440  _ => node.Position
441  };
442  }
443  wasInitializedByServer = true;
444  break;
445  }
446  default:
447  throw new ArgumentOutOfRangeException(nameof(header), header, "This opcode cannot be handled using entity events");
448  }
449  }
