656  public Item FindClosestItem(List<Item> itemCollection, Vector2 simPosition, float aimAssistModifier = 0.0f)
657  {
658  if (Submarine != null)
659  {
660  simPosition += Submarine.SimPosition;
661  }
662 
663  debugInteractablesInRange.Clear();
664  debugInteractablesAtCursor.Clear();
665  debugInteractablesNearCursor.Clear();
666 
667  bool draggingItemToWorld = CharacterInventory.DraggingItemToWorld;
668 
669  //reduce the amount of aim assist if an item has been selected
670  //= can't switch selection to another item without deselecting the current one first UNLESS the cursor is directly on the item
671  //otherwise it would be too easy to accidentally switch the selected item when rewiring items
672  float aimAssistAmount = SelectedItem == null ? 100.0f * aimAssistModifier : 1.0f;
673 
674  Vector2 displayPosition = ConvertUnits.ToDisplayUnits(simPosition);
675 
676  Item closestItem = null;
677  float closestItemDistance = Math.Max(aimAssistAmount, 2.0f);
678  foreach (var item in itemCollection)
679  {
680  if (draggingItemToWorld)
681  {
682  if (item.OwnInventory == null ||
683  !item.OwnInventory.Container.AllowDragAndDrop ||
684  !item.OwnInventory.CanBePut(CharacterInventory.DraggingItems.First()) ||
685  !CanAccessInventory(item.OwnInventory))
686  {
687  continue;
688  }
689  }
690 
691  float distanceToItem = float.PositiveInfinity;
692  if (item.IsInsideTrigger(displayPosition, out Rectangle transformedTrigger))
693  {
694  debugInteractablesAtCursor.Add(item);
695  //distance is between 0-1 when the cursor is directly on the item
696  distanceToItem =
697  Math.Abs(transformedTrigger.Center.X - displayPosition.X) / transformedTrigger.Width +
698  Math.Abs((transformedTrigger.Y - transformedTrigger.Height / 2.0f) - displayPosition.Y) / transformedTrigger.Height;
699  //modify the distance based on the size of the trigger (preferring smaller items)
700  distanceToItem *= MathHelper.Lerp(0.05f, 2.0f, (transformedTrigger.Width + transformedTrigger.Height) / 250.0f);
701  }
702  else if (!item.Prefab.RequireCursorInsideTrigger)
703  {
704  Rectangle itemDisplayRect = new Rectangle(item.InteractionRect.X, item.InteractionRect.Y - item.InteractionRect.Height, item.InteractionRect.Width, item.InteractionRect.Height);
705 
706  if (itemDisplayRect.Contains(displayPosition))
707  {
708  debugInteractablesAtCursor.Add(item);
709  //distance is between 0-1 when the cursor is directly on the item
710  distanceToItem =
711  Math.Abs(itemDisplayRect.Center.X - displayPosition.X) / itemDisplayRect.Width +
712  Math.Abs(itemDisplayRect.Center.Y - displayPosition.Y) / itemDisplayRect.Height;
713  //modify the distance based on the size of the item (preferring smaller ones)
714  distanceToItem *= MathHelper.Lerp(0.05f, 2.0f, (itemDisplayRect.Width + itemDisplayRect.Height) / 250.0f);
715  }
716  else
717  {
718  if (closestItemDistance < 2.0f) { continue; }
719  //get the point on the itemDisplayRect which is closest to the cursor
720  Vector2 rectIntersectionPoint = new Vector2(
721  MathHelper.Clamp(displayPosition.X, itemDisplayRect.X, itemDisplayRect.Right),
722  MathHelper.Clamp(displayPosition.Y, itemDisplayRect.Y, itemDisplayRect.Bottom));
723  distanceToItem = 2.0f + Vector2.Distance(rectIntersectionPoint, displayPosition);
724  }
725  }
726 
727  if (distanceToItem > closestItemDistance) { continue; }
728  if (!CanInteractWith(item)) { continue; }
729 
730  debugInteractablesNearCursor.Add((item, 1.0f - distanceToItem / (100.0f * aimAssistModifier)));
731  closestItem = item;
